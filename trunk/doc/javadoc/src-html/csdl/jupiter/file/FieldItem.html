001    package csdl.jupiter.file;
002    
003    import java.util.List;
004    
005    /**
006     * Provides field item data structure.
007     * @author Takuya Yamashita
008     * @version $Id: FieldItem.java,v 1.2 2005/03/25 18:32:07 takuyay Exp $
009     */
010    public class FieldItem {
011      private String fieldItemId;
012      private String defaultKey;
013      private List entryNameList;
014      
015      /**
016       * Instantiates field item data structure.
017       * @param fieldItemId the field item id.
018       * @param defaultKey the default key.
019       * @param entryNameList the entry name list.
020       */
021      public FieldItem(String fieldItemId, String defaultKey, List entryNameList) {
022        this.fieldItemId = fieldItemId;
023        this.defaultKey = defaultKey;
024        this.entryNameList = entryNameList;
025      }
026      
027      /**
028       * Gets the default key of the field item.
029       * @return the default key of the field item.
030       */
031      public String getDefaultKey() {
032        return defaultKey;
033      }
034      
035      /**
036       * Sets the default key of the field item.
037       * @param defaultKey the default key.
038       */
039      public void setDefaultKey(String defaultKey) {
040        this.defaultKey = defaultKey;
041      }
042      
043      /**
044       * Gets the list of the <code>String</code> entry name.
045       * @return the list of the <code>String</code> entry name.
046       */
047      public List getEntryNameList() {
048        return entryNameList;
049      }
050      
051      /**
052       * gets the field item id.
053       * @return the field item id.
054       */
055      public String getFieldItemId() {
056        return fieldItemId;
057      }
058      
059      /**
060       * sets the field item id.
061       * @param fieldItemId the field item id.
062       */
063      public void setFieldItemId(String fieldItemId) {
064        this.fieldItemId = fieldItemId;
065      }
066    }
