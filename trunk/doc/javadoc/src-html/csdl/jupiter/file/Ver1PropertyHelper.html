056      /**
057       * Creates the new property config document in the <code>IProject</code>.
058       * @param project the project
059       * @return the new <code>Document</code> instance.
060       * @throws ReviewException if an error occurs during the new document creation.
061       */
062      public static Document newDocument(IProject project) throws ReviewException {
063        try {
064          SAXBuilder builder = new SAXBuilder();
065          IFile jupiterConfigIFile = project.getFile(".jupiter");
066          File jupiterConfigFile = jupiterConfigIFile.getLocation().toFile();
067          Document document = null;
068          if (jupiterConfigIFile.getLocation().toFile().exists()) {
069            document = builder.build(jupiterConfigFile);
070          }
071          else {
072            document = builder.build(copyDefaultConfigFileTo(jupiterConfigFile));
073            jupiterConfigIFile.refreshLocal(IResource.DEPTH_ONE, null);
074          }
075          return document;
076        }
077        catch (CoreException e) {
078          throw new ReviewException("CoreException: " + e.getMessage());
079        }
080        catch (JDOMException e) {
081          throw new ReviewException("JDOMException: " + e.getMessage());
082        }
083        catch (IOException e) {
084          throw new ReviewException("IOException: " + e.getMessage());      
085        }
086      }

088      /**
089       * Serializes <code>Document</code> to the jupiter config file.
090       * @param document the document.
091       * @param project the project.
092       * @throws ReviewException if an error occurs during serialization.
093       */
094      public static void serializeDocument(Document document, IProject project)
095                                           throws ReviewException {
096        try {
097          IFile outputPropertyIFile = project.getFile(".jupiter");
098          File outputPropertyFile = outputPropertyIFile.getLocation().toFile();
099          FileOutputStream fileOutputStream = new FileOutputStream(outputPropertyFile);
100          OutputStreamWriter writer = new OutputStreamWriter(fileOutputStream, "UTF-8");
101          XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
102          outputter.output(document, writer);
103          writer.close();
104        }
105        catch (IOException e) {
106          throw new ReviewException("IOException: " + e.getMessage());
107        }
108      }

110      /**
111       * Copies default config file in the <code>Project</code>. Leave the current config file in the
112       * project if the file already exists.
113       * @param outputPropertyFile the output property file.
114       * @return the config file <code>File</code> instance.
115       * @throws IOException if problems occur.
116       * @throws CoreException if problems occur.
117       */
118      private static File copyDefaultConfigFileTo(File outputPropertyFile) 
119                          throws IOException, CoreException {
120        if (!outputPropertyFile.exists()) {
121          outputPropertyFile.createNewFile();
122        }
123        URL pluginUrl = ReviewPlugin.getInstance().getInstallURL();
124        URL xmlUrl = FileLocator.toFileURL(new URL(pluginUrl, "default.jupiter"));
125        File sourceXmlFile = new File(xmlUrl.getFile());
126        // copy XML file in the plug-in directory to the state location.
127        Ver1PropertyHelper.copy(sourceXmlFile, outputPropertyFile);
128        return outputPropertyFile;
129      }

131      /**
132      * Copy the source file to the destination file.
133      * @param srouceFile the source <code>File</code>.
134      * @param destinationFile the destination <code>File</code>.
135      * @throws IOException if problems occur.
136      */
137     private static void copy(File srouceFile, File destinationFile) throws IOException {
138       FileChannel sourceChannel = new FileInputStream(srouceFile).getChannel();
139       FileChannel destinationChannel = new FileOutputStream(destinationFile).getChannel();
140       destinationChannel.transferFrom(sourceChannel, 0, sourceChannel.size());
141       sourceChannel.close();
142       destinationChannel.close();
143     }

174     /**
175      * Checks if the element contains the review id. Returns the contained <code>Element</code> if 
176      * the review id exists. Returns <code>null</code> if not.
177      * 
178      * @param jupiterConfigElement the jupiter config <code>Element</code> instance.
179      * @param reviewId the <code>reviewId</code> instance.
180      * @return the contained <code>Element</code> if the review id exists. 
181      * Returns <code>null</code> if not.
182      */
183     private static Element getReviewId(Element jupiterConfigElement, ReviewId reviewId) {
184       List reviewIdElementList = jupiterConfigElement.getChildren(ELEMENT_REVIEW_ID);
185       for (Iterator i = reviewIdElementList.iterator(); i.hasNext();) {
186         Element reviewIdElement = (Element) i.next();
187         if (reviewIdElement.getAttributeValue(ATTRIBUTE_NAME).equals(reviewId.getReviewId())) {
188           return reviewIdElement;
189         }
190       }
191       return null;
192     }

194     /**
195      * Reads the <code>IProject</code> instance to create the list of <code>ReviewId</code> instances
196      * and returns it.
197      * @param project the <code>IProject</code> instance.
198      * @param isDefaultLoaded sets <code>true</code> if the default review id is loaded too. 
199      * @return the created list of the <code>ReviewId</code> instances.
200      * @throws ReviewException if an error occurs during process.
201      */
202     public static List loadReviewIds(IProject project, boolean isDefaultLoaded)
203       throws ReviewException {
204       List reviewIdList = new ArrayList();
205       if (project != null) {
206         Document document = Ver1PropertyHelper.newDocument(project);
207         Element jupiterConfigElement = document.getRootElement();
208         List reviewIdElementList = jupiterConfigElement.getChildren(ELEMENT_REVIEW_ID);
209         for (Iterator i = reviewIdElementList.iterator(); i.hasNext();) {
210           Element reviewIdElement = (Element) i.next();
211           if (reviewIdElement != null) {
212             String reviewIdName = reviewIdElement.getAttributeValue(ATTRIBUTE_NAME);
213             String description = reviewIdElement.getAttributeValue(ATTRIBUTE_DESCRIPTION);
214             String author = reviewIdElement.getAttributeValue(ATTRIBUTE_AUTHOR);
215             author = (author != null) ? author : "";
216             // Supports 1.4.303 version. should be removed in the future.
217             description = (description != null) ? description : "";
218             if (isDefaultLoaded || !reviewIdName.equals(ATTRIBUTE_VALUE_DEFAULT)) {
219               String dateFormat = reviewIdElement.getAttributeValue(ATTRIBUTE_DATE_FORMAT);
220               String creationDate = reviewIdElement.getAttributeValue(ATTRIBUTE_CREATION_DATE);
221               Date date = createDate(creationDate, dateFormat);
222               Element storagePreferenceElement = getPreferenceElement(ATTRIBUTE_VALUE_STORAGE, 
223                                                                       reviewIdElement);
224               String directory = createDirectory(document, storagePreferenceElement);
225               Element reviewerPreferenceElement = getPreferenceElement(ATTRIBUTE_VALUE_REVIEWER, 
226                                                                        reviewIdElement);
227               Map reviewers = createReviewers(document, reviewerPreferenceElement);
228               Element categoryItemPrefElement = getPreferenceElement(ATTRIBUTE_VALUE_CATEGORY_ITEM, 
229                                                                            reviewIdElement);
230               Map categoryMap = createCategoryMap(categoryItemPrefElement);
231               reviewIdList.add(new ReviewId(reviewIdName, description, author, directory,
232                                             reviewers, categoryMap, date));
233             }
234           }
235         } 
236       }
237       return reviewIdList;
238     }

240     /**
241      * Gets the preference element with the attribute name. Returns null if the element is not found.
242      * @param attributeValueName the attribute name of the preference element.
243      * @param reviewIdElement the review id element.
244      * @return the preference <code>Element</code> instance.
245      */
246     private static Element getPreferenceElement(String attributeValueName, Element reviewIdElement) {
247       List preferenceList = reviewIdElement.getChildren(ELEMENT_PREFERENCE);
248       Element preferenceElement = null;
249       for (Iterator i = preferenceList.iterator(); i.hasNext();) {
250         Element tempElement = (Element) i.next();
251         String name = tempElement.getAttributeValue(ATTRIBUTE_NAME);
252         if (name.equals(attributeValueName)) {
253           preferenceElement = tempElement;
254           break;
255         }
256       }
257       return preferenceElement;
258     }

260     /**
261      * Creates the <code>Map</code> of the reviewer names. Returns empty map if 
262      * <code>Element</code> is null.
263      * @param document the <code>Document</code> instance.
264      * @param reviewerPreferenceElement the reviewer preference element.
265      * @return the <code>Map</code> of the reviewer names.
266      */
267     private static Map createReviewers(Document document, Element reviewerPreferenceElement) {
268       Map reviewers = new TreeMap();
269       if (reviewerPreferenceElement != null) {
270         List reviewerElementList = reviewerPreferenceElement.getChildren(ELEMENT_REVIEWER);
271         if (reviewerElementList.size() > 0) {
272           for (Iterator i = reviewerElementList.iterator(); i.hasNext();) {
273             Element reviewerElement = (Element) i.next();
274             String reviewerId = reviewerElement.getAttributeValue(ATTRIBUTE_ID);
275             String reviewerName = reviewerElement.getAttributeValue(ATTRIBUTE_NAME);
276             String enable = reviewerElement.getAttributeValue(ATTRIBUTE_ENABLE);
277             boolean isEnabled = new Boolean(enable).booleanValue();
278             reviewers.put(reviewerId, new ReviewerId(reviewerId, reviewerName));
279           }
280         }
281         else {
282           reviewers.putAll(getDefaultReviewers(reviewerPreferenceElement.getDocument()));        
283         }
284       }
285       else {
286    //     log.debug("reviewerPreferenceElement is null: there is no reviewer preference element."
287    //               + "read default reviewer values...");
288         reviewers.putAll(getDefaultReviewers(document));        
289       }
290       return reviewers;
291     }

293     /**
294      * Creates directory.
295      * @param document the <code>Document</code>.
296      * @param storagePreferenceElement the storage preference <code>Element</code>.
297      * @return the directory.
298      */
299     private static String createDirectory(Document document, Element storagePreferenceElement) {
300       String directory = "";
301       if (storagePreferenceElement != null) {
302         directory = storagePreferenceElement.getAttributeValue(ATTRIBUTE_DIRECTORY);
303       }
304       else {
305         directory = getDefaultDirectory(document);
306       }
307       return directory;
308     }

310     /**
311      * Gets default reviewer map from the  document.
312      * @param document the <code>Document</code> instance.
313      * @return the map of the <code>String</code> reviewer IDs.
314      */
315     private static Map getDefaultReviewers(Document document) {
316       Map defaultReviewers = new TreeMap();
317       Element jupiterConfigElement = document.getRootElement();
318       List reviewIdElementList = jupiterConfigElement.getChildren(ELEMENT_REVIEW_ID);
319       Element defaultPrefElement = getDefaultPrefElement(ATTRIBUTE_VALUE_REVIEWER, document);
320       if (defaultPrefElement != null) {
321         List reviewerElementList = defaultPrefElement.getChildren(ELEMENT_REVIEWER);
322         for (Iterator j = reviewerElementList.iterator(); j.hasNext();) {
323           Element reviewerElement = (Element) j.next();
324           String reviewerId = reviewerElement.getAttributeValue(ATTRIBUTE_ID);
325           String reviewerName = reviewerElement.getAttributeValue(ATTRIBUTE_NAME);
326           String enable = reviewerElement.getAttributeValue(ATTRIBUTE_ENABLE);
327           boolean isEnabled = new Boolean(enable).booleanValue();
328           ReviewerId reviewer = new ReviewerId(reviewerId, reviewerName);
329           defaultReviewers.put(reviewerId, reviewer);
330         }
331       }
332       return defaultReviewers;
333     }

336      * Gets default directory. Returns empty string if the default review id element does not exist
337      * or the value of the name attribute of the preference element in the default review id
338      * does not exist.
339      * @param document the document
340      * @return the default directory.

342     private static String getDefaultDirectory(Document document) { 
343       String directory = "";
344       Element defaultPrefElement = getDefaultPrefElement(ATTRIBUTE_VALUE_STORAGE, document);
345       if (defaultPrefElement != null) {
346         defaultPrefElement.getAttributeValue(ATTRIBUTE_DIRECTORY);
347       }
348       return directory;

351     /**
352      * Gets default preference element whose name attribute is <code>attributeValueName</code>.
353      * @param attributeValueName the value of the name attribute of the preference element in the 
354      * default review id.
355      * @param document the document.
356      * @return the default preference element whose name attribute is <code>attributeValueName</code>.
357      * Returns <code>null</code> if the default review id element does not exist or the value of the
358      * name attribute of the preference element in the default review id does not exist.
359      */
360     private static Element getDefaultPrefElement(String attributeValueName, Document document) {
361       Element jupiterConfigElement = document.getRootElement();
362       List reviewIdElementList = jupiterConfigElement.getChildren(ELEMENT_REVIEW_ID);
363       for (Iterator i = reviewIdElementList.iterator(); i.hasNext();) {
364         Element reviewIdElement = (Element) i.next();
365         if (reviewIdElement.getAttributeValue(ATTRIBUTE_NAME).equals(ATTRIBUTE_VALUE_DEFAULT)) {
366           Element defaultPrefElement = getPreferenceElement(attributeValueName, reviewIdElement);
367           return defaultPrefElement;
368         }
369       }
370       return null;
371     }

374     /**
375      * Creates the <code>Map</code> with the category <code>String</code> name and <code>List</code>
376      * of items. Returns empty list if <code>Element</code> is null.
377      * @param categoryItemPreferenceElement the category item preference element.
378      * @return the <code>Map</code> with the category <code>String</code> name and <code>List</code>
379      * of items.
380      */
381     private static Map createCategoryMap(Element categoryItemPreferenceElement) {
382       Map categoryMap = new HashMap();
383       if (categoryItemPreferenceElement != null) {
384         List categoryElementList = categoryItemPreferenceElement.getChildren(ELEMENT_CATEGORY);
385         for (Iterator i = categoryElementList.iterator(); i.hasNext();) {
386           Element categoryElement = (Element) i.next();
387           String categoryName = categoryElement.getAttributeValue(ATTRIBUTE_NAME);
388           List itemElementList = categoryElement.getChildren(ELEMENT_ITEM);
389           List itemList = new ArrayList();
390           for (Iterator j = itemElementList.iterator(); j.hasNext();) {
391             Element itemElement = (Element) j.next();
392             String itemName = itemElement.getAttributeValue(ATTRIBUTE_NAME);
393             itemList.add(itemName);
394           }
395           categoryMap.put(categoryName, itemList);
396         }
397       }
398       return categoryMap;
399     }

401     /**
402      * Creates the <code>Date</code> instance associated with the <code>dateString</code>. Note the
403      * this returns current time <code>Date</code> instance if <code>dateString</code> could not be
404      * parsed with <code>dateFormat</code>.
405      *
406      * @param dateString the date string to be parsed.
407      * @param dateFormat the date format to let parser know the date string to be parsed.
408      *
409      * @return the <code>Date</code> instance associated with the <code>dateString</code>.
410      */
411     private static Date createDate(String dateString, String dateFormat) {
412       if (dateString == null || dateFormat == null) {
413         return new Date();
414       }
415       try {
416         return new SimpleDateFormat(dateFormat).parse(dateString);
417       }
418       catch (ParseException e) {
419         log.warning(e.getMessage());
420         return new Date();
421       }
422     }

435     /**
436      * Creates the reviewer preference element
437      * @param reviewerList the <code>List</code> instance of the reviewer names.
438      * @return the <code>Element</code> of the review preference.
439      */
440     private static Element createReviewerPreferenceElement(List reviewerList) {
441       Element reviewerPreferenceElement = new Element(ELEMENT_PREFERENCE);
442       reviewerPreferenceElement.setAttribute(ATTRIBUTE_NAME, ATTRIBUTE_VALUE_REVIEWER);
443       for (Iterator i = reviewerList.iterator(); i.hasNext();) {
444         String reviewer = (String) i.next();
445         Element reviewerElement = new Element(ELEMENT_REVIEWER);
446         reviewerElement.setAttribute(ATTRIBUTE_NAME, reviewer);
447         reviewerPreferenceElement.addContent(reviewerElement);
448       }
449       return reviewerPreferenceElement;
450     }

452     /**
453      * Creates root review id element.
454      * @param reviewId the <code>ReviewId</code> instance.
455      * @return the <code>Element</code> of the review id.
456      */
457     private static Element createReviewIdElement(ReviewId reviewId) {
458       Element reviewIdElement = new Element(ELEMENT_REVIEW_ID);
459       reviewIdElement.setAttribute(ATTRIBUTE_NAME, reviewId.getReviewId());
460       reviewIdElement.setAttribute(ATTRIBUTE_DESCRIPTION, reviewId.getDescription());
461       reviewIdElement.setAttribute(ATTRIBUTE_AUTHOR, reviewId.getAuthor());
462       String formatString = "yyyy-MM-dd :: HH:mm:ss z";
463       reviewIdElement.setAttribute(ATTRIBUTE_DATE_FORMAT, formatString);
464       SimpleDateFormat format = new SimpleDateFormat(formatString);
465       reviewIdElement.setAttribute(ATTRIBUTE_CREATION_DATE, format.format(reviewId.getDate()));
466       return reviewIdElement;
467     }

469     /**
470      * Gets the list of the <code>ReviewId</code> instances.
471      * @param projectName the project name.
472      * @param isDefaultLoaded sets <code>true</code> if the default review id is loaded too.
473      * @return the list of the <code>ReviewId</code> instances.
474      */
475     public static List getReviewIdList(String projectName, boolean isDefaultLoaded) {
476       IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
477       IProject selectedProject = root.getProject(projectName);
478       List reviewIdList = new ArrayList();
479       if (selectedProject != null && projectName != null) {
480         try {
481           reviewIdList = Ver1PropertyHelper.loadReviewIds(selectedProject, 
482                                                                    isDefaultLoaded);
483           Collections.sort(reviewIdList, new Comparator() {
484             public int compare(Object object1, Object object2) {
485               ReviewId reviewId1 = (ReviewId) object1;
486               ReviewId reviewId2 = (ReviewId) object2;
487               return reviewId2.getDate().compareTo(reviewId1.getDate());
488             }
489           });
490         }
491         catch (Exception e) {
492           e.printStackTrace();
493           log.error(e);
494         }
495       }
496       return reviewIdList;
497     }
