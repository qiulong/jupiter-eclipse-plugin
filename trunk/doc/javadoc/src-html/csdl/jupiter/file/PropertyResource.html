001    package csdl.jupiter.file;
002    
003    import java.util.ArrayList;
004    import java.util.Comparator;
005    import java.util.Date;
006    import java.util.Iterator;
007    import java.util.List;
008    import java.util.Map;
009    import java.util.TreeMap;
010    
011    import org.eclipse.core.resources.IProject;
012    import org.jdom.Document;
013    import org.jdom.Element;
014    
015    import csdl.jupiter.ReviewException;
016    import csdl.jupiter.model.review.ReviewId;
017    import csdl.jupiter.util.JupiterLogger;
018    
019    /**
020     * Provides the review id wrapper singleton class. 
021     * 
022     * @author Takuya Yamashita
023     * @version $Id: PropertyResource.java,v 1.9 2005/03/25 18:32:07 takuyay Exp $
024     */
025    public class PropertyResource {
026      /** Jupiter logger */
027      private JupiterLogger log = JupiterLogger.getLogger();
028      
029      private static PropertyResource theInstance = new PropertyResource();
030      private Map reviewIdMap;
031      private IProject project;
032      private boolean isDefaultLoaded;
033      private Document propertyDocument;
034      private Map reviewIdReviewElementMap;
035    
036      /** The default review id. */
037      public static final String DEFAULT_ID = PropertyConstraints.DEFAULT_REVIEW_ID;
038      /**
039       * Prohibits clients from instantiating this.
040       */
041      private PropertyResource() {
042        this.reviewIdMap = new TreeMap();
043        this.reviewIdReviewElementMap = new TreeMap();
044      }
045    
046      /**
047       * Gets the review id wrapper class for the project.
048       * 
049       * @param project the project.
050       * @param isDefaultLoaded sets <code>true</code> if the default review id is loaded too.
051       * @return the review id wrapper class which contains set of <code>ReviewID</code> instances.
052       */
053      public static PropertyResource getInstance(IProject project, boolean isDefaultLoaded) {
054        theInstance.setValues(project, isDefaultLoaded);
055        theInstance.fillReviewIdReviewElementMap(project, isDefaultLoaded);
056        theInstance.fillReviewIdMap();
057        return theInstance;
058      }
059    
060      /**
061       * Sets project and isDefaultLoaded values.
062       * 
063       * @param project the project.
064       * @param isDefaultLoaded set true if default review id is also loaded.
065       */
066      private void setValues(IProject project, boolean isDefaultLoaded) {
067        this.project = project;
068        this.isDefaultLoaded = isDefaultLoaded;
069      }
070      
071      /**
072       * Loads the 'Review' <code>Element</code>.
073       * @param project the project.
074       * @param isDefaultLoaded true if default review id is loaded too.
075       */
076      private void fillReviewIdReviewElementMap(IProject project, boolean isDefaultLoaded) {
077        try {
078          this.propertyDocument = PropertyXmlSerializer.newDocument(project);
079        }
080        catch (ReviewException e) {
081          log.error(e);
082        }
083        if (this.propertyDocument == null) {
084          return;
085        }
086        Element propertyElement = this.propertyDocument.getRootElement();
087        List reviewElementList = propertyElement.getChildren(PropertyConstraints.ELEMENT_REVIEW);
088        this.reviewIdReviewElementMap.clear();
089        for (Iterator i = reviewElementList.iterator(); i.hasNext();) {
090          Element reviewElement = (Element) i.next();
091          String reviewId = reviewElement.getAttributeValue(PropertyConstraints.ATTRIBUTE_ID);
092          if (reviewId.equals(PropertyConstraints.DEFAULT_REVIEW_ID) && !isDefaultLoaded) {
093            continue;
094          }
095          this.reviewIdReviewElementMap.put(reviewId, reviewElement);
096        }
097      }
098      
099      /**
100       * Gets the <code>ReviewResource</code> instance associating with the review id.
101       * Returns null if the review id does not exist.
102       * @param reviewId the review id.
103       * @param isClone true if the <code>ReviewResource</code> contains the clone of the
104       *  review element. false if it contains the review element of the document.
105       * @return the <code>ReviewResource</code> instance associating with the review id.
106       * Returns null if the review id does not exist.
107       */
108      public ReviewResource getReviewResource(String reviewId, boolean isClone) {
109        Element reviewElement = (Element) this.reviewIdReviewElementMap.get(reviewId);
110        if (reviewElement != null) {
111          reviewElement = (isClone) ? (Element) reviewElement.clone() : reviewElement;
112         return new ReviewResource(reviewElement);
113        }
114        return null;
115      }
116    
117      /**
118       * Loads <code>ReviewId</code> instances for the project into review id map.
119       */
120      private void fillReviewIdMap() {
121        this.reviewIdMap.clear();
122        for (Iterator i = this.reviewIdReviewElementMap.values().iterator(); i.hasNext();) {
123          Element reviewIdElement = (Element) i.next();
124          ReviewResource reviewResource = new ReviewResource(reviewIdElement);
125          ReviewId reviewId = reviewResource.getReviewId();
126          this.reviewIdMap.put(reviewId.getReviewId(), reviewId);
127        }
128      }
129    
130      /**
131       * Gets the array of the <code>String</code> review id names. Note that the order of the
132       * elements is reverse chronological order in which review id was created.
133       * 
134       * @return the array of the <code>String</code> review id names
135       */
136      public String[] getReviewIdNames() {
137        List reviewIdList = getReviewIdList();
138        Map reviewIdNameMap = new TreeMap(new Comparator() {
139          public int compare(Object object1, Object object2) {
140            return ((Date) object2).compareTo((Date) object1);
141          }
142        });
143        for (Iterator i = reviewIdList.iterator(); i.hasNext();) {
144          ReviewId reviewId = (ReviewId) i.next();
145          reviewIdNameMap.put(reviewId.getDate(), reviewId.getReviewId());
146        }
147        return (String[]) new ArrayList(reviewIdNameMap.values()).toArray(new String[]{});
148      }
149    
150      /**
151       * Gets the array of the <code>String</code> reviewer id names.
152       * 
153       * @param reviewIdName the review id.
154       * @return the array of the <code>String</code> review id names
155       */
156      public String[] getReviewerIdNames(String reviewIdName) {
157        Map reviewers = getReviewers(reviewIdName);
158        return (String[]) new ArrayList(reviewers.keySet()).toArray(new String[]{});
159      }
160    
161      /**
162       * Gets the map of the <code>ReviewerId</code> instances. Returns the map of default reviewers
163       * if no reviewer exists.
164       * 
165       * @param reviewIdName the review id name.
166       * @return the <code>Map</code> of the <code>ReviewerId</code> instance.
167       */
168      public Map getReviewers(String reviewIdName) {
169        ReviewId reviewId = (ReviewId) this.reviewIdMap.get(reviewIdName);
170        Map reviewersMap = new TreeMap();
171        if (reviewId != null) {
172          reviewersMap = reviewId.getReviewers();
173        }
174        return reviewersMap;
175      }
176    
177      /**
178       * Gets the list of the <code>ReviewId</code> instances.
179       * 
180       * @return the list of the <code>ReviewId</code> instances.
181       */
182      public List getReviewIdList() {
183        return new ArrayList(this.reviewIdMap.values());
184      }
185    
186      /**
187       * Gets the <code>ReviewId</code> instance from the reviewIdName. Returns null if the
188       * <code>reviewIdName</code> does not exist.
189       * 
190       * @param reviewIdName the review id name.
191       * @return the <code>ReviewId</code> instance from the reviewIdName. Returns null if the review
192       *         id name does not exist.
193       */
194      public ReviewId getReviewId(String reviewIdName) {
195        return (ReviewId) this.reviewIdMap.get(reviewIdName);
196      }
197    
198      /**
199       * Adds <code>ReviewResource</code> instance to the property config file.
200       * @param reviewResource the <code>ReviewResource</code> instance.
201       * @throws ReviewException if the review id could not be written.
202       * @return <code>true</code> if review id does not exist and could be written.
203       * <code>false</code> if review id already exist.
204       */
205      public boolean addReviewResource(ReviewResource reviewResource) throws ReviewException  {
206        Element propertyElement = this.propertyDocument.getRootElement();
207        if (reviewResource != null) {
208          Element reviewElement = reviewResource.getReviewElement();
209          ReviewId reviewId = reviewResource.getReviewId();
210          propertyElement.addContent(reviewElement);
211          PropertyXmlSerializer.serializeDocument(propertyDocument, project);
212          this.reviewIdReviewElementMap.put(reviewId.getReviewId(), reviewElement);
213          this.reviewIdMap.put(reviewId.getReviewId(), reviewId);
214          return true;
215        }
216        else {
217          return false;
218        }
219      }
220    
221      /**
222       * Removes <code>ReviewId</code> instance from the property config file.
223       * @param reviewId the <code>ReviewId</code> instance.
224       * @throws ReviewException if the review id could not be written.
225       * @return <code>true</code> if review id exists and could be written.
226       * <code>false</code> if review id does not exist.
227       */
228      public boolean removeReviewResource(ReviewId reviewId) throws ReviewException {
229        Element propertyElement = this.propertyDocument.getRootElement();
230        ReviewResource reviewResource = getReviewResource(reviewId.getReviewId(), false);
231        // if the element exists, i.e. the review id exists.
232        if (reviewResource != null) {
233          Element targetReviewElement = reviewResource.getReviewElement();
234          propertyElement.removeContent(targetReviewElement);
235          PropertyXmlSerializer.serializeDocument(this.propertyDocument, project);
236          this.reviewIdReviewElementMap.remove(reviewId.getReviewId());
237          this.reviewIdMap.remove(reviewId.getReviewId());
238          return true;
239        }
240        else {
241          return false;
242        }
243      }
244    }
