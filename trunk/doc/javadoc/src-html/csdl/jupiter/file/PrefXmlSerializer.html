001    package csdl.jupiter.file;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.io.FileOutputStream;
006    import java.io.IOException;
007    import java.io.OutputStreamWriter;
008    import java.net.URL;
009    import java.nio.channels.FileChannel;
010    
011    import org.eclipse.core.runtime.FileLocator;
012    import org.jdom.Document;
013    import org.jdom.Element;
014    import org.jdom.JDOMException;
015    import org.jdom.input.SAXBuilder;
016    import org.jdom.output.Format;
017    import org.jdom.output.XMLOutputter;
018    
019    import csdl.jupiter.ReviewException;
020    import csdl.jupiter.ReviewPlugin;
021    import csdl.jupiter.util.JupiterLogger;
022    
023    /**
024     * Provides the function to read and write information in the config.xml file.
025     *
026     * @author Takuya Yamashita
027     * @version $Id: PrefXmlSerializer.java,v 1.4 2005/03/25 18:32:08 takuyay Exp $
028     */
029    public class PrefXmlSerializer {
030      /** Jupiter logger */
031      private static JupiterLogger log = JupiterLogger.getLogger();
032    
033      /** The singleton instance */
034      private static PrefXmlSerializer theInstance;
035      /** The preference file. */
036      private static final String PREFERENCE_XML_FILE = "preference.xml";
037    
038      /**
039       * Prohibits the instantiation from clients.
040       */
041      private PrefXmlSerializer() {
042      }
043      
044      /**
045       * Gets the instance of <code>PrefXmlSerializer</code>. Note that the lazy instantiation is
046       * necessary to call<code>ReviewPlugin.getInstance()</code>. Otherwise the method returns
047       * null.
048       * @return the instance of <code>PrefXmlSerializer</code>.
049       */
050      public static PrefXmlSerializer getInstance() {
051        if (theInstance == null) {
052          theInstance = new PrefXmlSerializer();
053        }
054        return theInstance;
055      }
056      
057      /**
058       * Loads root preference element from the PREFERENCE_XML_FILE file in the plug-in directory.
059       * @return the <code>Element</code> of the default Review.
060       */
061      public static Element loadPreferenceElement() {
062        ReviewPlugin plugin = ReviewPlugin.getInstance();
063        File stateLocationXmlFile = plugin.getStateLocation().append(PREFERENCE_XML_FILE).toFile();
064        URL pluginUrl = ReviewPlugin.getInstance().getInstallURL();
065        try {
066          URL xmlUrl = FileLocator.toFileURL(new URL(pluginUrl, PREFERENCE_XML_FILE));
067          File pluginXmlFile = new File(xmlUrl.getFile());
068          if (!stateLocationXmlFile.exists()) {
069            copy(pluginXmlFile, stateLocationXmlFile);
070          }
071          SAXBuilder builder = new SAXBuilder();
072          Document document = builder.build(stateLocationXmlFile);
073          Element preferenceElement = document.getRootElement();
074          return preferenceElement;
075        }
076        catch (IOException e) {
077          log.error(e);
078        }
079        catch (JDOMException e) {
080          log.error(e);
081        }
082        return null;
083      }
084      
085      /**
086       * Serializes <code>Document</code> to the PREFERENCE_XML_FILE file.
087       * @param document the document.
088       * @throws ReviewException if an error occurs during serialization.
089       */
090      public static void serializeDocument(Document document) throws ReviewException {
091        try {
092          ReviewPlugin plugin = ReviewPlugin.getInstance();
093          File stateLocationXmlFile = plugin.getStateLocation().append(PREFERENCE_XML_FILE).toFile();
094          FileOutputStream fileOutputStream = new FileOutputStream(stateLocationXmlFile);
095          OutputStreamWriter writer = new OutputStreamWriter(fileOutputStream, "UTF-8");
096          XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
097          outputter.output(document, writer);
098          writer.close();
099        }
100        catch (IOException e) {
101          throw new ReviewException("IOException: " + e.getMessage());
102        }
103      }
104      
105      /**
106       * Copy the source file to the destination file.
107       * @param srouceFile the source <code>File</code>.
108       * @param destinationFile the destination <code>File</code>.
109       * @throws IOException if problems occur.
110       */
111      static void copy(File srouceFile, File destinationFile) throws IOException {
112        FileChannel sourceChannel = new FileInputStream(srouceFile).getChannel();
113        FileChannel destinationChannel = new FileOutputStream(destinationFile).getChannel();
114        destinationChannel.transferFrom(sourceChannel, 0, sourceChannel.size());
115        sourceChannel.close();
116        destinationChannel.close();
117      }
118    }
