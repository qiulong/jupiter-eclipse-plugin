102      /**
103       * Gets the root element of the tree element from the ReviewIssue instance.
104       *
105       * @param reviewIssue The ReviewIssue instance.
106       *
107       * @return The root element of the tree element.
108       */
109      private static Element createReviewIssueElement(ReviewIssue reviewIssue) {
110        if (reviewIssue == null) {
111          throw new NullPointerException("ReviewIssue instance is null.");
112        }
113        Element codeReviewRoot = new Element(REVIEW_ISSUE);
114        codeReviewRoot.setAttribute(REVIEW_ISSUE_ATTRIBUTE_ID, reviewIssue.getIssueId());
115        Element codeReviewMetaElement = new Element(REVIEW_ISSUE_META);
116        codeReviewRoot.addContent(codeReviewMetaElement);
117        Element creationDateElement = new Element(CREATION_DATE);
118        creationDateElement.setAttribute(FORMAT_ATTRIBUTE, DATE_FORMAT_PATTERN);
119        SimpleDateFormat format = new SimpleDateFormat(DATE_FORMAT_PATTERN);
120        creationDateElement.setText(format.format(reviewIssue.getCreationDate()));
121        codeReviewMetaElement.addContent(creationDateElement);
122        Element lastModificationDateElement = new Element(LAST_MODIFICATION_DATE);
123        lastModificationDateElement.setAttribute(FORMAT_ATTRIBUTE, DATE_FORMAT_PATTERN);
124        lastModificationDateElement.setText(format.format(reviewIssue.getModificationDate()));
125        codeReviewMetaElement.addContent(lastModificationDateElement);
126        Element reviewerElement = new Element(REVIEWER_ID);
127        reviewerElement.setText(reviewIssue.getReviewer());
128        codeReviewRoot.addContent(reviewerElement);
129        Element assignedToElement = new Element(ASSIGNED_TO);
130        assignedToElement.setText(reviewIssue.getAssignedTo());
131        codeReviewRoot.addContent(assignedToElement);
132        Element fileElement = new Element(FILE);
133        fileElement.setText(reviewIssue.getTargetFile());
134        fileElement.setAttribute(FILE_ATTRIBUTE_LINE, reviewIssue.getLine());
135        codeReviewRoot.addContent(fileElement);
136        Element typeElement = new Element(TYPE);
137        typeElement.setText(reviewIssue.getType().getKey());
138        codeReviewRoot.addContent(typeElement);
139        Element severityElement = new Element(SEVERITY);
140        severityElement.setText(reviewIssue.getSeverity().getKey());
141        codeReviewRoot.addContent(severityElement);
142        Element summaryElement = new Element(SUMMARY);
143        summaryElement.setText(reviewIssue.getSummary());
144        codeReviewRoot.addContent(summaryElement);
145        Element descriptionElement = new Element(DESCRIPTION);
146        descriptionElement.setText(reviewIssue.getDescription());
147        codeReviewRoot.addContent(descriptionElement);
148        Element annotationElement = new Element(ANNOTATION);
149        annotationElement.setText(reviewIssue.getAnnotation());
150        codeReviewRoot.addContent(annotationElement);
151        Element revisionElement = new Element(REVISION);
152        revisionElement.setText(reviewIssue.getRevision());
153        codeReviewRoot.addContent(revisionElement);
154        Element resolutionElement = new Element(RESOLUTION);
155        resolutionElement.setText(reviewIssue.getResolution().getKey());
156        codeReviewRoot.addContent(resolutionElement);
157        Element statusElement = new Element(STATUS);
158        statusElement.setText(reviewIssue.getStatus().getKey());
159        codeReviewRoot.addContent(statusElement);
160        return codeReviewRoot;
161      }

163      /**
164       * Gets the filled <code>ReviewIssue</code> instance from the <code>Element</code> instance whose
165       * sub elements contain information of code review.
166       *
167       * @param codeReviewElement The <code>Element</code> instance to hold the root element whose sub
168       *        elements contain information for <code>ReviewIssue</code> instance.
169       * @param reviewIFile the review <code>iFile</code> instance to be stored in the
170       *        <code>ReviewIssue</code> instance.
171       *
172       * @return The filled <code>ReviewIssue</code> instance.
173       *
174       * @throws ReviewException if the passing <code>Element</code> is null or the problem occurs
175       *         when <code>ReviewIssue</code> is being created.
176       */
177      private static ReviewIssue createReviewIssue(Element codeReviewElement, IFile reviewIFile)
178        throws ReviewException {
179        if (codeReviewElement == null) {
180          new ReviewException("element instance is null");
181        }
182        try {
183          Element codeReviewMetaElement = codeReviewElement.getChild(REVIEW_ISSUE_META);
184          Element creationDateElement = codeReviewMetaElement.getChild(CREATION_DATE);
185          Element lastModificationDateElement = codeReviewMetaElement.getChild(LAST_MODIFICATION_DATE);
186          Element reviewerElement = codeReviewElement.getChild(REVIEWER_ID);
187          Element assignedToElement = codeReviewElement.getChild(ASSIGNED_TO);
188          Element fileElement = codeReviewElement.getChild(FILE);
189          Element typeElement = codeReviewElement.getChild(TYPE);
190          Element severityElement = codeReviewElement.getChild(SEVERITY);
191          Element summaryElement = codeReviewElement.getChild(SUMMARY);
192          Element descriptionElement = codeReviewElement.getChild(DESCRIPTION);
193          Element annotationElement = codeReviewElement.getChild(ANNOTATION);
194          Element revisionElement = codeReviewElement.getChild(REVISION);
195          Element resolutionElement = codeReviewElement.getChild(RESOLUTION);
196          Element statusElement = codeReviewElement.getChild(STATUS);
197          String creationDateFormat = getAttributeText(creationDateElement, FORMAT_ATTRIBUTE);
198          String creationDateString = getElementText(creationDateElement);
199          Date creationDate = createDate(creationDateString, creationDateFormat);
200          String lastModDateFormat = getAttributeText(lastModificationDateElement, FORMAT_ATTRIBUTE);
201          String lastModificationDateString = getElementText(lastModificationDateElement);
202          Date lastModificationDate = createDate(lastModificationDateString, lastModDateFormat);
203          String reviewer = getElementText(reviewerElement);
204          String assignedTo = getElementText(assignedToElement);
205          String targetFilePath = getElementText(fileElement);
206          String summary = getElementText(summaryElement);
207          String description = getElementText(descriptionElement);
208          String annotation = getElementText(annotationElement);
209          String revision = getElementText(revisionElement);
210          String id = getAttributeText(codeReviewElement, REVIEW_ISSUE_ATTRIBUTE_ID);
211          String fullyQualifiedClass = getAttributeText(fileElement, FILE_ATTRIBUTE_CLASS);
212          String lineNumber = getAttributeText(fileElement, FILE_ATTRIBUTE_LINE);
213          Type type = getType(typeElement);
214          Severity severity = getSeverity(severityElement);
215          Resolution resolution = getResolution(resolutionElement);
216          Status status = getStatus(statusElement);
217          return new ReviewIssue(creationDate, lastModificationDate, reviewer, assignedTo,
218                                 targetFilePath, lineNumber, type, severity, summary, description,
219                                 annotation, revision, resolution, status, reviewIFile);
220        }
221        catch (NullPointerException e) {
222          throw new ReviewException(e.getMessage());
223        }
224      }

243       *
244       * @param typeElement the type <code>Element</code> instance. Note that this supports the 1.4.212
245       *        version or below to convert old text to a key.
246       *
247       * @return the <code>Type</code> instance.

250        String typeText = getElementText(typeElement);
251        String typeKey = "";
252        if (typeText.equals("Defect")) {
253          typeKey = ResourceBundleKey.ITEM_KEY_TYPE_DEFECT;
254        }
255        else if (typeText.equals("External_Issue")) {
256          typeKey = ResourceBundleKey.ITEM_KEY_TYPE_EXTERNAL_ISSUE;
257        }
258        else if (typeText.equals("Question")) {
259          typeKey = ResourceBundleKey.ITEM_KEY_TYPE_QUESTION;
260        }
261        else if (typeText.equals("Praise")) {
262          typeKey = ResourceBundleKey.ITEM_KEY_TYPE_PRAISE;
263        }
264        else {
265          typeKey = typeText;
266        }
267        ReviewModel reviewModel = ReviewModel.getInstance();
268        IProject project = reviewModel.getProjectManager().getProject();
269        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
270        return new Type(typeKey, TypeKeyManager.getInstance(project, reviewId).getOrdinal(typeKey));
271      }

411      /**
412       * Writes the elements which starts for the element into XML output file.
413       *
414       * @param outputXml The XML output file.
415       * @param element The root element to be written to the output.
416       *
417       * @throws IOException if problems occur.
418       */
419      private static void write(File outputXml, Element element)
420        throws IOException {
421        OutputStreamWriter writer = new OutputStreamWriter(new FileOutputStream(outputXml), "UTF-8");
422    
423        // output JDOM tree to file
424        Document elementDocument = new Document(element);
425        XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
426        outputter.output(elementDocument, writer);
427        writer.close();
428      }

430      /**
431       * Writes the content of all the code review data stored in the model into the XML file. Note
432       * that clients does not need to check if the specified file path (either directory or file)
433       * exists.
434       * <p>
435       * Clients should check if the passing <code>File</code> or/and <code>ReviewIssueModel</code>
436       * instance is not null.
437       * Otherwise the <code>IllegalArgumentException</code> is thrown.
438       *
439       * @param reviewId the review ID.
440       * @param model The model which contains the <code>ReviewIssue</code> instances.
441       * @param xmlFile The output XML file with absolute path.
442       *
443       * @throws IOException if problems occur.
444       */
445      public static void write(ReviewId reviewId, ReviewIssueModel model, File xmlFile)
446        throws IOException {
447        if (xmlFile == null) {
448          log.debug("XML file instance is null.");
449          throw new IllegalArgumentException("File instance is null.");
450        }
451        if (model == null) {
452          log.debug("Model is null.");
453          throw new IllegalArgumentException("ReviewIssueModel instance is null.");
454        }
455        // if outputFile does not exit, create them.
456        if (!xmlFile.getParentFile().exists()) {
457          try {
458            xmlFile.getParentFile().mkdirs();
459          }
460          catch (SecurityException e) {
461            // show can-not-create directory message.
462            log.debug(e.getMessage());
463            throw new SecurityException(e.getMessage());
464          }
465        }
466        if (!xmlFile.exists()) {
467          try {
468            xmlFile.createNewFile();
469          }
470          catch (SecurityException e) {
471            log.debug(e.getMessage());
472            throw new SecurityException(e.getMessage());
473          }
474        }
475        if (!xmlFile.canWrite()) {
476          // show can-not-write message in a file.
477          log.debug(xmlFile + " can not be written");
478          throw new RuntimeException(xmlFile + " is not writable to record review issue. " + 
479                "Please make it writable before proceed. Issue from " + reviewId.getAuthor() +
480                reviewId.getAuthor() + " with message \"" + reviewId.getDescription() + 
481                "\" is not saved.");      
482        }
483        
484        Element reviewRootElement = new Element(REVIEW);
485        reviewRootElement.setAttribute(REVIEW_ATTRIBUTE_ID, reviewId.getReviewId());
486        if (model != null) {
487          for (Iterator i = model.iterator(); i.hasNext();) {
488            ReviewIssue reviewIssue = (ReviewIssue) i.next();
489            if (xmlFile.equals(reviewIssue.getReviewIFile().getLocation().toFile())) {
490              Element reviewIssueElement = createReviewIssueElement(reviewIssue);
491              reviewRootElement.addContent(reviewIssueElement);
492            }
493          }
494        }
495        log.debug("writing " + xmlFile + " ...");
496        write(xmlFile, reviewRootElement);
497      }

521      /**
522       * Reads the xmlFile of the <code>IFile</code> instance to create <code>ReviewIssue</code>
523       * instances in the <code>ReviewIssueModel</code> instance. Note that the review file will
524       * be skipped if the problems occur on the reading process (e.g. XML file is not valid due to
525       * the file confliction).
526       *
527       * @param reviewId the review id.
528       * @param model The <code>ReviewIssueModel</code> instance to hold <code>ReviewIssue</code>
529       *        instances.
530       * @param iFiles The array of IFile implementing class instance to hold file path.
531       * @return <code>true</code> if all files are successfully read. <code>false</code> if 
532       * there is a file to be unread.
533       */
534      public static boolean read(ReviewId reviewId, ReviewIssueModel model, IFile[] iFiles) {
535        if (reviewId == null || model == null || iFiles == null) {
536          throw new IllegalArgumentException ("ReviewId, ReviewIssueModel, or IFile[] is null");
537        }
538        boolean isSuccessIterationForAll = true;
539        for (int i = 0; i < iFiles.length; i++) {
540          log.debug("reading " + iFiles[i].getLocation() + " ...");
541          File xmlFile = new File(iFiles[i].getLocation().toString());
542          SAXBuilder builder = new SAXBuilder();
543          Document document = null;
544          try {
545            document = builder.build(xmlFile);
546          }
547          catch (Exception e) {
548            log.error(e);
549            isSuccessIterationForAll = false;
550            continue;
551          }
552          Element codeReviewsElement = document.getRootElement();
553          String reviewIdName = codeReviewsElement.getAttributeValue(REVIEW_ATTRIBUTE_ID);
554          if (reviewIdName != null && reviewId != null && reviewIdName.equals(reviewId.getReviewId())) {
555            List codeReviewList = codeReviewsElement.getChildren();
556            List tempCodeReviewList = new ArrayList();
557            boolean isSuccessIterationForFile = true;
558            for (Iterator iterator = codeReviewList.iterator(); iterator.hasNext();) {
559              Element codeReviewElement = (Element) iterator.next();
560              try {
561                ReviewIssue reviewIssue = createReviewIssue(codeReviewElement, iFiles[i]);
562                tempCodeReviewList.add(reviewIssue);
563              }
564              catch (ReviewException e) {
565                log.error(e);
566                isSuccessIterationForFile = false;
567                isSuccessIterationForAll = false;
568                break;
569              }
570            }
571      
572            // adds the list to the model only when the iteration of the file succeed.
573            if (isSuccessIterationForFile) {
574              model.addAll(tempCodeReviewList);
575            }
576          }
577        }
578        return isSuccessIterationForAll;
579      }

581      /**
582       * Checks if the passing <code>File</code> instance is associated with <code>String</code>
583       * review ID.
584       * @param reviewId the review ID.
585       * @param reviewFile the review file to be checked.
586       * @return <code>true</code> if the code>File</code> instance is associated with
587       * <code>String</code>.
588       * @throws ReviewException if problems occur during file reading process.
589       */
590      static boolean isReviewIdAssociatedFile(String reviewId, File reviewFile) throws ReviewException {
591        SAXBuilder builder = new SAXBuilder();
592        Document document = null;
593        try {
594          document = builder.build(reviewFile);
595        }
596        catch (JDOMException e) {
597          throw new ReviewException("JDOMException: " + e.getMessage());
598        }
599        catch (IOException e) {
600          throw new ReviewException("IOException: " + e.getMessage());
601        }
602        Element codeReviewsElement = document.getRootElement();
603        String reviewIdName = codeReviewsElement.getAttributeValue(REVIEW_ATTRIBUTE_ID);
604        return (reviewIdName != null && reviewIdName.equals(reviewId));
605      }

610       * @throws ReviewException if problems occur during the file deletion.

614          try {
615            iFiles[i].delete(true, false, null);
616          }
617          catch (CoreException e) {
618            throw new ReviewException(e.getMessage());
619          }
620        }
621      }
