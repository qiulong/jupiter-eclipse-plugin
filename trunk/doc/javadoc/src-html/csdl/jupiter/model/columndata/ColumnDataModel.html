001    package csdl.jupiter.model.columndata;
002    
003    import java.util.ArrayList;
004    import java.util.HashMap;
005    import java.util.Iterator;
006    import java.util.List;
007    import java.util.Map;
008    
009    import org.eclipse.jface.viewers.ColumnPixelData;
010    
011    import csdl.jupiter.ReviewI18n;
012    import csdl.jupiter.util.ResourceBundleKey;
013    
014    /**
015     * Provides the container class for <code>ColumnData</code> instances.
016     *
017     * @author Takuya Yamashita
018     * @version $Id: ColumnDataModel.java,v 1.3 2005/03/25 18:32:09 takuyay Exp $
019     */
020    public class ColumnDataModel {
021      /** The list of the all <code>ColumnData</code> instances. */
022      private List allColumnNameKeyList;
023      /** The map for the column name key and <code>ColumnData</code> value. */
024      private Map columnNameKeyMap;
025    
026      /**
027       * Initializes the lists of <code>ColumnData</code> and default review phase name.
028       */
029      public ColumnDataModel() {
030        this.allColumnNameKeyList = new ArrayList();
031        this.columnNameKeyMap = new HashMap();
032      }
033    
034      /**
035       * Adds the <code>ColumnData</code> instance.
036       *
037       * @param columnData the <code>ColumnData</code> instance
038       */
039      public void add(ColumnData columnData) {
040        this.columnNameKeyMap.put(columnData.getColumnNameKey(), columnData);
041        this.allColumnNameKeyList.add(columnData.getColumnNameKey());
042      }
043      
044      /**
045       * Adds <code>ColumnData</code> list. Throws <code>ClassCastException</code> if the element
046       * of the list is not <code>ColumnData</code> instance.
047       * @param columnDataList the <code>ColumnData</code> list.
048       */
049      public void addAll(List columnDataList) {
050        for (Iterator i = columnDataList.iterator(); i.hasNext();) {
051          add((ColumnData) i.next());
052        }
053      }
054      
055      /**
056       * Removes the element at the specified position in this list. 
057       * Shifts any subsequent elements to the left (subtracts one from their indices).
058       * @param index the index of the element to removed.
059       * @return the <code>ColumnData</code> instance to be removed.
060       */
061      public ColumnData remove(int index) {
062        String removedColumnNameKey = (String) this.allColumnNameKeyList.remove(index);
063        return (ColumnData) this.columnNameKeyMap.remove(removedColumnNameKey);
064      }
065      
066      /**
067       * Inserts the specified element at the specified position in this list. 
068       * Shifts the element currently at that position (if any) and any subsequent elements 
069       * to the right (adds one to their indices).
070       * @param index index at which the specified element is to be inserted.
071       * @param columnData column data to be inserted.
072       */
073      public void insert(int index, ColumnData columnData) {
074        this.allColumnNameKeyList.add(index, columnData.getColumnNameKey());
075        this.columnNameKeyMap.put(columnData.getColumnNameKey(), columnData);
076      }
077      
078      /**
079       * Gets the <code>ColumnData</code> instance from the <code>columnNameKey</code>.
080       * @param columnNameKey the column name key.
081       * @return the <code>ColumnData</code> instance.
082       */
083      public ColumnData get(String columnNameKey) {
084        return (ColumnData) this.columnNameKeyMap.get(columnNameKey);
085      }
086    
087      /**
088       * Gets the array of the <code>String</code> column header names.
089       *
090       * @return the array of the <code>String</code> column header names
091       */
092      public String[] getEnabledColumnNameArray() {
093        List columnNameList = new ArrayList();
094        for (Iterator i = this.allColumnNameKeyList.iterator(); i.hasNext();) {
095          String columnNameKey = (String) i.next();
096          ColumnData columnData = (ColumnData) this.columnNameKeyMap.get(columnNameKey);
097          if (columnData.isEnabled()) {
098            String columnName = ReviewI18n.getString(columnNameKey);
099            if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_LINK_ICON)) {
100              columnName = "";
101            }
102            columnNameList.add(columnName);
103          }
104        }
105        return (String[]) columnNameList.toArray(new String[] {});
106      }
107      
108      /**
109       * Gets the array of the <code>String</code> column header name keys.
110       *
111       * @return the array of the <code>String</code> column header name keys
112       */
113      public String[] getEnabledColumnNameKeyArray() {
114        List columnNameKeyList = new ArrayList();
115        for (Iterator i = this.allColumnNameKeyList.iterator(); i.hasNext();) {
116          String columnNameKey = (String) i.next();
117          ColumnData columnData = (ColumnData) this.columnNameKeyMap.get(columnNameKey);
118          if (columnData.isEnabled()) {
119            columnNameKeyList.add(columnNameKey);
120          }
121        }
122        return (String[]) columnNameKeyList.toArray(new String[] {});
123      }
124    
125      /**
126       * Gets the array of the <code>ColumnPixelData</code> instances
127       *
128       * @return the array of the <code>ColumnPixelData</code> instances
129       */
130      public ColumnPixelData[] getEnabledColumnPixelDataArray() {
131        List columnPixelDataList = new ArrayList();
132        for (Iterator i = this.allColumnNameKeyList.iterator(); i.hasNext();) {
133          String columnNameKey = (String) i.next();
134          ColumnData columnData = (ColumnData) this.columnNameKeyMap.get(columnNameKey);
135          if (columnData.isEnabled()) {
136            columnPixelDataList.add(columnData.getColumnPixelData());
137          }
138        }
139        return (ColumnPixelData[]) columnPixelDataList.toArray(new ColumnPixelData[] {});
140      }
141        
142      /**
143       * Gets the array of the enabled <code>ColumnData</code> instances.
144       * @return the array of the enabled <code>ColumnData</code> instances.
145       */
146      public ColumnData[] getEnabledColumnDataArray() {
147        List enabledColumnDataList = new ArrayList();
148        for (Iterator i = this.allColumnNameKeyList.iterator(); i.hasNext();) {
149          String columnNameKey = (String) i.next();
150          ColumnData columnData = (ColumnData) this.columnNameKeyMap.get(columnNameKey);
151          if (columnData.isEnabled()) {
152            enabledColumnDataList.add(columnData);
153          }
154        }
155        return (ColumnData[]) enabledColumnDataList.toArray(new ColumnData[] {});
156      }
157      
158      /**
159       * Gets the array of all <code>ColumnData</code> instances.
160       * @return the array of all <code>ColumnData</code> instances.
161       */
162      public ColumnData[] getAllColumnDataArray() {
163        List columnDataList = new ArrayList();
164        for (Iterator i = this.allColumnNameKeyList.iterator(); i.hasNext();) {
165          String columnNameKey = (String) i.next();
166          columnDataList.add((ColumnData) this.columnNameKeyMap.get(columnNameKey));
167        }
168        return (ColumnData[]) columnDataList.toArray(new ColumnData[] {});
169      }
170      
171      /**
172       * Clears all elements in this model. Note that review phase name is not cleared.
173       */
174      public void clear() {
175        this.columnNameKeyMap.clear();
176        this.allColumnNameKeyList.clear();
177      }
178      
179      /**
180       * Provides the size of the all elements.
181       *
182       * @return the size of the all elements.
183       */
184      public int allSize() {
185        return this.allColumnNameKeyList.size();
186      }
187    }
