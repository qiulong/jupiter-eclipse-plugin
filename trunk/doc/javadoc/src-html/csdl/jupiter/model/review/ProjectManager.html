001    package csdl.jupiter.model.review;
002    
003    import org.eclipse.core.resources.IProject;
004    
005    import csdl.jupiter.file.FileResource;
006    
007    /**
008     * Provides the singleton project manager.
009     * @author Takuya Yamashita
010     * @version $Id: ProjectManager.java,v 1.4 2005/03/28 14:39:31 takuyay Exp $
011     */
012    public class ProjectManager implements IReviewModelElementChangeListener {
013      private static ProjectManager theInstance = new ProjectManager();
014      private IProject project;
015      
016      /**
017       * Sets default project Prohibits clients from instantiating this.
018       * This project instance could be <code>null</code> if an active editor does not
019       * exist or <code>IFileEditorInput</code> instance does not exist (meaning the case that
020       * welcome page is opened, for example).
021       */
022      private ProjectManager() {
023        this.project = FileResource.getActiveProject();
024      }
025      
026      /**
027       * Gets the singleton instance.
028       * @return the singleton instance.
029       */
030      static ProjectManager getInstance() {
031        return theInstance;
032      }
033      
034      /**
035       * Sets the IProject if the notified object is the instance of the IProject.
036       * @param object The object to be notified.
037       */
038      public void elementChanged(Object object) {
039        if (object instanceof IProject) {
040          setProject((IProject) object);
041        }
042      }
043      
044      /**
045       * Sets the <code>IProject</code> instance.
046       * @param project the project.
047       */
048      private void setProject(IProject project) {
049        this.project = project;
050      }
051      
052      /**
053       * Gets the <code>IProject</code> instance. Clients should check <code>null</code> before
054       * using the <code>IProject</code> instance.
055       * @return the project.
056       */
057      public IProject getProject() {
058        return this.project;
059      }
060    }
