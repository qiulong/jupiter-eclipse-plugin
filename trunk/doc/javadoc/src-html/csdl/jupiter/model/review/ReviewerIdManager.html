001    package csdl.jupiter.model.review;
002    
003    /**
004     * Provides the singleton reviewer id manager class.
005     * @author Takuya Yamashita
006     * @version $Id: ReviewerIdManager.java,v 1.3 2005/03/28 14:39:31 takuyay Exp $
007     */
008    public class ReviewerIdManager implements IReviewModelElementChangeListener {
009      private static ReviewerIdManager theInstance = new ReviewerIdManager();
010      private ReviewerId reviewerId;
011      
012      /**
013       * Prohibits clients from instantiating this.
014       */
015      private ReviewerIdManager() {
016      }
017      
018      /**
019       * Gets the singleton instance.
020       * @return the singleton instance.
021       */
022      static ReviewerIdManager getInstance() {
023        return theInstance;
024      }
025      
026      /**
027       * Sets the ReviewerId if the object is the instance of the ReviewerId.
028       * @param object The object to be notified.
029       */
030      public void elementChanged(Object object) {
031        if (object instanceof ReviewerId) {
032          setReviewerId((ReviewerId) object);
033        }
034      }
035      
036      /**
037       * Sets the <code>ReviewerId</code> instance.
038       * @param reviewerId the <code>ReviewerId</code> instance.
039       */
040      private void setReviewerId(ReviewerId reviewerId) {
041        this.reviewerId = reviewerId;
042      }
043      
044      /**
045       * Gets the <code>ReviewerId</code> instance. Returns null if there is no <code>ReviewerId</code>
046       * instance is set.
047       * @return the <code>ReviewerId</code> instance.
048       */
049      public ReviewerId getReviewerId() {
050        return this.reviewerId;
051      }
052    }
