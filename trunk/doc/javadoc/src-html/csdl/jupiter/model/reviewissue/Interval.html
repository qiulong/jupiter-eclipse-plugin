001    package csdl.jupiter.model.reviewissue;
002    
003    /**
004     * Provides the review interval. Clients can instantiate this with a key.
005     *
006     * @author Takuya Yamashita
007     * @version $Id: Interval.java,v 1.1 2004/07/26 06:32:09 takuyay Exp $
008     */
009    public class Interval {
010      /** The key of the type. */
011      private String key;
012    
013      /**
014       * Instantiates this with the key of the type.
015       *
016       * @param key the key of the type.
017       */
018      public Interval(String key) {
019        this.key = key;
020      }
021    
022      /**
023       * Gets the key for the <code>Interval</code> instance.
024       *
025       * @return the key for the <code>Interval</code> instance.
026       */
027      public String getKey() {
028        return this.key;
029      }
030    }
