001    package csdl.jupiter.model.reviewissue;
002    
003    /**
004     * Provides severity information Clients can instantiate this with a key and its ordinal number.
005     * The ordinal number is used for the comparison of this instances.
006     *
007     * @author Takuya Yamashita
008     * @version $Id: Severity.java,v 1.2 2005/03/25 18:32:03 takuyay Exp $
009     */
010    public class Severity {
011      /** The key of the severity. */
012      private String key;
013      /** Assigns an ordinal to this key. */
014      private int ordinal;
015    
016      /**
017       * Constructor for the Severity object. Sets the key string and its ordinal number.
018       *
019       * @param key the severity of the code review.
020       * @param ordinal the ordinal of the key.
021       */
022      public Severity(String key, int ordinal) {
023        this.key = key;
024        this.ordinal = ordinal;
025      }
026    
027      /**
028       * Gets the key for the severity value.
029       *
030       * @return the key for the severity value.
031       */
032      public String getKey() {
033        return this.key;
034      }
035    
036      /**
037       * Compare this object by the ordinal number.
038       *
039       * @see java.lang.Comparable#compareTo(java.lang.Object)
040       */
041      public int compareTo(Object object) {
042        return ordinal - ((Severity) object).ordinal;
043      }
044    }
