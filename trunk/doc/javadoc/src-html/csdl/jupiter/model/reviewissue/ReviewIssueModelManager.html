001    package csdl.jupiter.model.reviewissue;
002    
003    import org.eclipse.core.resources.IFile;
004    import org.eclipse.core.resources.IProject;
005    
006    import csdl.jupiter.file.FileResource;
007    import csdl.jupiter.file.ReviewIssueXmlSerializer;
008    import csdl.jupiter.file.Ver1ReviewHelper;
009    import csdl.jupiter.model.review.ReviewId;
010    
011    /**
012     * Provides review issue model manager.
013     * @author Takuya Yamashita
014     * @version $Id: ReviewIssueModelManager.java,v 1.12 2005/03/25 18:32:03 takuyay Exp $
015     */
016    public class ReviewIssueModelManager {
017      private static ReviewIssueModelManager theInstance = new ReviewIssueModelManager();
018      private ReviewIssueModel model;
019      
020      /**
021       * Prohibits clients from instantiating this.
022       */
023      private ReviewIssueModelManager() {
024        this.model = new ReviewIssueModel();
025      }
026      
027      /**
028       * Gets the singleton instance.
029       * @return the singleton instance.
030       */
031      public static ReviewIssueModelManager getInstance() {
032        return theInstance;
033      }
034      
035      /**
036       * Gets the <code>ReviewIssueModel</code> instance. Note that 
037       * <code>IllegalArgumentException</code> will be thrown if either <code>IProject</code> or 
038       * <code>ReviewId</code> were null.
039       * 
040       * @param project the project.
041       * @param reviewId the review id.
042       * @return the <code>ReviewIssueModel</code> instance.
043       */
044      public ReviewIssueModel getModel(IProject project, ReviewId reviewId) {
045        if (project == null) {
046          throw new IllegalArgumentException("project is null");
047        }
048        if (reviewId == null) {
049          throw new IllegalArgumentException("reviewId is null");
050        }
051        this.model.clear();
052        IFile[] reviewIFiles = FileResource.getReviewIFiles(project, reviewId);
053        ReviewIssueXmlSerializer.read(reviewId, this.model, reviewIFiles);
054        // compatible to the version 1.
055        if (this.model.size() <= 0) {
056          Ver1ReviewHelper.read(reviewId, this.model, reviewIFiles);
057        }
058        this.model.sortByPreviousComparator();
059        this.model.clearComparator();
060        return this.model;
061      }
062      
063      /**
064       * Gets the current model.
065       * @return the current model.
066       */
067      public ReviewIssueModel getCurrentModel() {
068        return this.model;
069      }
070    }
