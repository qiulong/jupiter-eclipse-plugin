080       * @param type the type if it's defect
081       * @param severity the severity.

096        this.setReviewIFile(reviewIFile);
097        this.setCreationDate(creationDate);
098        this.setModificationDate(modificationDate);

101        this.setTargetFile(targetFile);
102        this.setLine(line);
103        this.setType(type);

106        this.setDescription(description);
107        this.setAnnotation(annotation);
108        this.setRevision(revision);
109        this.setResolution(resolution);
110        this.setStatus(status);
111        // Used for calculation if java hash code is changed.
112        this.previousJavaHashCode = getJavaHashCode();
113        this.currentJavaHashCode = getJavaHashCode();
114      }

118       * Gets the java hash code from <code>fullyQualifiedClass</code>, <code>method</code>, and
119       * <code>line</code>.
120       *
121       * @return the java hash code.
122       */
123      private int getJavaHashCode() {
124        int hash = 17;
125        hash = (37 * hash) + this.line.hashCode();
126        return hash;
127      }

416        if (status == null) {
417          ReviewModel reviewModel = ReviewModel.getInstance();
418          IProject project = reviewModel.getProjectManager().getProject();
419          ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
420          status = (Status) StatusKeyManager.getInstance(project, reviewId).getItem(0);
421        }
422        this.status = status;
423      }

510      /**
511       * Checks if the <code>ReviewIssue</code> instance is equal. Returns <code>true</code> if all the
512       * passing variables to this constructor are equal. Returns <code>false</code> otherwise.
513       *
514       * @see java.lang.Object#equals(java.lang.Object)
515       */
516      public boolean equals(Object object) {
517        if (this == object) {
518          return true;
519        }
520        if (object instanceof ReviewIssue) {
521          return (this.hashCode() == object.hashCode());
522        }
523        return false;
524      }

526      /**
527       * Gets hash code calculated by all the passing variables to this constructor so that the hash
528       * code will be different if one of the variables differs.
529       *
530       * @see java.lang.Object#hashCode()
531       */
532      public int hashCode() {
533        int hash = 17;
534        hash = (37 * hash) + ((this.reviewIFile != null) 
535                               ? this.reviewIFile.getFullPath().toString().hashCode() 
536                               : "".hashCode());
537        hash = (37 * hash) + this.form.format(creationDate).hashCode();
538        hash = (37 * hash) + this.form.format(modificationDate).hashCode();
539        hash = (37 * hash) + this.reviewer.hashCode();
540        hash = (37 * hash) + this.assignedTo.hashCode();
541        hash = (37 * hash) + this.targetFile.hashCode();
542        hash = (37 * hash) + this.line.hashCode();   
543        hash = (37 * hash) + this.type.toString().hashCode();
544        hash = (37 * hash) + this.severity.toString().hashCode();
545        hash = (37 * hash) + this.summary.hashCode();
546        hash = (37 * hash) + this.description.hashCode();
547        hash = (37 * hash) + this.annotation.hashCode();
548        hash = (37 * hash) + this.revision.hashCode();
549        hash = (37 * hash) + this.resolution.toString().hashCode();
550        hash = (37 * hash) + this.status.toString().hashCode();
551        hash = (37 * hash) + String.valueOf(this.isLinked).hashCode();
552        return hash;
553      }

555      /**
556       * Checks if the content of <code>ReviewIssue</code> instance is equal. Note that the content
557       * means the all field information which are shown in the UI. Returns <code>true</code> if 
558       * this content are equal to the passing object. Returns <code>false</code> otherwise.
559       * @param object the object to be compared.
560       * @return <code>true</code> if the content of <code>ReviewIssue</code> instance is equal.
561       */
562      public boolean contentEquals(Object object) {
563        if (object instanceof ReviewIssue) {
564          ReviewIssue reviewIssue = (ReviewIssue) object;
565          String fullPath = reviewIssue.getReviewIFile().getFullPath().toString();
566          // This normalization is needed because the delimiter of the String from JDOM 
567          // and String from eclipse Text is different. I'm not sure there is more better way.
568          String thisNormalizedDescription = new Text(this.description).getTextNormalize();
569          String normalizedDescription = new Text(reviewIssue.getDescription()).getTextNormalize();
570          if (this.reviewIFile.getFullPath().toString().equals(fullPath)
571              && this.reviewer.equals(reviewIssue.getReviewer())
572              && this.assignedTo.equals(reviewIssue.getAssignedTo())
573              && this.targetFile.equals(reviewIssue.getTargetFile())
574              && this.line.equals(reviewIssue.getLine())
575              && this.type.getKey().equals(reviewIssue.getType().getKey())
576              && this.severity.getKey().equals(reviewIssue.getSeverity().getKey())
577              && this.resolution.getKey().equals(reviewIssue.getResolution().getKey())
578              && this.status.getKey().equals(reviewIssue.getStatus().getKey())
579              && this.summary.equals(reviewIssue.getSummary())
580              && thisNormalizedDescription.equals(normalizedDescription)
581              && this.annotation.equals(reviewIssue.getAnnotation())
582              && this.revision.equals(reviewIssue.getRevision())) {
583            return true;
584          }
585        }
586        return false;
587      }
