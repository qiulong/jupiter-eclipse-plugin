001    package csdl.jupiter.model.reviewissue;
002    
003    import csdl.jupiter.ReviewTestCase;
004    import csdl.jupiter.util.ResourceBundleKey;
005    
006    /**
007     * Tests the Status class.
008     *
009     * @author Takuya Yamashita
010     * @version $Id: TestStatus.java,v 1.1 2004/07/26 06:32:08 takuyay Exp $
011     */
012    public class TestStatus extends ReviewTestCase {
013      /**
014       * Tests for toString() method. The return value should be equal to each string below.
015       */
016      public void testToString() {
017        String key = ResourceBundleKey.ITEM_KEY_STATUS_UNRESOLVED;
018        assertEquals("Testing the key of UNRESOVLED", key, unresolved.getKey());
019        key = ResourceBundleKey.ITEM_KEY_STATUS_RESOLVED;
020        assertEquals("Testing the key of RESOVLED", key, resolved.getKey());
021      }
022    
023      /**
024       * Tests for compareTo() method. Since unresolved &lt; resolved,
025       * <code>unresolved.compareTo(resolved)</code> should return <code>-1</code>,
026       * <code>resolved.compareTo(unresolved.compareTo)</code> should return <code>1</code>,
027       * <code>resolved.compareTo(resolved)</code> should return <code>0</code>.
028       * <code>unresolved.compareTo(unresolved)</code> should return <code>0</code>.
029       */
030      public void testCompareTo() {
031        assertEquals("Testing compareTo with unresolved and resolved", -1,
032          unresolved.compareTo(resolved));
033        assertEquals("Testing compareTo with resolved and unresolved", 1,
034          resolved.compareTo(unresolved));
035        assertEquals("Testing compareTo with resolved and resolved", 0,
036          resolved.compareTo(resolved));
037        assertEquals("Testing compareTo with unresolved and unresolved", 0,
038            unresolved.compareTo(unresolved));
039      }
040    }
