017    public class ReviewTestCase extends TestCase {
018      /** The defect <code>Type</code> instance. */
019      protected Type defect;
020      /** The externalIssue <code>Type</code> instance. */
021      protected Type externalIssue;
022      /** The question <code>Type</code> instance. */
023      protected Type question;
024      /** The praise <code>Type</code> instance. */
025      protected Type praise;
026      /** The critical <code>Severity</code> instance. */
027      protected Severity critical;
028      /** The major <code>Severity</code> instance. */
029      protected Severity major;
030      /** The normal <code>Severity</code> instance. */
031      protected Severity normal;
032      /** The minor <code>Severity</code> instance. */
033      protected Severity minor;
034      /** The trivial <code>Severity</code> instance. */
035      protected Severity trivial;
036      /** The unset <code>Resolution</code> instance. */
037      protected Resolution unset;
038      /** The validNeedsfixing <code>Resolution</code> instance. */
039      protected Resolution validNeedsfixing;
040      /** The validFixlater <code>Resolution</code> instance. */
041      protected Resolution validFixlater;
042      /** The validDuplicate <code>Resolution</code> instance. */
043      protected Resolution validDuplicate;
044      /** The validWontfix <code>Resolution</code> instance. */
045      protected Resolution validWontfix;
046      /** The invalidWontfix <code>Resolution</code> instance. */
047      protected Resolution invalidWontfix;
048      /** The unsureValidity <code>Resolution</code> instance. */
049      protected Resolution unsureValidity;
050      /** The unresolved <code>Status</code> instance. */

053      protected Status resolved;
054    
055      /**
056       * Sets up the necessary item type instances.
057       *
058       * @exception Exception if errors occur in implementing class.
059       */
060      protected void setUp() throws Exception {
061        this.defect = new Type(ResourceBundleKey.ITEM_KEY_TYPE_DEFECT, 0);
062        this.externalIssue = new Type(ResourceBundleKey.ITEM_KEY_TYPE_EXTERNAL_ISSUE, 1);
063        this.question = new Type(ResourceBundleKey.ITEM_KEY_TYPE_QUESTION, 2);
064        this.praise = new Type(ResourceBundleKey.ITEM_KEY_TYPE_PRAISE, 3);
065        this.critical = new Severity(ResourceBundleKey.ITEM_KEY_SEVERITY_CRITICAL, 0);
066        this.major = new Severity(ResourceBundleKey.ITEM_KEY_SEVERITY_MAJOR, 1);
067        this.normal = new Severity(ResourceBundleKey.ITEM_KEY_SEVERITY_NORMAL, 2);
068        this.minor = new Severity(ResourceBundleKey.ITEM_KEY_SEVERITY_MINOR, 3);
069        this.trivial = new Severity(ResourceBundleKey.ITEM_KEY_SEVERITY_TRIVIAL, 4);
070        String resolutionKey = ResourceBundleKey.ITEM_KEY_UNSET;
071        this.unset = new Resolution(resolutionKey, 0);
072        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_NEEDSFIXING;
073        this.validNeedsfixing = new Resolution(resolutionKey, 1);
074        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_FIXLATER;
075        this.validFixlater = new Resolution(resolutionKey, 2);
076        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_DUPLICATE;
077        this.validDuplicate = new Resolution(resolutionKey, 3);
078        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_WONTFIX;
079        this.validWontfix = new Resolution(resolutionKey, 4);
080        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_INVALID_WONTFIX;
081        this.invalidWontfix = new Resolution(resolutionKey, 5);
082        resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_UNSURE_VALIDITY;
083        this.unsureValidity = new Resolution(resolutionKey, 6);
084        this.unresolved = new Status(ResourceBundleKey.ITEM_KEY_STATUS_UNRESOLVED, 0);
085        this.resolved = new Status(ResourceBundleKey.ITEM_KEY_STATUS_RESOLVED, 1);
086      }
087      
088      /**
089       * Checks if the instantiated references are not null.
090       */
091      public void testNullInstances() {
092        assertNotNull("Checking the defect type instance.", this.defect);
093        assertNotNull("Checking the defect type instance.", this.externalIssue);
094        assertNotNull("Checking the defect type instance.", this.question);
095        assertNotNull("Checking the defect type instance.", this.praise);
096        assertNotNull("Checking the defect type instance.", this.critical);
097        assertNotNull("Checking the defect type instance.", this.major);
098        assertNotNull("Checking the defect type instance.", this.normal);
099        assertNotNull("Checking the defect type instance.", this.minor);
100        assertNotNull("Checking the defect type instance.", this.trivial);
101        assertNotNull("Checking the defect type instance.", this.unset);
102        assertNotNull("Checking the defect type instance.", this.validNeedsfixing);
103        assertNotNull("Checking the defect type instance.", this.validFixlater);
104        assertNotNull("Checking the defect type instance.", this.validDuplicate);
105        assertNotNull("Checking the defect type instance.", this.validWontfix);
106        assertNotNull("Checking the defect type instance.", this.unsureValidity);
107        assertNotNull("Checking the defect type instance.", this.validWontfix);
108        assertNotNull("Checking the defect type instance.", this.unresolved);
109        assertNotNull("Checking the defect type instance.", this.resolved);
110      }
