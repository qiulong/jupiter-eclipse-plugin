001    package csdl.jupiter;

003    import junit.framework.TestCase;
004    
005    /**
006     * Provides the test cases for the <code>CodeReviewException</code> class.
007     * @author Takuya Yamashita
008     * @version $Id: TestCodeReviewException.java,v 1.2 2004/07/26 06:32:02 takuyay Exp $
009     */
010    public class TestCodeReviewException extends TestCase {
011      /**
012       * Tests the <code>CodeReviewException</code> constructor.
013       */
014      public void testCodeReviewException() {
015        ReviewException codeReviewException = new ReviewException("one parameter");
016        assertEquals("Testing one parameter constructor: message", "one parameter",
017                                                          codeReviewException.getMessage());
018        codeReviewException = new ReviewException("two parameter", new Exception("cause instance"));
019        assertEquals("Testing two parameter constructor: message", "two parameter", 
020                                                          codeReviewException.getMessage());
021        assertEquals("Testing two parameter constructor: cause message", "cause instance", 
022            codeReviewException.getCause().getMessage());
023      }
