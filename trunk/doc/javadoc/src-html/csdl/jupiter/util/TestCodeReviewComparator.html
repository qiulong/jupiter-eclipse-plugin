012    /**
013     * Tests the <code>ReviewComparator</code> insntace.
014     *
015     * @author Takuya Yamashita

017     */
018    public class TestCodeReviewComparator extends ReviewTestCase {
019      /** The <code>List</code> interface to hold <code>ReviewIssue</code> instance. */
020      List codeReviews = new ArrayList();
021    
022      /**
023       * Sets up the initial condition. Instantiates three <code>ReviewIssue</code> instances, then adds
024       * them to the List interface.
025       * 
026       * @throws Exception if problems occur.
027       */
028      protected void setUp() throws Exception {
029        super.setUp();
030        codeReviews.add(new ReviewIssue(new Date(), new Date(), "fooReviewer", "fooRespondent",
031            "doc/review/foo/bar/Baz.java", "10", praise, major,
032            "This is summary.", "This is nice method!", "This is annotation for review.",
033            "This is revision comment.", validFixlater, unresolved, null));
034        codeReviews.add(new ReviewIssue(new Date(), new Date(), "barReviewer", "barRespondent",
035            "doc/review/bar/baz/Foo.java", "20", defect, minor,
036            "This is summary.", "Should throw Exception.", "This is annotation for review.",
037            "This is revision comment.", validNeedsfixing, resolved, null));
038        codeReviews.add(new ReviewIssue(new Date(), new Date(), "bazReviewer", "bazRespondent",
039            "doc/review/baz/foo/Bar.java", "30", question, trivial,
040            "This is summary", "is this method needed?", "This is annotation for review.",
041            "This is revision comment.", unset, unresolved, null));
042      }
043    
044      /**
045       * Tests <code>ReviewComparator.REVIEWER</code> instance.
046       */
047      public void testReviewNameComparator() {
048        Collections.sort(codeReviews, ReviewComparator.REVIEWER);
049        Iterator i = codeReviews.iterator();
050        i.hasNext();
051        assertEquals("Testing first element of the reviewer's name.", "barReviewer",
052          ((ReviewIssue) i.next()).getReviewer());
053        i.hasNext();
054        assertEquals("Testing second element of the reviewer's name.", "bazReviewer",
055          ((ReviewIssue) i.next()).getReviewer());
056        i.hasNext();
057        assertEquals("Testing third element of the reviewer's name.", "fooReviewer",
058          ((ReviewIssue) i.next()).getReviewer());
059      }
060    
061      /**
062       * Tests <code>ReviewComparator.TYPE</code> instance.
063       */
064      public void testTypeComparator() {
065        Collections.sort(codeReviews, ReviewComparator.TYPE);
066        Iterator i = codeReviews.iterator();
067        i.hasNext();
068        assertEquals("Testing first element of the type.", defect,
069          ((ReviewIssue) i.next()).getType());
070        i.hasNext();
071        assertEquals("Testing second element of the type.", question,
072          ((ReviewIssue) i.next()).getType());
073        i.hasNext();
074        assertEquals("Testing third element of the type.", praise,
075          ((ReviewIssue) i.next()).getType());
076      }
