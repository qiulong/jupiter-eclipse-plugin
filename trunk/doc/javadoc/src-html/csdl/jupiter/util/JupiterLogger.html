001    package csdl.jupiter.util;
002    
003    import java.util.logging.Level;
004    import java.util.logging.Logger;
005    
006    import org.eclipse.core.runtime.IStatus;
007    import org.eclipse.core.runtime.Status;
008    
009    import csdl.jupiter.ReviewPlugin;
010    
011    /**
012     * Provides the logging error function in Eclipse. Facilitates the logging in source codes. Clients
013     * can use something like:
014     * <pre>
015     * Exception exception = ...;
016     * Logger.log(Logger.ERROR, exception.getMessage(), exception);
017     * </pre>
018     *
019     * @author Takuya Yamashita
020     * @version $Id: Logger.java,v 1.7 2005/03/25 18:32:07 takuyay Exp $
021     */
022    public class JupiterLogger {
023      /** Singleton logger object. */
024      private static JupiterLogger theInstance = new JupiterLogger();
025      
026      /** Console log for debug. */
027      private Logger consoleLog;
028      
029      /**
030       * Construct an empty logger object.
031       *
032       */
033      private JupiterLogger() {
034        this.consoleLog = Logger.getLogger("csdl.jupiter");
035        
036        String levelName = System.getProperty("LogLevel", Level.WARNING.getName()); 
037        this.consoleLog.setLevel(Level.parse(levelName));
038      }
039      
040      /**
041       * Gets singleton logger object. 
042       * 
043       * @return Logger object.
044       */
045      public static JupiterLogger getLogger() {
046        return theInstance;
047      }
048    
049      /**
050       * Debugger message.
051       *
052       * @param message Error message.
053       */
054      public void debug(String message) {
055        this.consoleLog.info(message);
056      }
057    
058      
059      /**
060       * Logs an warning information with exception into Eclipse log.
061       *
062       * @param message Error message.
063       * @param e The exception is thrown.
064       */
065      public void warning(String message, Exception e) {
066        this.log(IStatus.WARNING, message, e);
067      }
068    
069      /**
070       * Logs an warning information with exception into Eclipse log.
071       *
072       * @param message Error message.
073       */
074      public void warning(String message) {
075        this.log(IStatus.WARNING, message, null);
076      }
077    
078      /**
079       * Logs an information into Eclipse log.
080       *
081       * @param message Error message.
082       */
083      public void info(String message) {
084        this.log(IStatus.INFO, message, null);
085      }
086    
087      /**
088       * Logs exception into eclipse error log
089       *
090       * @param e The exception is thrown.
091       */
092      public void error(Exception e) {
093        this.log(IStatus.ERROR, "", e);
094      }
095      
096      /**
097       * Logs an error message and exception into eclipse error log
098       *
099       * @param message Error message.
100       * @param e The exception is thrown.
101       */
102      public void error(String message, Exception e) {
103        this.log(IStatus.ERROR, message, e);
104      }
105      
106      /**
107       * Logs an error message into eclipse error log. Severity should be one of the type of
108       * <code>org.eclipse.core.runtime.IStatus</code>. For example, for the error, the severity
109       * should be <code>IStatus.ERROR</code>. Message summary parameter might be the message, which
110       * is the  <code>Exception.getMessage()</code> value.
111       *
112       * @param severity the severity type represented by the static field value of this
113       *        <code>Logger</code>
114       * @param messageSummary the message summary.
115       * @param exception the exception to be thrown.
116       */
117      private void log(int severity, String messageSummary, Exception exception) {
118        ReviewPlugin plugin = ReviewPlugin.getInstance();
119        IStatus status = new Status(severity, plugin.getBundle().getSymbolicName(), IStatus.OK,
120            messageSummary, exception);
121        plugin.getLog().log(status);
122      }
123    }
