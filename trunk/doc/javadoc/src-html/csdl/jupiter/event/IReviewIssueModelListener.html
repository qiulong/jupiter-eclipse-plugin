001    package csdl.jupiter.event;
002    
003    
004    /**
005     * Provides an interface for the ReviewIssue model change listener. 
006     * The <code>EventIssueModelEvent</code> will be notified when ReviewIssueModel is
007     * changed in such a way that a new ReviewIssue is added, an existing ReviewIssue is deleted,
008     * and so forth. See the <code>ReviewIssueModelEvent</code> class for all the possible event.
009     * Clients can add an implementing sub class into the <code>ReviewIssueModel</code>.
010     *
011     * @author Takuya Yamashita
012     * @version $Id: IReviewIssueModelListener.java,v 1.3 2005/03/25 18:32:02 takuyay Exp $
013     */
014    public interface IReviewIssueModelListener {
015      /**
016       * Called when a notifier notifies to this listener. <code>ReviewIssueModelEvent</code>
017       * instance contains the event type such as <code>ReviewIssueModelEvent.ADD</code>, etc.
018       * @param event the codeReview event to be notified.
019       */
020      void reviewIssueModelChanged(ReviewIssueModelEvent event);
021    }
