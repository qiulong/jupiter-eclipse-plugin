001    package csdl.jupiter.event;
002    
003    /**
004     * Provides an interface for the Review event listener. 
005     * The <code>ReviewEvent</code> will be notified when Review event is invoked in such a way
006     * that review command (for example, go to button is clicked) is triggered. 
007     * @author Takuya Yamashita
008     * @version $Id: IReviewListener.java,v 1.3 2005/03/25 18:32:02 takuyay Exp $
009     */
010    public interface IReviewListener {
011      /**
012       * Called when a notifier notifies to this listener. <code>ReviewEvent</code> instance might
013       * be ant instance of the command.
014       * @param event the review event to be notified.
015       */
016      void reviewInvoked(ReviewEvent event);
017    }
