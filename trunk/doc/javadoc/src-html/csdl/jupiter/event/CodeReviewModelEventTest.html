001    package csdl.jupiter.event;
002    
003    import junit.framework.TestCase;
004    import csdl.jupiter.ReviewPlugin;
005    
006    /**
007     * Provides test cases for <code>CodeReviewModeEvent</code>.
008     * @author Takuya Yamashita
009     * @version $Id: TestCodeReviewModelEvent.java,v 1.4 2005/03/18 09:30:48 takuyay Exp $
010     */
011    public class CodeReviewModelEventTest extends TestCase {
012      /**
013       * Initializes plugin for test. 
014       */
015      protected void setUp() {
016        ReviewPlugin plugin = new ReviewPlugin();
017      }
018      
019      /**
020       * Tests the <code>getEventType</code> method.
021       */
022      public void testGetEventType() {
023        int mergeType = ReviewIssueModelEvent.MERGE;
024        int addType = ReviewIssueModelEvent.ADD;
025        int deleteType = ReviewIssueModelEvent.DELETE;    
026        
027        ReviewIssueModelEvent event = new ReviewIssueModelEvent(null, mergeType | addType);
028        int eventType = event.getEventType();
029        assertEquals("Checking merge type. ", true, (eventType & mergeType) != 0);
030        assertEquals("Checking addType type. ", true, (eventType & addType) != 0);
031        assertEquals("Checking deleteType type. ", false, (eventType & deleteType) != 0);    
032        assertEquals("Checking mergeType and addType type. ", true, 
033                    (eventType & (mergeType | addType)) != 0);
034      }
035    }
