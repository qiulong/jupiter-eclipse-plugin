001    package csdl.jupiter.event;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    import org.eclipse.core.resources.IProject;
007    
008    import csdl.jupiter.model.review.ReviewId;
009    import csdl.jupiter.model.review.ReviewModel;
010    import csdl.jupiter.model.reviewissue.ResolutionKeyManager;
011    import csdl.jupiter.model.reviewissue.ReviewIssue;
012    import csdl.jupiter.model.reviewissue.SeverityKeyManager;
013    import csdl.jupiter.model.reviewissue.StatusKeyManager;
014    import csdl.jupiter.model.reviewissue.TypeKeyManager;
015    
016    /**
017     * Provides review issue model event. This wraps review issue entry with an event type. This
018     * is used for the argument of the <code>reviewIssueModelChanged</code> in the
019     * <code>IReviewIssueModelListener</code> interface.
020     * 
021     * @author Takuya Yamashita
022     * @version $Id: ReviewIssueModelEvent.java,v 1.9 2005/03/28 14:39:31 takuyay Exp $
023     */
024    public class ReviewIssueModelEvent {
025      // note that a new type should be the sequence of the power 2. next should be 2^4.
026      /** The merge type to merge review issues in the model. */
027      public static final int MERGE = 1;
028      /** The save type to save a review issue in the model. */
029      public static final int ADD = 2;
030      /** The delete type to delete a review issue in the model. */
031      public static final int DELETE = 4;
032      /** The edit type to edit a review issue in the model. */
033      public static final int EDIT = 8;
034      /** The clear type to clear all review issues in the model. */
035      public static final int CLEAR = 16;
036      /** Type category key */
037      public static final String TYPE = "type";
038      /** Severity category key */
039      public static final String SEVERITY = "severity";
040      /** Resolution category key */
041      public static final String RESOLUTION = "resolution";
042      /** Status category key */
043      public static final String STATUS = "status";
044      /** The ReviewModel instance */
045      private ReviewModel reviewModel;
046      /** The ReviewIssue instance */
047      private ReviewIssue reviewIssue;
048      /** The event type */
049      private int eventType;
050      /** String category key -> the String array of the item value.*/
051      private Map categoryToItems;
052      /**
053       * Instantiates this event with <code>ReviewIssue</code> instance and event type.
054       * @param reviewIssue the <code>ReviewIssue</code> instance.
055       * @param eventType the event type.
056       */
057      public ReviewIssueModelEvent(ReviewIssue reviewIssue, int eventType) {
058        this.reviewModel = ReviewModel.getInstance();
059        this.reviewIssue = reviewIssue;
060        this.eventType = eventType;
061        this.categoryToItems = new HashMap();
062        IProject project = this.reviewModel.getProjectManager().getProject();
063        ReviewId reviewId = this.reviewModel.getReviewIdManager().getReviewId();
064        if (project == null || reviewId == null) {
065          return;
066        }
067        String[] types = TypeKeyManager.getInstance(project, reviewId).getElements();
068        this.categoryToItems.put(TYPE, types);
069        String[] severities = SeverityKeyManager.getInstance(project, reviewId).getElements();
070        this.categoryToItems.put(SEVERITY, severities);
071        String[] resolutions = ResolutionKeyManager.getInstance(project, reviewId).getElements();
072        this.categoryToItems.put(RESOLUTION, resolutions);
073        String[] status = StatusKeyManager.getInstance(project, reviewId).getElements();
074        this.categoryToItems.put(STATUS, status);
075      }
076    
077      /**
078       * Gets the <code>ReviewModel</code> instance.
079       * @return the <code>ReviewModel</code> instance.
080       */
081      public ReviewModel getReviewModel() {
082        return this.reviewModel;
083      }
084      
085      /**
086       * Gets the <code>ReviewIssue</code> instance.
087       * 
088       * @return the <code>ReviewIssue</code> instance.
089       */
090      public ReviewIssue getReviewIssue() {
091        return this.reviewIssue;
092      }
093    
094      /**
095       * Gets the event type.
096       * 
097       * @return the event type.
098       */
099      public int getEventType() {
100        return this.eventType;
101      }
102      
103      /**
104       * Gets the map of the String category key -> the String ordinal array of 
105       * the item value.
106       * For example, 
107       * <pre>
108       * String[] types = (String[]) event.getCategories.get(ReviewIssueModelEvent.TYPE);
109       * </pre>
110       * @return the map of the String category key -> the String array of the item value.
111       */
112      public Map getCategories() {
113        return this.categoryToItems;
114      }
115    }
