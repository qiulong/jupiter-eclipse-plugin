015     */
016    public class ReviewConversation {
017      protected ProjectMockup fooProjectMockup;
018      
019      protected ProjectMockup barProjectMockup;
020      
021      protected ProjectMockup bazProjectMockup;
022      
023      protected static final String REVIEW_DIR = "review";
024      protected static final String FOO_REVIEW = "foo.review";
025      protected static final String BAR_REVIEW = "bar.review";
026      protected static final String BAZ_REVIEW = "baz.review";
027      protected static final String FOO_REVIEW_PATH = REVIEW_DIR + "/" +  FOO_REVIEW;
028      protected static final String BAR_REVIEW_PATH = REVIEW_DIR + "/" +  BAR_REVIEW;
029      protected static final String BAZ_REVIEW_PATH = REVIEW_DIR + "/" +  BAZ_REVIEW;
030      protected static final String FOO_PROJECT = "fooProject";
031      protected static final String BAR_PROJECT = "barProject";  
032      protected static final String BAZ_PROJECT = "bazProject"; 
033      
034    
035      /**
036       * Sets up 
037       * @throws CoreException if problems occur.
038       */
039      public ReviewConversation() throws CoreException {
040        ReviewPlugin plugin = ReviewPlugin.getInstance();
041        IWorkbenchWindow[] windows = plugin.getWorkbench().getWorkbenchWindows();
042        this.fooProjectMockup = new ProjectMockup(FOO_PROJECT, "bin");
043        this.barProjectMockup = new ProjectMockup(BAR_PROJECT, "bin");
044        this.bazProjectMockup = new ProjectMockup(BAZ_PROJECT, "bin");
045        IProject fooProject = this.fooProjectMockup.getProject();
046        IProject barProject = this.barProjectMockup.getProject();
047        IProject bazProject = this.bazProjectMockup.getProject();
048        this.fooProjectMockup.copy(FOO_REVIEW, FOO_REVIEW_PATH);
049        this.barProjectMockup.copy(BAR_REVIEW, BAR_REVIEW_PATH);
050        IFile fooReviewFile = fooProject.getFile(FOO_REVIEW_PATH);
051        IFile barReviewFile = barProject.getFile(BAR_REVIEW_PATH);
052      }
053    }
