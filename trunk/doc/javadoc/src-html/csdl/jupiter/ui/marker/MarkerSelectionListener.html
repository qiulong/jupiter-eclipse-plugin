001    package csdl.jupiter.ui.marker;
002    
003    import org.eclipse.jface.text.ITextSelection;
004    import org.eclipse.jface.viewers.ISelection;
005    import org.eclipse.swt.widgets.Table;
006    import org.eclipse.swt.widgets.TableItem;
007    import org.eclipse.ui.ISelectionListener;
008    import org.eclipse.ui.IWorkbenchPart;
009    
010    import csdl.jupiter.model.reviewissue.ReviewIssue;
011    import csdl.jupiter.ui.view.table.ReviewTableView;
012    import csdl.jupiter.ui.view.table.ReviewTableViewAction;
013    
014    /**
015     * Provides the marker selection listener. When a marker is selected, selects the review issue
016     * located in the marker line number, then run NOTIFY_EDITOR to show the issue information in the
017     * review editor view.
018     *
019     * @author Takuya Yamashita
020     * @version $Id: MarkerSelectionListener.java,v 1.2 2005/02/05 04:37:36 takuyay Exp $
021     */
022    public class MarkerSelectionListener implements ISelectionListener {
023      
024      /**
025       * @see org.eclipse.ui.ISelectionListener
026       * #selectionChanged(org.eclipse.ui.IWorkbenchPart, org.eclipse.jface.viewers.ISelection)
027       */
028      public void selectionChanged(IWorkbenchPart part, ISelection selection) {
029        if (selection instanceof ITextSelection) {
030          String lineNumber = String.valueOf(((ITextSelection) selection).getStartLine() + 1);
031          ReviewTableView[] tableViews = ReviewTableView.getViews();
032          for (int i = 0; i < tableViews.length; i++) {
033            ReviewTableView tableView = tableViews[i];
034            if (tableView != null) {
035              Table table = tableView.getTable();
036              TableItem[] items = table.getItems();
037              for (int j = 0; j < items.length; j++) {
038                ReviewIssue reviewIssue = (ReviewIssue) items[j].getData();
039                if (lineNumber.equals(reviewIssue.getLine())) {
040                  table.select(j);
041                  table.setTopIndex(j);
042                  break;
043                }
044              }
045              ReviewTableViewAction.NOTIFY_EDITOR.run();
046            }
047          }
048        }
049      }
050    }
