037    public class ReviewMarker {
038      /** Jupiter logger */
039      private static JupiterLogger log = JupiterLogger.getLogger();
040    
041      /** The review marker type. */
042      public static final String REVIEW_MARKER = "csdl.jupiter.reviewmarker";
043      /** The review issue id attribute. */
044      public static final String ATTRIBUTE_REVIEW_ISSUE = "ReviewIssue";
045      /**
046       * Adds marker.
047       * 
048       * @param reviewIssue the <code>ReviewIssue</code> instance.
049       */
050      public static void addMarker(ReviewIssue reviewIssue) {
051        ReviewModel reviewModel = ReviewModel.getInstance();
052        IProject project = reviewModel.getProjectManager().getProject();
053        if (project != null && !reviewIssue.getTargetFile().equals("")) {
054          Map attributes = getInitialAttributes(reviewIssue);
055          try {
056            createMarker(project.getFile(reviewIssue.getTargetFile()), attributes);
057            log.debug("Created marker in the add marker.");
058          }
059          catch (CoreException e) {
060            log.error("Error when adding the review marker: ", e);
061          }
062        }
063      }
064    
065      /**
066       * Removes the marker associated with the <code>ReviewIssue</code> instance.
067       * 
068       * @param reviewIssue the <code>ReviewIssue</code> instance.
069       */
070      public static void removeMarker(ReviewIssue reviewIssue) {
071        ReviewModel reviewModel = ReviewModel.getInstance();
072        IProject project = reviewModel.getProjectManager().getProject();
073        String targetFile = reviewIssue.getTargetFile();
074        String issueId = reviewIssue.getIssueId();
075        if (project != null && !targetFile.equals("")) {
076          IFile targetIFile = project.getFile(targetFile);
077          try {
078            IMarker[] markers = targetIFile.findMarkers(REVIEW_MARKER, true, IResource.DEPTH_ZERO);
079            for (int i = 0; i < markers.length; i++) {
080              IMarker marker = markers[i];
081              String reviewIssueKey = ReviewMarker.ATTRIBUTE_REVIEW_ISSUE;
082              ReviewIssue reviewIssueInMarker = (ReviewIssue) marker.getAttribute(reviewIssueKey);
083              if (reviewIssueInMarker != null && reviewIssueInMarker.getIssueId().equals(issueId)) {
084                marker.delete();
085              }
086            }
087          }
088          catch (CoreException e) {
089            log.error("Error when removing the review marker: ", e);
090          }
091        }
092      }
093    
094      /**
095       * Updates the marker associated with the <code>ReviewIssue</code> instance.
096       * 
097       * @param reviewIssue the <code>ReviewIssue</code> instance.
098       */
099      public static void updateMarker(ReviewIssue reviewIssue) {
100        if (reviewIssue != null) {
101          ReviewModel reviewModel = ReviewModel.getInstance();
102          IProject project = reviewModel.getProjectManager().getProject();
103          String targetFile = reviewIssue.getTargetFile();
104          if (project != null && !targetFile.equals("")) {
105            IFile targetIFile = project.getFile(targetFile);
106            try {
107              IMarker[] markers = targetIFile.findMarkers(REVIEW_MARKER, true, IResource.DEPTH_ZERO);
108              for (int i = 0; i < markers.length; i++) {
109                IMarker marker = markers[i];
110                String reviewIssueKey = ATTRIBUTE_REVIEW_ISSUE;
111                ReviewIssue reviewIssueInMarker = (ReviewIssue) marker.getAttribute(reviewIssueKey);
112                if (reviewIssueInMarker != null
113                    && reviewIssueInMarker.getIssueId().equals(reviewIssue.getIssueId())) {
114                  marker.setAttribute(IMarker.MESSAGE, reviewIssue.getSummary());
115                }
116              }
117            }
118            catch (CoreException e) {
119              log.error("Error when updating (adding) the review marker: ", e);
120            }
121          }
122        }
123      }
124    
125      /**
126       * Updates all markers in a review ID.
127       */
128      public static void updateAllMarkersInReviewId() {
129        log.debug("updateAllMarkersInReviewId are called.");
130        ReviewModel reviewModel = ReviewModel.getInstance();
131        IProject project = reviewModel.getProjectManager().getProject();
132        if (project != null) {
133          clearMarkersInReviewId();
134          ReviewTableView view = ReviewTableView.getActiveView();
135          if (view == null) {
136            return;
137          }
138          TableItem[] items = view.getViewer().getTable().getItems();
139          for (int i = 0; i < items.length; i++) {
140            ReviewIssue reviewIssue = (ReviewIssue) items[i].getData();
141            if (!reviewIssue.getTargetFile().equals("")) {
142              Map attributes = getInitialAttributes(reviewIssue);
143              try {
144                createMarker(project.getFile(reviewIssue.getTargetFile()), attributes);
145                log.debug("Created marker in the update marker.");
146              }
147              catch (CoreException e) {
148                log.error("Error when updating (adding) the review markers: ", e);
149              }
150            }
151          }
152        }
153      }
154      
155      /**
156       * Updates markers in the file. What happens is that it cleans all markers for the file
157       * and re-creates new markers for the file. 
158       * @param file the file that contains updating markers.
159       */
160      public static void updateMarkers(IFile file) {
161        ReviewModel reviewModel = ReviewModel.getInstance();
162        IProject project = reviewModel.getProjectManager().getProject();
163        if (project != null) {
164          String fileFullPath = file.getFullPath().toString();
165          clearMarkers(file);
166          ReviewTableView view = ReviewTableView.getActiveView();
167          if (view == null) {
168            return;
169          }
170          TableItem[] items = view.getViewer().getTable().getItems();
171          for (int i = 0; i < items.length; i++) {
172            ReviewIssue reviewIssue = (ReviewIssue) items[i].getData();
173            if (!reviewIssue.getTargetFile().equals("")) {
174              Map attributes = getInitialAttributes(reviewIssue);
175              try {
176                IFile targetFileInReviewIssue = project.getFile(reviewIssue.getTargetFile());
177                String targetFileFullPath = targetFileInReviewIssue.getFullPath().toString();
178                if (targetFileFullPath.equals(fileFullPath)) {
179                  createMarker(file, attributes);
180                }
181              }
182              catch (CoreException e) {
183                log.error("Error when updating (adding) the review markers: ", e);
184              }
185            }
186          }
187        }
188      }

190      /**
191       * Creates a marker on the given resource with the given type and attributes.
192       * <p>
193       * This method modifies the workspace (progress is not reported to the user).
194       * </p>
195       * 
196       * @param resource the resource
197       * @param attributes the attribute map (key type: <code>String</code>, value type:
198       *        <code>Object</code>)
199       * @throws CoreException if this method fails
200       * @see IResource#createMarker(java.lang.String)
201       */
202      private static void createMarker(final IResource resource, final Map attributes) 
203      throws CoreException {
204        IWorkspaceRunnable workbenchRunnable = new IWorkspaceRunnable() {
205          public void run(IProgressMonitor monitor) throws CoreException {
206            IMarker marker = resource.createMarker(REVIEW_MARKER);
207            marker.setAttributes(attributes);
208          }
209        };
210        resource.getWorkspace().run(workbenchRunnable, null, IWorkspace.AVOID_UPDATE, null);
211      }

213      /**
214       * Returns the attributes with which a newly created marker will be initialized.
215       * 
216       * @param reviewIssue the <code>ReviewIssue</code> instance.
217       * @return the initial marker attributes
218       */
219      private static Map getInitialAttributes(ReviewIssue reviewIssue) {
220        Map attributes = new HashMap(11);
221        if (reviewIssue != null) {
222          int line = 0;
223          try {
224            line = Integer.parseInt(reviewIssue.getLine());
225          }
226          catch (NumberFormatException e) {
227            // ignore.
228          }
229          ReviewModel reviewModel = ReviewModel.getInstance();
230          IProject project = reviewModel.getProjectManager().getProject();
231          IFile targetFile = project.getFile(reviewIssue.getTargetFile());
232          IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
233          final IWorkbenchPage page = workbench.getActiveWorkbenchWindow().getActivePage();
234          IFileEditorInput fileEditorInput = new FileEditorInput(targetFile);
235          IEditorPart editorPart = page.findEditor(fileEditorInput);
236          if (editorPart != null && editorPart instanceof ITextEditor) {
237            ITextEditor textEditor = (ITextEditor) editorPart;
238            IDocument document = textEditor.getDocumentProvider().getDocument(fileEditorInput);
239            int start = -1;
240            int end = -1;
241            int length = 0;
242            try {
243              IRegion lineInformation = document.getLineInformation(line - 1);
244              start = lineInformation.getOffset();
245              length = lineInformation.getLength();
246              end = start + length;
247      
248            }
249            catch (BadLocationException e) {
250              // ignore.
251            }
252      
253            // marker line numbers are 1-based
254            attributes.put(IMarker.LINE_NUMBER, new Integer(line));
255            attributes.put(IMarker.CHAR_START, new Integer(start));
256            attributes.put(IMarker.CHAR_END, new Integer(start));
257            String summary = reviewIssue.getSummary() + " [" + reviewIssue.getReviewer() + "]";
258            attributes.put(IMarker.MESSAGE, summary);
259            attributes.put(ATTRIBUTE_REVIEW_ISSUE, reviewIssue);
260          }
261        }
262        return attributes;
263      }

265      /**
266       * Clears all markers in the current review ID.
267       */
268      public static void clearMarkersInReviewId() {
269        ReviewModel reviewModel = ReviewModel.getInstance();
270        IProject project = reviewModel.getProjectManager().getProject();
271        if (project != null) {
272          try {
273            log.debug("clearing all markers in the review id...");
274            IMarker[] markers = project.findMarkers(REVIEW_MARKER, true, IResource.DEPTH_INFINITE);
275            for (int i = 0; i < markers.length; i++) {
276              markers[i].delete();
277            }
278          }
279          catch (CoreException e) {
280            log.debug(e.getMessage());
281          }
282        }
283      }
284      
285      /**
286       * Clears markers in the file.
287       * @param file the file that contains the deleting markers.
288       */
289      public static void clearMarkers(IFile file) {
290        try {
291          log.debug("clearing markers in a file...");
292          IMarker[] markers = file.findMarkers(REVIEW_MARKER, true, IResource.DEPTH_ZERO);
293          for (int i = 0; i < markers.length; i++) {
294            markers[i].delete();
295          }
296        }
297        catch (CoreException e) {
298          log.debug(e.getMessage());
299        }
300      }
