001    package csdl.jupiter.ui.menu;
002    
003    import java.util.Iterator;
004    import java.util.Set;
005    
006    import org.eclipse.core.resources.IFile;
007    import org.eclipse.core.resources.IProject;
008    import org.eclipse.swt.SWT;
009    import org.eclipse.swt.events.SelectionAdapter;
010    import org.eclipse.swt.events.SelectionEvent;
011    import org.eclipse.swt.widgets.Menu;
012    import org.eclipse.swt.widgets.MenuItem;
013    
014    import csdl.jupiter.ReviewI18n;
015    import csdl.jupiter.file.FileResource;
016    import csdl.jupiter.file.PropertyResource;
017    import csdl.jupiter.file.ReviewResource;
018    import csdl.jupiter.model.review.ReviewId;
019    import csdl.jupiter.model.review.ReviewModel;
020    
021    /**
022     * Provides review file selection menu.
023     * @author Takuya Yamashita
024     * @version $Id: ReviewFileSelectionMenu.java,v 1.8 2005/05/06 05:22:47 takuyay Exp $
025     */
026    public class ReviewFileSelectionMenu {
027    
028      /**
029       * Creates file selection pulldown menu.
030       * @param menu the menu of the parent.
031       * @return the menu of the pulldown menu.
032       */
033      public static Menu createPulldownMenu(Menu menu) {
034        create(menu);
035        return menu;
036      }
037      
038      /**
039       * Creates target file selection menu.
040       * @param menu the menu of the parent.
041       */
042      private static void create(Menu menu) {
043        ReviewModel reviewModel = ReviewModel.getInstance();
044        final IProject project = reviewModel.getProjectManager().getProject();
045        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId(); 
046        if (project != null && reviewId != null) {
047          PropertyResource propertyResource = PropertyResource.getInstance(project, false);
048          String reviewIdString = reviewId.getReviewId();
049          ReviewResource reviewResource = propertyResource.getReviewResource(reviewIdString, true);
050          
051          // Get current editor file to highlight it if it is in the list of
052          // our review files
053          IFile activeFile = FileResource.getActiveFile();
054          String activeFileName = null;
055          if (activeFile != null) {
056            activeFileName = activeFile.getProjectRelativePath().toString();        
057          }
058    
059          if (reviewResource != null) {
060            Set targetFileSet = reviewResource.getFileSet();
061            int index = 0;
062            if (targetFileSet.size() > 0) {
063              for (Iterator i = targetFileSet.iterator(); i.hasNext(); index++) {
064                MenuItem menuItem = new MenuItem(menu, SWT.NONE);
065                String targetFile = (String) i.next();
066                menuItem.setText((index + 1) + " " + targetFile);
067                menuItem.setData(targetFile);
068                if (targetFile.equalsIgnoreCase(activeFileName)) {
069                  // Highlight current editor file
070                  menu.setDefaultItem(menuItem);
071                }
072    
073                menuItem.addSelectionListener(new SelectionAdapter() {
074                  public void widgetSelected(SelectionEvent event) {
075                    String targetFile = (String) event.widget.getData();
076                    if (!targetFile.equals("")) {
077                      IFile targetIFile = project.getFile(targetFile);
078                      int lineNumber =  0;
079                      FileResource.goToLine(targetIFile, lineNumber);
080                    }
081                  }
082                });
083              }
084            }
085            else {
086              MenuItem menuItem = new MenuItem(menu, SWT.NONE);
087              menuItem.setText(ReviewI18n.getString("ReviewFileSelectionMenu.menuItem.none"));
088            }
089          }
090        }
091      }
092    
093    }
