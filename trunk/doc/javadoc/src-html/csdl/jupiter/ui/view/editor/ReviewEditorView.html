122      /**
123       * @see org.eclipse.ui.part.WorkbenchPart#createPartControl(org.eclipse.swt.widgets.Composite)
124       */
125      public void createPartControl(Composite ancestor) {
126        this.ancestor = ancestor;
127        IWorkbenchPage page = getSite().getPage();
128        page.addPartListener(this.partListener);
129        Composite dialog = createGeneralComposite(ancestor, 0, 0);
130        this.folder = createTabFolder(dialog);
131        createIndividualPhaseTabContent(folder);
132        createTeamPhaseTabContent(folder);
133        createReworkPhaseTabContent(folder);
134        hookListenerOnTabFolder(folder);
135        hookTextActionHandler();
136        createActions();
137        this.setEnable(false);
138      }

140      /**
141       * Hooks listener on tab folder.
142       * @param folder the folder
143       */
144      private void hookListenerOnTabFolder(TabFolder folder) {
145        folder.setData(folder);
146        folder.addListener(SWT.Selection, new Listener() {
147          public void handleEvent(Event event) {
148            TabFolder folder = (TabFolder) event.widget.getData();
149            setActiveTabNameKey(folder);
150            updateTitle();
151            handleTabChangeEvent();
152          }
153        });
154      }

156      /**
157       * Sets the active tab name key.

159       */
160      private void setActiveTabNameKey(TabFolder folder) {
161        TabItem tabItem = folder.getItem(folder.getSelectionIndex());
162        this.activeTabNameKey = ReviewI18n.getKey(tabItem.getText());
163      }

173      /**
174       * Creates view preference frame and return the child composite.
175       * @param parent the parent composite.
176       * @param height the margin height.
177       * @param width the margin width.
178       * @return the child composite.
179       */
180      private Composite createGeneralComposite(Composite parent, int height, int width) {
181        Composite child = new Composite(parent, SWT.LEFT);
182        FormLayout layout = new FormLayout();
183        layout.marginHeight = height;
184        layout.marginWidth = width;
185        child.setLayout(layout);
186        return child;
187      }

189      /**
190       * Creates tab folder.
191       * @param dialog the composite.
192       * @return the <code>TabFolder</code> instance.
193       */
194      private TabFolder createTabFolder(Composite dialog) {
195        TabFolder folder = new TabFolder(dialog, SWT.BOTTOM);
196        folder.setLayout(new TabFolderLayout());
197        FormData folderData = new FormData();
198        folderData.top = new FormAttachment(0, -3);
199        folderData.left = new FormAttachment(0, -2);
200        folderData.right = new FormAttachment(100, +2);
201        folderData.bottom = new FormAttachment(100, 0);
202        folder.setLayoutData(folderData);
203        return folder;
204      }

263      /**
264       * Creates type content.
265       * @param parent the parent composite
266       */
267      private void createTypeSeverityContent(Composite parent) {    
268        Label typeLabel = new Label(parent, SWT.NONE);
269        typeLabel.setText(ReviewI18n.getString("ReviewEditorView.label.type"));
270        this.typeCombo = new Combo(parent, SWT.READ_ONLY);
271        FormData typeLabelData = new FormData();
272        typeLabelData.top = new FormAttachment(typeCombo, 0, SWT.CENTER);
273        typeLabel.setLayoutData(typeLabelData);
274        FormData typeComboData = new FormData();
275        typeComboData.left = new FormAttachment(typeLabel, 75, SWT.LEFT);
276        typeComboData.right = new FormAttachment(50, 0);
277        typeCombo.setLayoutData(typeComboData);
278        ReviewModel reviewModel = ReviewModel.getInstance();
279        IProject project = reviewModel.getProjectManager().getProject();
280        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
281        String[] typeItems = TypeKeyManager.getInstance(project, reviewId).getElements();
282        typeCombo.setItems(typeItems);
283        typeCombo.addFocusListener(new FocusListener() {
284          public void focusGained(FocusEvent event) {
285            int type = ReviewEvent.TYPE_FOCUS;
286            int kind = ReviewEvent.KIND_EDITOR;
287            ReviewPlugin.getInstance().notifyListeners(type, kind);
288          }
289          public void focusLost(FocusEvent event) {
290          }     
291        });
292    
293        Label severityLabel = new Label(parent, SWT.NONE);
294        severityLabel.setText(ReviewI18n.getString("ReviewEditorView.label.severity"));
295        this.severityCombo = new Combo(parent, SWT.READ_ONLY);
296        FormData severityLabelData = new FormData();
297        severityLabelData.left = new FormAttachment(typeCombo, 10);
298        severityLabelData.top = new FormAttachment(typeCombo, 0, SWT.CENTER);
299        severityLabel.setLayoutData(severityLabelData);
300        FormData severityComboData = new FormData();
301        severityComboData.top = new FormAttachment(typeCombo, 0, SWT.CENTER);
302        severityComboData.left = new FormAttachment(severityLabel, 15);
303        severityComboData.right = new FormAttachment(100, 0);
304        severityCombo.setLayoutData(severityComboData);
305        String[] severityItems = SeverityKeyManager.getInstance(project, reviewId).getElements();
306        severityCombo.setItems(severityItems);
307        severityCombo.addListener(SWT.Selection, new Listener() {
308          public void handleEvent(Event event) {
309            String severityKey = ReviewI18n.getKey(severityCombo.getText());
310            if (severityKey.equals(ResourceBundleKey.ITEM_KEY_SEVERITY_TRIVIAL)) {
311              String resolutionKey = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_NEEDSFIXING;        
312              int index = resolutionCombo.indexOf(ReviewI18n.getString(resolutionKey));
313              if (index != -1) {
314                resolutionCombo.select(index);
315              }
316            }
317          }
318        });
319        severityCombo.addFocusListener(new FocusListener() {
320          public void focusGained(FocusEvent event) {
321            int type = ReviewEvent.TYPE_FOCUS;
322            int kind = ReviewEvent.KIND_EDITOR;
323            ReviewPlugin.getInstance().notifyListeners(type, kind);
324          }
325          public void focusLost(FocusEvent event) {
326          }     
327        });
328      }

407       * Creates assigned to and resolution content.

412        assignedToLabel.setText(ReviewI18n.getString("ReviewEditorView.label.assignedTo"));
413        this.assignedToCombo = new Combo(parent, SWT.READ_ONLY);
414        FormData assignedToLabelData = new FormData();
415        assignedToLabelData.top = new FormAttachment(assignedToCombo, 0, SWT.CENTER);
416        assignedToLabel.setLayoutData(assignedToLabelData);
417        FormData assginedToComboData = new FormData();
418        assginedToComboData.left = new FormAttachment(assignedToLabel, 75, SWT.LEFT);
419        assginedToComboData.right = new FormAttachment(50, 0);
420        assignedToCombo.setLayoutData(assginedToComboData);
421        assignedToCombo.addFocusListener(new FocusListener() {
422          public void focusGained(FocusEvent event) {
423            int type = ReviewEvent.TYPE_FOCUS;
424            int kind = ReviewEvent.KIND_EDITOR;
425            ReviewPlugin.getInstance().notifyListeners(type, kind);
426          }
427          public void focusLost(FocusEvent event) {
428          }     
429        });
430        ReviewModel reviewModel = ReviewModel.getInstance();
431        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
432        IProject project = reviewModel.getProjectManager().getProject();
433        if (reviewId != null) {
434          String reviewIdString = reviewId.getReviewId();
435          PropertyResource reviewIdResource = PropertyResource.getInstance(project, false);
436          String[] items = reviewIdResource.getReviewerIdNames(reviewIdString);
437          assignedToCombo.setItems((items == null) ? new String[] {} : items);
438          reviewModel = ReviewModel.getInstance();
439          String reviewerId = reviewModel.getReviewerIdManager().getReviewerId().getReviewerId();
440          assignedToCombo.setText(reviewerId);
441        }
442          
443        Label resolutionLabel = new Label(parent, SWT.NONE);
444        resolutionLabel.setText(ReviewI18n.getString("ReviewEditorView.label.resolution"));
445        this.resolutionCombo = new Combo(parent, SWT.READ_ONLY);
446        FormData resolutionLabelData = new FormData();
447        resolutionLabelData.left = new FormAttachment(assignedToCombo, 10);
448        resolutionLabelData.top = new FormAttachment(assignedToCombo, 0, SWT.CENTER);
449        resolutionLabel.setLayoutData(resolutionLabelData);
450        FormData resolutionComboData = new FormData();
451        resolutionComboData.top = new FormAttachment(assignedToCombo, 0, SWT.CENTER);
452        resolutionComboData.left = new FormAttachment(resolutionLabel, 15);
453        resolutionComboData.right = new FormAttachment(100, 0);

461            ReviewPlugin.getInstance().notifyListeners(type, kind);

687      /**
688       * Brings the code review view to the top no matter when review view is closed or not. Means that
689       * it is opened if it has not been opened yet, or it is brought to top if it already has been
690       * opened, but not brought to top.
691       * 
692       * @return the <code>ReviewTableView</code> instance to be opened and/or bright to top.
693       * @throws ReviewException if part could not be initialized.
694       */
695      public static ReviewEditorView bringViewToTop() throws ReviewException {
696        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
697        IWorkbenchPage page = workbench.getActiveWorkbenchWindow().getActivePage();
698        IViewPart viewPart = page.findView(EDITOR_VIEW_ID);
699        // if  the review view is not showed yet,
700        if (viewPart == null) {
701          try {
702            viewPart = page.showView(EDITOR_VIEW_ID);
703          }
704          catch (PartInitException e) {
705            throw new ReviewException(e.getMessage());
706          }
707        }
708        // if there exists the view, but if not on the top,
709        // then brings it to top when the view is already showed.
710        else if (!page.isPartVisible(viewPart)) {
711          page.bringToTop(viewPart);
712        }
713        if (viewPart instanceof ReviewEditorView) {
714          return (ReviewEditorView) viewPart;
715        }
716        else {
717          throw new ReviewException("The instance is not ReviewEditorView.");
718        }
719      }

863      /**
864       * Sets <code>ReviewIssue</code> instance.
865       * @param reviewIssue the code review instance.
866       */
867      public void setReviewIssue(ReviewIssue reviewIssue) {
868        if (!this.ancestor.isDisposed()) {
869          this.reviewIssue = reviewIssue;
870          ReviewModel reviewModel = ReviewModel.getInstance();
871          IProject project = reviewModel.getProjectManager().getProject();
872          ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
873          TypeKeyManager typeKeyManager = TypeKeyManager.getInstance(project, reviewId);
874          String localizedTypeText = typeKeyManager.getLocalizedLabel(reviewIssue.getType().getKey());
875          this.typeCombo.select(typeCombo.indexOf(localizedTypeText));
876          SeverityKeyManager severityKeyManager = SeverityKeyManager.getInstance(project, reviewId);
877          String severityKey = reviewIssue.getSeverity().getKey();
878          String localizedSeverityText = severityKeyManager.getLocalizedLabel(severityKey);
879          this.severityCombo.select(severityCombo.indexOf(localizedSeverityText));
880          this.summaryText.setText(reviewIssue.getSummary());
881          String description = reviewIssue.getDescription();
882          this.descriptionText.setText(description);
883          this.teamDescriptionText.setText(description);
884          String newDescription = createAnnotationDescription(reviewIssue.getAnnotation(), description);
885          this.reworkDescriptionText.setText(newDescription);
886          this.annotationText.setText(reviewIssue.getAnnotation());
887          this.revisionText.setText(reviewIssue.getRevision());
888          if (reviewId != null) {
889            // if no assigned to person was set yet, then it would be set.
890            String reviewIdString = reviewId.getReviewId();
891            PropertyResource reviewIdResource = PropertyResource.getInstance(project, false);
892            String[] items = reviewIdResource.getReviewerIdNames(reviewIdString);
893            assignedToCombo.setItems((items == null) ? new String[] {} : items);
894            String assginedTo = reviewIssue.getAssignedTo();
895            if (assginedTo.equals("")) {
896              this.assignedToCombo.setText(reviewId.getAuthor());
897            }
898            else {
899              this.assignedToCombo.setText(reviewIssue.getAssignedTo());
900            }
901          }
902          ResolutionKeyManager resolutionKeyManager = ResolutionKeyManager.getInstance(project, 
903                                                                                       reviewId);
904          String resolutionKey = reviewIssue.getResolution().getKey();
905          String localizedResolutionText = resolutionKeyManager.getLocalizedLabel(resolutionKey);
906          this.resolutionCombo.select(resolutionCombo.indexOf(localizedResolutionText));
907          this.resolutionText.setText(localizedResolutionText);
908          StatusKeyManager statusKeyManager = StatusKeyManager.getInstance(project, reviewId);
909          String statusKey = reviewIssue.getStatus().getKey();
910          String localizedStatusText = statusKeyManager.getLocalizedLabel(statusKey);
911          this.statusCombo.select(statusCombo.indexOf(localizedStatusText));
912          this.updateTitle();
913          ReviewEditorViewAction.GOTO.setEnabled(reviewIssue.isLinked());
914          EventFileManager.getInstance().setEventFilePath(this.reviewIssue.getTargetFile());
915        }
916      }

918      /**
919       * Gets the code review instance. Returns null if this view instance does not contains
920       * <code>ReviewIssue</code> instance.
921       * @return the code review instance.
922       */
923      public ReviewIssue getReviewIssue() {
924        if (this.reviewIssue == null) {
925         return null; 
926        }
927        try {
928          ReviewModel reviewModel = ReviewModel.getInstance();
929          IProject project = reviewModel.getProjectManager().getProject();
930          ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
931          TypeKeyManager typeKeyManager = TypeKeyManager.getInstance(project, reviewId);
932          SeverityKeyManager severityKeyManager = SeverityKeyManager.getInstance(project, reviewId);
933          ResolutionKeyManager resolutionKeyManager = ResolutionKeyManager.getInstance(project, 
934                                                                                       reviewId);
935          StatusKeyManager statusKeyManager = StatusKeyManager.getInstance(project, reviewId);
936          String resolutionKey = resolutionKeyManager.getKey(resolutionCombo.getText());
937          String severityKey = severityKeyManager.getKey(severityCombo.getText());
938          return new ReviewIssue(this.reviewIssue.getCreationDate(), new Date(), 
939                this.reviewIssue.getReviewer(), this.assignedToCombo.getText(),
940                this.reviewIssue.getTargetFile(), this.reviewIssue.getLine(),
941                (Type) typeKeyManager.getItemObject(typeKeyManager.getKey(typeCombo.getText())),
942                (Severity) severityKeyManager.getItemObject(severityKey),
943                new org.jdom.Text(this.summaryText.getText()).getText(),
944                this.descriptionText.getText(), this.annotationText.getText(), 
945                this.revisionText.getText(),
946                (Resolution) resolutionKeyManager.getItemObject(resolutionKey),
947                (Status) statusKeyManager.getItemObject(statusKeyManager.getKey(statusCombo.getText())),
948                this.reviewIssue.getReviewIFile());
949        }
950        catch (ReviewException e) {
951          e.printStackTrace();
952          return null;
953        }
954      }

1000      /**
1001       * Updates title.
1002       */
1003      private void updateTitle() {
1004        String title = "";
1005        if (reviewIssue != null) {
1006          if (this.activeTabNameKey.equals(ResourceBundleKey.PHASE_INFIVIDUAL)) {
1007            String individualTitle = reviewIssue.getReviewer() + " : " + reviewIssue.getTargetFile()
1008            + " : " + reviewIssue.getLine();
1009            title = individualTitle;
1010          }
1011          else if (this.activeTabNameKey.equals(ResourceBundleKey.PHASE_TEAM)) {
1012            String teamTitle = reviewIssue.getReviewer() + " : " + summaryText.getText();
1013            title = teamTitle;
1014          }
1015          else if (this.activeTabNameKey.equals(ResourceBundleKey.PHASE_REWORK)) {
1016            String reworkTitle = reviewIssue.getReviewer() + " : " + summaryText.getText();
1017            title = reworkTitle;
1018          }
1019          ReviewTableView tableView = ReviewTableView.getActiveView();
1020          if (tableView == null) {
1021            tableView = ReviewTableView.bringViewToTop();
1022          }
1023        }
1024        this.setContentDescription(title);
1025      }
