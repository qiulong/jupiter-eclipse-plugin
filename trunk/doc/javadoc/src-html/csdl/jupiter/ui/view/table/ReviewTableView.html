117      /**
118       * Removes all table columns in the current view.
119       */
120      private void removeAllColumns() {
121        if (!getViewer().getControl().isDisposed()) {
122          TableViewer tableViewer = getViewer();
123          tableViewer.setSorter(null);
124          Table table = tableViewer.getTable();
125          while (table.getColumnCount() > 0) {
126            int lastIndex = table.getColumnCount() - 1;
127            TableColumn column = table.getColumn(lastIndex);
128            column.dispose();
129          }
130        }
131      }

133      /**
134       * Updates column name, width, and resizable information of the set of <code>ColumnData</code>
135       * which are stored in <code>ColumnDataManager</code>.
136       * @param columnDataManager <code>ColumnDataManager</code> instance.
137       * @param codeReviewView the <code>ReviewTableView</code> instance.
138       */
139      public void updateColumnDataManager(ColumnDataModel columnDataManager,
140                                          ReviewTableView codeReviewView) {
141        ColumnData[] columnDataArray = columnDataManager.getEnabledColumnDataArray();
142        TableViewer tableViewer = codeReviewView.getViewer();
143        Table table = tableViewer.getTable();
144        int length = table.getColumnCount();
145        for (int i = 0; i < length; i++) {
146          TableColumn column = table.getColumn(0);
147          ColumnPixelData columnPixelData = columnDataArray[i].getColumnPixelData();
148          columnPixelData.width = column.getWidth();
149          columnPixelData.resizable = column.getResizable();
150        }
151      }

216      /**
217       * Updates the tile of table. Adds the number of displayed item information. Also adds the total
218       * number of issues in project if filter button in the Jupiter issues view is selected.
219       */
220      public void updateTitle() {
221        ReviewModel reviewModel = ReviewModel.getInstance();
222        IProject project = reviewModel.getProjectManager().getProject();
223        String projectName = (project != null) ? project.getName() : "";
224        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
225        String phaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
226        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
227        ColumnDataModel columnDataModel = columnDataModelManager.getModel(phaseNameKey);
228        String reviewIdName = (reviewId != null) ? " - " + reviewId.getReviewId() : "";
229        String reviewPhaseName = ReviewI18n.getString(phaseNameKey);
230        updateTitle(projectName, reviewIdName, reviewPhaseName);
231      }

240      /**
241       * Updates the title of table.
242       * @param projectName the project name.
243       * @param reviewIdName the review id name.
244       * @param reviewPhaseName the review phase name.
245       */
246      private void updateTitle(String projectName, String reviewIdName, String reviewPhaseName) {
247        int displayedItems = getTable().getItemCount();
248        ReviewIssueModelManager reviewIssueModelManager = ReviewIssueModelManager.getInstance();
249        ReviewIssueModel reviewIssueModel = reviewIssueModelManager.getCurrentModel();
250        int totalItems = reviewIssueModel.size();
251        if (ReviewTableViewAction.FILTER.isChecked()) {
252          IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
253          String prefFilterKey = FilterPreferencePage.ENABLE_FILTER_STORE_KEY;
254          boolean isPrefFilterEnabled = store.getBoolean(prefFilterKey);
255          String titleFirst = (isPrefFilterEnabled)
256                               ? "ReviewTableView.view.title.firstPreference"
257                               : "ReviewTableView.view.title.firstProperty";
258          setContentDescription(projectName + reviewIdName + " - " + reviewPhaseName + " ("
259              + ReviewI18n.getString(titleFirst) + displayedItems
260              + ReviewI18n.getString("ReviewTableView.view.title.between") + totalItems
261              + ((totalItems <= 1)
262              ? ReviewI18n.getString("ReviewTableView.view.title.last.singular")
263              : ReviewI18n.getString("ReviewTableView.view.title.last.plural")) + ")");
264        }
265        else {
266          setContentDescription(projectName + reviewIdName + " - " + reviewPhaseName
267              + " (" + displayedItems 
268              + ((displayedItems <= 1)
269              ? ReviewI18n.getString("ReviewTableView.view.title.last.singular")
270              : ReviewI18n.getString("ReviewTableView.view.title.last.plural")) + ")");
271        }
272      }

301      /**
302       * Called when column header is selected. Checks the header name and triggers
303       * <code>csdl.jupiter.model.ReviewIssueModel.sortBy(Comparator)</code> method.
304       *
305       * @param event the event triggered when column header is selected.
306       */
307      protected void columnHeaderSelected(SelectionEvent event) {
308        String columnName = ((TableColumn) event.getSource()).getText();
309        // if column name is link icon, (i.e. column name is empty), use link icon constant.
310        String columnHeaderLinkIcon = ReviewI18n.getString(ResourceBundleKey.COLUMN_HEADER_LINK_ICON);
311        String columnNameKey = (!columnName.equals("")) ? columnName
312                                                        : columnHeaderLinkIcon;
313        this.model.sortBy(ReviewComparator.getComparator(columnNameKey));
314        this.getViewer().refresh();
315        this.getTable().redraw();
316      }
317      
318      /**
319       * Called when column header is resized.
320       *
321       * @param event the event triggered when column header is resized.
322       */
323      protected void columnHeaderResized(ControlEvent event) {
324        TableColumn column = (TableColumn) event.getSource();
325        String columnName = column.getText();
326        int width = column.getWidth();
327        boolean resizable = column.getResizable();
328        
329        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
330        ColumnDataModel columnDataModel = columnDataModelManager.getModel();
331        
332        // if column name is link icon, (i.e. column name is empty), use link icon constant.
333        String columnNameKey = (!columnName.equals("")) ? ReviewI18n.getKey(columnName)
334                                                        : ResourceBundleKey.COLUMN_HEADER_LINK_ICON;
335        ColumnData columnData = columnDataModel.get(columnNameKey);
336        // set the resized column information into column data.
337        if (columnData != null) {
338          columnData.setColumnPixelData(new ColumnPixelData(width, resizable));
339        }
340        ReviewModel reviewModel = ReviewModel.getInstance();    
341        String phaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
342        // store the modified column data model into config XML file. 
343        PrefResource.getInstance().storeColumnDataModel(phaseNameKey, columnDataModel);
344      }

358      /**
359       * Brings the code review view to the top no matter when review view is closed or not. Means that
360       * it is opened if it has not been opened yet, or it is brought to top if it already has been
361       * opened, but not brought to top.
362       * 
363       * @return the <code>ReviewTableView</code> instance to be opened and/or bright to top.
364       */
365      public static ReviewTableView bringViewToTop() {
366        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
367        IWorkbenchPage page = workbench.getActiveWorkbenchWindow().getActivePage();
368        IViewPart viewPart = page.findView(VIEW_ID);
369        // if  the review view is not showed yet,
370        if (viewPart == null) {
371          try {
372            viewPart = page.showView(VIEW_ID);
373          }
374          catch (PartInitException e) {
375            throw new IllegalStateException(e.getMessage());
376          }
377        }
378        // if there exists the view, but if not on the top,
379        // then brings it to top when the view is already showed.
380        else if (!page.isPartVisible(viewPart)) {
381          page.bringToTop(viewPart);
382        }
383        if (!(viewPart instanceof ReviewTableView)) {
384          throw new IllegalStateException("viewPart is not instance of ReviewTableView");
385        }
386        else {
387          ((ReviewTableView) viewPart).updateTitle();
388          return (ReviewTableView) viewPart;
389        }
390      }

400        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();

403          return null;
404        }

409      /**
410       * Provides the array of the Review table view. A review table view in the array
411       * associates with the opened window. The array might contain <code>null</code> 
412       * if there is no active review table view in the active window.
413       * @return the array of the review table views. 
414       */
415      public static ReviewTableView[] getViews() {
416        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
417        IWorkbenchWindow[] windows = workbench.getWorkbenchWindows();
418        ReviewTableView[] views = new ReviewTableView[windows.length];
419        for (int i = 0; i < windows.length; i++) {
420          IWorkbenchPage page = windows[i].getActivePage();
421          IViewPart viewPart = page.findView(VIEW_ID);
422          views[i] = (ReviewTableView) viewPart;
423        }
424        return views;
425      }

427      /**
428       * Closes the opened code review view if it's opened.
429       */
430      public static void closeView() {
431        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
432        IWorkbenchWindow activeWindow = workbench.getActiveWorkbenchWindow();
433        if (activeWindow != null) {
434          IWorkbenchPage page = activeWindow.getActivePage();
435          if (page != null) {
436            IViewPart viewPart = page.findView(VIEW_ID);
437            if (viewPart != null) {
438              page.hideView(viewPart);
439            }
440          }
441        }
442      }

444      /**
445       * Gets the view id of this plug-in. Returns <code>null</code> if a view id was not found.
446       *
447       * @return the view id of this plug-in.
448       */
449      private static String getViewId() {
450        String pluginId = ReviewPlugin.PLUGIN_ID;
451        IExtension[] extensions = Platform.getExtensionRegistry().getExtensions(pluginId);
452        for (int i = 0; i < extensions.length; i++) {
453          if (extensions[i].getExtensionPointUniqueIdentifier().equals("org.eclipse.ui.views")) {
454            IConfigurationElement[] elements = extensions[i].getConfigurationElements();
455            for (int j = 0; j < elements.length; j++) {
456              if (elements[j].getName().equals("view")) {
457                return elements[j].getAttribute("id");
458              }
459            }
460          }
461        }
462        return null;
463      }
