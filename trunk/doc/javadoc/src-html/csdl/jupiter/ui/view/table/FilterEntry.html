001    package csdl.jupiter.ui.view.table;
002    
003    /**
004     * Provides the filter entry data structure.
005     * @author Takuya Yamashita
006     * @version $Id: FilterEntry.java,v 1.4 2005/03/25 18:32:01 takuyay Exp $
007     */
008    public class FilterEntry {
009      private String filterName;
010      private String valueKey;
011      private boolean isEnabled;
012      
013      /**
014       * Instantiates the filter entry.
015       * @param filterName the filter name.
016       * @param valueKey the filter value key.
017       * @param isEnabled <code>true</code> if the filter entry is enabled.
018       */
019      public FilterEntry(String filterName, String valueKey, boolean isEnabled) {
020        this.filterName = filterName;
021        this.valueKey = valueKey;
022        this.isEnabled = isEnabled;
023      }
024     
025      /**
026       * Gets the filter name.
027       * @return the filter name.
028       */
029      public String getFilterName() {
030        return filterName;
031      }
032      
033      /**
034       * Checks if the filter entry is enabled. Returns <code>true</code> if the filter entry is
035       * enabled.
036       * @return <code>true</code> if the filter entry is enabled.
037       */
038      public boolean isEnabled() {
039        return isEnabled;
040      }
041      
042      /**
043       * Sets the enable status of the filter.
044       * @param isEnabled <code>true</code> if the filter is enabled.
045       */
046      public void setEnabled(boolean isEnabled) {
047        this.isEnabled = isEnabled;
048      }
049      
050      /**
051       * Sets the value key of the filter entry.
052       * @return the value key of the filter entry.
053       */
054      public String getValueKey() {
055        return this.valueKey;
056      }
057      
058      /**
059       * Sets the default value key of the filter.
060       * @param valueKey the default value key of the filter.
061       */
062      public void setKey(String valueKey) {
063        this.valueKey = valueKey;
064      }
065    }
