026    /**
027     * Provides the manager for the <code>ReviewFilter</code> instance. Clients can check the 
028     * filter status (if the filter instance is created and added to the view or not), and/or
029     * add/remove the filter instance to/from the <code>ReviewTableView</code> view.
030     * 
031     * @author Takuya Yamashita
032     * @version $Id: ReviewFilterManager.java,v 1.15 2005/03/28 14:39:30 takuyay Exp $
033     */
034    public class ReviewFilterManager {
035      /** Singleton instance. */
036      private static ReviewFilterManager theInstance = new ReviewFilterManager();
037      /** The <code>ViewerFilter</code> instance. */
038      private ViewerFilter viewerFilter;
039      private StructuredViewer viewer;
040      /** The boolean flag to check if the filter is created or not. */
041      private boolean isCreated;
042      /**
043       * Creates <code>ReviewFilterManager</code> instance.
044       * The field instance is set as null for <code>viewerFilter</code> and false for
045       * <code>isCreated</code>
046       */
047      private ReviewFilterManager() {
048        viewerFilter = new ReviewFilter();
049      }
050      
051      /**
052       * Gets the <code>ReviewFilterManager</code> singleton instance.
053       * @param structuredViewer the structure viewer.
054       * @param viewerFilter the <code>ViewerFilter</code> instance.
055       * @return the the <code>ReviewFilterManager</code> singleton instance.
056       */
057      public static ReviewFilterManager getInstance(StructuredViewer structuredViewer, 
058                                                    ViewerFilter viewerFilter) {
059        theInstance.viewer = structuredViewer;
060        theInstance.setFilter(viewerFilter);
061        return theInstance;
062      }
063      
064      /**
065       * Gets the <code>ReviewFilterManager</code> singleton instance.
066       * @param structuredViewer the structure viewer.
067       * @return the the <code>ReviewFilterManager</code> singleton instance.
068       */
069      public static ReviewFilterManager getInstance(StructuredViewer structuredViewer) {
070        theInstance.viewer = structuredViewer;
071        return theInstance;
072      }
073      
074      /**
075       * Sets the <code>ViewerFilter</code> instance.
076       * @param viewerFilter the <code>ViewerFilter</code> instance.
077       */
078      private void setFilter(ViewerFilter viewerFilter) {
079        if (this.viewerFilter != viewerFilter) {
080          this.viewer.removeFilter(this.viewerFilter);
081        }
082        this.viewerFilter = viewerFilter;
083      }
084      
085      /**
086       * Adds filter.
087       */
088      public void addFilter() {
089        this.viewer.addFilter(this.viewerFilter);
090      }
091      
092      /**
093       * Removes filter.
094       */
095      public void removeFilter() {
096        this.viewer.removeFilter(this.viewerFilter);
097      }
098      
099      /**
100       * Gets the new <code>ReviewFilter</code> instance, whose filter is retrieved from preference
101       * setting.
102       *
103       * @return the new <code>ReviewFilter</code> instance.
104       */
105      public static ViewerFilter createFilterFromPreference() {
106        ReviewFilter filter = new ReviewFilter();
107        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
108        boolean isFilterEanbled = store.getBoolean(FilterPreferencePage.ENABLE_FILTER_STORE_KEY);
109        if (isFilterEanbled) {
110          ReviewModel reviewModel = ReviewModel.getInstance();
111          IProject project = reviewModel.getProjectManager().getProject();
112          ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
113          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_SEVERITY_STORE_KEY)) {
114            String severityKey = store.getString(FilterPreferencePage.FILTER_SEVERITY_COMBO_KEY);
115            SeverityKeyManager severityKeyManager = SeverityKeyManager.getInstance(project, reviewId);
116            int serverityOrdinal = severityKeyManager.getOrdinal(severityKey);
117            filter.setSeverityFilter(new Severity(severityKey, serverityOrdinal));
118          }
119          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_STATUS_STORE_KEY)) {
120            String statusKey = store.getString(FilterPreferencePage.FILTER_STATUS_COMBO_KEY);
121            StatusKeyManager statusKeyManager = StatusKeyManager.getInstance(project, reviewId);
122            int statusOrdinal = statusKeyManager.getOrdinal(statusKey);
123            filter.setStatusFilter(new Status(statusKey, statusOrdinal));
124          }
125          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_RESOLUTION_STORE_KEY)) {
126            String resolutionKey = store.getString(FilterPreferencePage.FILTER_RESOLUTION_COMBO_KEY);
127            ResolutionKeyManager resolutionKeyManager = ResolutionKeyManager.getInstance(project,
128                                                                                         reviewId);
129            int resolutionOrdinal = resolutionKeyManager.getOrdinal(resolutionKey);
130            filter.setResolutionFilter(new Resolution(resolutionKey, resolutionOrdinal));
131          }
132          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_TYPE_STORE_KEY)) {
133            String typeKey = store.getString(FilterPreferencePage.FILTER_TYPE_COMBO_KEY);
134            TypeKeyManager typeKeyManager = TypeKeyManager.getInstance(project, reviewId);
135            int typeOrdinal = typeKeyManager.getOrdinal(typeKey);
136            filter.setTypeFilter(new Type(typeKey, typeOrdinal));
137          }
138          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_ASSIGNED_TO_STORE_KEY)) {
139            String assignedTo = store.getString(FilterPreferencePage.FILTER_ASSIGNED_TO_COMBO_KEY);
140            filter.setAssignedToFilter(assignedTo);
141          }
142          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_REVIEWER_STORE_KEY)) {
143            String reviewer = store.getString(FilterPreferencePage.FILTER_REVIEWER_COMBO_KEY);
144            filter.setReviewerFilter(reviewer);
145          }
146          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_FILE_STORE_KEY)) {
147            String file = store.getString(FilterPreferencePage.FILTER_FILE_COMBO_KEY);
148            filter.setFileFilter(file);
149          }
150          if (store.getBoolean(FilterPreferencePage.ENABLE_FILTER_INTERVAL_STORE_KEY)) {
151            String intervalString = store.getString(FilterPreferencePage.FILTER_INTERVAL_TEXT_KEY);
152            int interval = 0;
153            try {
154              interval = Integer.parseInt(intervalString);
155              interval = (interval >= 0) ? interval : 0;
156            }
157            catch (NumberFormatException e) {
158              // use interval = 0.
159            }
160            filter.setIntervalFilter(interval);
161          }
162        }
163        return filter;
164      }
165      
166      /**
167       * Creates the <code>ViewerFilter</code> from property setting.
168       * @param project the project.
169       * @param reviewId the review id.
170       * @param phaseNameKey the phase name key. e.g. phase.individual, etc.
171       * @return the <code>ViewerFilter</code> from property setting.
172       */
173      public static ViewerFilter createFilterFromProperty(IProject project, ReviewId reviewId,
174                                                         String phaseNameKey) {
175        ReviewFilter filter = new ReviewFilter();
176        PropertyResource propertyResource = PropertyResource.getInstance(project, true);
177        String reviewIdString = reviewId.getReviewId();
178        ReviewResource reviewResource = propertyResource.getReviewResource(reviewIdString, true);
179        FilterPhase filterPhase = reviewResource.getFilterPhase(phaseNameKey);
180        if (reviewResource != null && filterPhase.isEnabled()) {
181          String intervalName = PropertyConstraints.ATTRIBUTE_VALUE_INTERVAL;
182          FilterEntry entry = filterPhase.getFilterEntry(intervalName);
183          if (entry != null && entry.isEnabled()) {
184            int interval = 0;
185            try {
186              interval = Integer.parseInt(entry.getValueKey());
187              interval = (interval >= 0) ? interval : 0;
188            }
189            catch (NumberFormatException e) {
190              // use interval = 0;
191            }
192            filter.setIntervalFilter(interval);
193          }
194          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_REVIEWER);
195          if (entry != null && entry.isEnabled()) {
196            String reviewer = entry.getValueKey();
197            if (reviewer.equals(ReviewerId.AUTOMATIC_KEY)) {
198              ReviewModel reviewModel = ReviewModel.getInstance();
199              ReviewerId reviewerId = reviewModel.getReviewerIdManager().getReviewerId();
200              reviewer = reviewerId.getReviewerId();
201            }
202            filter.setReviewerFilter(reviewer);
203          }
204          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_TYPE);
205          if (entry != null && entry.isEnabled()) {
206            TypeKeyManager manager = TypeKeyManager.getInstance(project, reviewId);
207            Type type = (Type) manager.getItemObject(entry.getValueKey());
208            filter.setTypeFilter(type);
209          }
210          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY);
211          if (entry != null && entry.isEnabled()) {
212            SeverityKeyManager manager = SeverityKeyManager.getInstance(project, reviewId);
213            Severity severity = (Severity) manager.getItemObject(entry.getValueKey());
214            filter.setSeverityFilter(severity);
215          }
216          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_ASSIGNED_TO);
217          if (entry != null && entry.isEnabled()) {
218            String assignedTo = entry.getValueKey();
219            if (assignedTo.equals(ReviewerId.AUTOMATIC_KEY)) {
220              ReviewModel reviewMode = ReviewModel.getInstance();
221              ReviewerId reviewerId = reviewMode.getReviewerIdManager().getReviewerId();
222              assignedTo = reviewerId.getReviewerId();
223            }
224            filter.setAssignedToFilter(assignedTo);
225          }
226          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION);
227          if (entry != null && entry.isEnabled()) {
228            ResolutionKeyManager manager = ResolutionKeyManager.getInstance(project, reviewId);
229            Resolution resolution = (Resolution) manager.getItemObject(entry.getValueKey());
230            filter.setResolutionFilter(resolution);
231          }
232          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_STATUS);
233          if (entry != null && entry.isEnabled()) {
234            StatusKeyManager manager = StatusKeyManager.getInstance(project, reviewId);
235            Status status = (Status) manager.getItemObject(entry.getValueKey());
236            filter.setStatusFilter(status);
237          }
238          entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_FILE);
239          if (entry != null && entry.isEnabled()) {
240            filter.setFileFilter(ReviewI18n.getString(entry.getValueKey()));
241          }
242        }
243        return filter;
244      }
