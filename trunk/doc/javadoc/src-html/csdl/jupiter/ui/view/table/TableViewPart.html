031    /**
032     * Provides the implementation for table view.
033     *
034     * @author Takuya Yamashita
035     * @version $Id: TableViewPart.java,v 1.5 2005/03/25 18:32:02 takuyay Exp $
036     */
037    public class TableViewPart extends ViewPart {
038      /** The TAG_COLUMN type. */
039      protected static final String TAG_COLUMN = "column";
040      /** The TAG_NUMBER type. */
041      protected static final String TAG_NUMBER = "number";
042      /** The TAG_NAME type, which means column header name. */
043      protected static final String TAG_NAME = "name";
044      /** The TAG_WIDTH type. */
045      protected static final String TAG_WIDTH = "width";
046      /** The array of the column header string. */
047      private String[] columnHeaders;
048      /** The array of the column layouts */
049      private ColumnLayoutData[] columnLayouts;
050      /** The double click action. */
051      private IAction doubleClickAction;
052      /** The single click action. */
053      private IAction singleClickAction;
054      /** The <code>Table</code> instance. */
055      private Table table;
056      /** The <code>TableViewer</code> instance. */
057      private TableViewer viewer;
058    
059      /**
060       * Creates table view part.
061       *
062       * @param parent The <code>Composite</code> instance to be hooked.
063       */
064      public void createPartControl(Composite parent) {
065        viewer = new TableViewer(parent, SWT.SINGLE | SWT.FULL_SELECTION | SWT.H_SCROLL | SWT.V_SCROLL);
066        table = viewer.getTable();
067        table.setHeaderVisible(true);
068        table.setLinesVisible(true);
069        createColumns();
070        createActions();
071        hookMenus();
072        hookEvents();
073        contributeToActionBars();
074      }
075    
076      /**
077       * Creates columns of the table view.
078       */
079      protected void createColumns() {
080        TableLayout layout = new TableLayout();
081        table.setLayout(layout);
082        for (int i = 0; i < columnHeaders.length; i++) {
083          final TableColumn tableColumn = new TableColumn(table, SWT.NONE, i);
084          tableColumn.setText(columnHeaders[i]);
085          tableColumn.setResizable(columnLayouts[i].resizable);
086          // :SOLVED: 4/12/04 sets column width.
087          if (columnLayouts[i] instanceof ColumnPixelData) {
088            ColumnPixelData columnPixeldata = (ColumnPixelData) columnLayouts[i];
089            tableColumn.setWidth(columnPixeldata.width);
090          }
091          tableColumn.addControlListener(new ControlListener() {
092            public void controlMoved(ControlEvent event) {       
093            }
094            public void controlResized(ControlEvent event) {
095              TableViewPart.this.columnHeaderResized(event);
096            }
097          });
098          tableColumn.addSelectionListener(new SelectionListener() {
099              public void widgetSelected(SelectionEvent event) {
100                TableViewPart.this.columnHeaderSelected(event);
101              }
102    
103              public void widgetDefaultSelected(SelectionEvent e) {
104                // nothing so for.
105              }
106            });
107          layout.addColumnData(columnLayouts[i]);
108        }
109      }
110    
111      /**
112       * Hooks pop up menus, and add menu listener to it in order to fill context menu. The listener is
113       * notified when the menu is about to show.
114       */
115      protected void hookMenus() {
116        MenuManager menuManager = new MenuManager("#PopupMenu");
117        menuManager.setRemoveAllWhenShown(true);
118        menuManager.addMenuListener(new IMenuListener() {
119            public void menuAboutToShow(IMenuManager manager) {
120              TableViewPart.this.fillContextMenu(manager);
121            }
122          });
123    
124        Menu menu = menuManager.createContextMenu(viewer.getControl());
125        viewer.getControl().setMenu(menu);
126        getSite().registerContextMenu(menuManager, viewer);
127      }
128    
129      /**
130       * Hooks event listeners to listen to events. When selection is changed, the
131       * <code>selectionChanged</code> is called. When the each row is clicked, the
132       * <code>doubleClickAction</code> runs.
133       */
134      protected void hookEvents() {
135        viewer.addSelectionChangedListener(new ISelectionChangedListener() {
136            public void selectionChanged(SelectionChangedEvent event) {
137              if (event.getSelection() != null) {
138                TableViewPart.this.selectionChanged(event);
139                if (singleClickAction != null) {
140                  singleClickAction.run();
141                }
142              }
143            }
144          });
145        viewer.addDoubleClickListener(new IDoubleClickListener() {
146            public void doubleClick(DoubleClickEvent event) {
147              doubleClickAction.run();
148            }
149          });
150      }
151    
152      /**
153       * Fills the action bars. The pull down menu and tool bar menu are filled.
154       */
155      protected void contributeToActionBars() {
156        IActionBars bars = getViewSite().getActionBars();
157        fillLocalPullDown(bars.getMenuManager());
158        fillLocalToolBar(bars.getToolBarManager());
159      }
160    
161      /**
162       * Initialized the table view part. Called by Eclipse platform. Clients should not call this.
163       *
164       * @param site The <code>IViewSite</code> instance.
165       * @param memento The <code>IMemento</code> instance.
166       *
167       * @exception PartInitException if problems occur.
168       */
169      public void init(IViewSite site, IMemento memento) throws PartInitException {
170        super.init(site, memento);
171      }
172    
173      /**
174       * Asks this part to take focus within the workbench.
175       */
176      public void setFocus() {
177        viewer.getControl().setFocus();
178      }
179    
180      /**
181       * Gets the <code>Table</code> instance.
182       *
183       * @return The <code>Table</code> instance.
184       */
185      public Table getTable() {
186        return table;
187      }
188    
189      /**
190       * Gets the <code>TableViewer</code> instance.
191       *
192       * @return The <code>TableViewer</code> instance.
193       */
194      public TableViewer getViewer() {
195        return viewer;
196      }
197

199       * Sets the array of the column header strings.
200       *

204        columnHeaders = strings;
205      }
206

209       *

211       */
212      public void setColumnLayouts(ColumnLayoutData[] data) {
213        columnLayouts = data;
214      }
215    
216      /**
217       * Sets the double click action.
218       *
219       * @param action The <code>IAction</code> instance.
220       */
221      public void setDoubleClickAction(IAction action) {
222        doubleClickAction = action;
223      }
224    
225      /**
226       * Sets the single click action.
227       *
228       * @param action The <code>IAction</code> instance.
229       */
230      public void setSingleClickAction(IAction action) {
231        singleClickAction = action;
232      }

251       * Fills the tool bar menu. Sub extending class should implements this.

272      protected void columnHeaderSelected(SelectionEvent event) {
273      }
274

276       * Called when the column header of the table view is resized. Sub extending class can listen

282      }

287       */
288      protected void createActions() {
289      }
