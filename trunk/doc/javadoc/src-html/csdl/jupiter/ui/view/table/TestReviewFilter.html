001    package csdl.jupiter.ui.view.table;
002    
003    import java.util.Calendar;
004    import java.util.Date;
005    
006    import csdl.jupiter.ReviewException;
007    import csdl.jupiter.ReviewTestCase;
008    import csdl.jupiter.model.reviewissue.ReviewIssue;
009    
010    /**
011     * Provides the test cases for the <code>ReviewFilter</code> class.
012     *
013     * @author Takuya Yamashita
014     * @version $Id: TestReviewFilter.java,v 1.6 2005/03/25 18:32:02 takuyay Exp $
015     */
016    public class TestReviewFilter extends ReviewTestCase {
017      /** The code review instance to be filtered. */
018      private ReviewIssue codeReview;
019    
020      /**
021       * Sets up the code review instance.
022       *
023       * @throws Exception if problems occur.
024       */
025      public void setUp() throws Exception {
026        super.setUp();
027        try {
028          this.codeReview = new ReviewIssue(new Date(), new Date(), "Taro Yamada", "John Smith",
029              "doc/review/bar/baz/Foo.java", "10", praise, normal, "This is summary.",
030              "This is nice description.!? The second sentence continues.",
031              "This is annotation for review.", "This is revision comment.", validFixlater, unresolved,
032              null);
033        }
034        catch (ReviewException e) {
035          fail("Should not throw CodeReviewException: " + e.getMessage());
036        }
037      }
038    
039      /**
040       * Tests the <code>select(Viewer, Object, Object)</code> method, given the each one filter is
041       * set.
042       */
043      public void testEachCodeReviewFilter() {
044        ReviewFilter filter = new ReviewFilter();
045        assertEquals("Checking the no filtering.", true, filter.select(null, null, this.codeReview));
046        // Sets filter type.
047        filter.setTypeFilter(praise);
048        assertEquals("Checking the filter matching with type.", true,
049          filter.select(null, null, this.codeReview));
050        filter.setTypeFilter(defect);
051        assertEquals("Checking the filter no matching with type.", false,
052          filter.select(null, null, codeReview));
053        // sets filter status.
054        filter.clearAllFilters();
055        filter.setStatusFilter(unresolved);
056        assertEquals("Checking the filter matching with status.", true,
057          filter.select(null, null, this.codeReview));
058        filter.setStatusFilter(resolved);
059        assertEquals("Checking the filter no matching with status.", false,
060          filter.select(null, null, this.codeReview));
061        // Sets filter resolution.
062        filter.clearAllFilters();
063        filter.setResolutionFilter(validFixlater);
064        assertEquals("Checking the filter matching with resolution.", true,
065          filter.select(null, null, this.codeReview));
066        filter.setResolutionFilter(validNeedsfixing);
067        assertEquals("Checking the filter no matching with resolution.", false,
068          filter.select(null, null, this.codeReview));
069        // Sets filter assigned to.
070        filter.clearAllFilters();
071        filter.setAssignedToFilter("John Smith");
072        assertEquals("Checking the filter matching with assigned to.", true,
073          filter.select(null, null, this.codeReview));
074        filter.setAssignedToFilter("Joe Danoe");
075        assertEquals("Checking the filter no matching with assigned to.", false,
076          filter.select(null, null, this.codeReview));
077        // Sets filter previous days.
078        filter.clearAllFilters();
079        filter.setIntervalFilter(0);
080        assertEquals("Checking the filter matching with previous days.", true,
081          filter.select(null, null, this.codeReview));
082        filter.setIntervalFilter(3);
083        assertEquals("Checking the filter matching with previous days.", true,
084          filter.select(null, null, this.codeReview));
085        Calendar fourDaysAgo = Calendar.getInstance();
086        fourDaysAgo.add(Calendar.DATE, -4);
087      }
088    }
