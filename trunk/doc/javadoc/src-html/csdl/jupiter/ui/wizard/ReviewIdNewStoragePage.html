001    package csdl.jupiter.ui.wizard;
002    
003    import org.eclipse.core.resources.IProject;
004    import org.eclipse.jface.wizard.WizardPage;
005    import org.eclipse.swt.SWT;
006    import org.eclipse.swt.layout.FormAttachment;
007    import org.eclipse.swt.layout.FormData;
008    import org.eclipse.swt.layout.FormLayout;
009    import org.eclipse.swt.widgets.Composite;
010    import org.eclipse.swt.widgets.Control;
011    import org.eclipse.swt.widgets.Label;
012    import org.eclipse.swt.widgets.Text;
013    
014    import csdl.jupiter.ReviewI18n;
015    import csdl.jupiter.ReviewPlugin;
016    import csdl.jupiter.file.FileResource;
017    import csdl.jupiter.file.PropertyConstraints;
018    import csdl.jupiter.file.PropertyResource;
019    import csdl.jupiter.file.ReviewResource;
020    
021    /**
022     * Provides storage page.
023     * @author Takuya Yamashita
024     * @version $Id: ReviewIdNewStoragePage.java,v 1.4 2005/03/25 18:32:06 takuyay Exp $
025     */
026    public class ReviewIdNewStoragePage extends WizardPage {
027      private Text storageText;
028      private IProject project;
029      private int clientWidth;
030      private static final double RATIO = 0.6;
031      /**
032       * Instantiates the review file configuration page.
033       * @param project the project.
034       * @param pageName the page name
035       * @param imageFilePath the image file path.
036       */
037      protected ReviewIdNewStoragePage(IProject project, String pageName, String imageFilePath) {
038        super(pageName);
039        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
040        setTitle(ReviewI18n.getString("ReviewIdNewStoragePage.label.title"));
041        setDescription(ReviewI18n.getString("ReviewIdNewStoragePage.label.title.description"));
042        this.project = project;
043      }
044    
045      /**
046       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
047       */
048      public void createControl(Composite ancestor) {
049        this.clientWidth = ancestor.computeSize(SWT.DEFAULT, SWT.DEFAULT).x;
050        Composite parent = createsGeneralComposite(ancestor);
051        createStorage(parent);
052        setControl(parent);
053      }
054      
055      /**
056       * Creates view preference frame and return the child composite.
057       * @param parent the parent composite.
058       * @return the child composite.
059       */
060      private Composite createsGeneralComposite(Composite parent) {
061        Composite child = new Composite(parent, SWT.LEFT);
062        FormLayout layout = new FormLayout();
063        layout.marginWidth = 7;
064        clientWidth = clientWidth - (7 * 2);
065        layout.marginHeight = 7;
066        child.setLayout(layout);
067        return child;
068      }
069      
070      /**
071       * Creates the storage content.
072       * @param composite the composite.
073       * @return the control.
074       */
075      private Control createStorage(Composite composite) {
076        Label storageLabel = new Label(composite, SWT.NONE);
077        storageLabel.setText(ReviewI18n.getString("ReviewIdEditDialog.label.storage"));
078        this.storageText = new Text(composite, SWT.BORDER);
079        IProject project = FileResource.getProject(this.project);
080        PropertyResource propertyResource = PropertyResource.getInstance(project, true);
081        String defaultReviewId = PropertyConstraints.DEFAULT_REVIEW_ID;
082        ReviewResource reviewResource = propertyResource.getReviewResource(defaultReviewId, true);
083        if (reviewResource != null) {
084          storageText.setText(reviewResource.getReviewId().getDirectory());
085        }
086        
087        FormData storageLabelData = new FormData();
088        storageLabelData.width = (int) (clientWidth * RATIO);
089        storageLabelData.top = new FormAttachment(storageText, 0, SWT.CENTER);
090        storageLabel.setLayoutData(storageLabelData);
091        FormData storageTextData = new FormData();
092        storageTextData.left = new FormAttachment(storageLabel, 0);
093        storageTextData.right = new FormAttachment(100, 0);
094        storageText.setLayoutData(storageTextData);
095        return composite;
096      }
097      
098      /**
099       * Gets the directory.
100       * @return the directory.
101       */
102      public String getDirectory() {
103        return this.storageText.getText();
104      }
105    
106    }
