031    /**
032     * Provides default items page
033     * @author Takuya Yamashita
034     * @version $Id: ReviewIdNewDefaultItemsPage.java,v 1.6 2005/03/25 18:32:06 takuyay Exp $
035     */
036    public class ReviewIdNewDefaultItemsPage extends WizardPage {
037      private IProject project;
038      private Combo typeCombo;
039      private Combo severityCombo;
040      private Combo resolutionCombo;
041      private Combo statusCombo;
042      private int clientWidth;
043      private static final double RATIO = 0.6;
044      /**
045       * Instantiates the default items configuration page.
046       * @param project the project.
047       * @param pageName the page name.
048       * @param imageFilePath the image file path.
049       */
050      protected ReviewIdNewDefaultItemsPage(IProject project, String pageName, String imageFilePath) {
051        super(pageName);
052        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
053        setTitle(ReviewI18n.getString("ReviewIdNewDefaultItemsPage.label.title"));
054        setDescription(ReviewI18n.getString("ReviewIdNewDefaultItemsPage.label.title.description"));
055        this.project = project;
056      }
057    
058      /**
059       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
060       */
061      public void createControl(Composite ancestor) {
062        this.clientWidth = ancestor.computeSize(SWT.DEFAULT, SWT.DEFAULT).x;
063        Composite parent = createsGeneralComposite(ancestor);
064        createDefaultItems(parent);
065        setControl(parent);
066      }
067      
068      /**
069       * Creates view preference frame and return the child composite.
070       * @param parent the parent composite.
071       * @return the child composite.
072       */
073      private Composite createsGeneralComposite(Composite parent) {
074        Composite child = new Composite(parent, SWT.LEFT);
075        FormLayout layout = new FormLayout();
076        layout.marginWidth = 7;
077        clientWidth = clientWidth - (7 * 2);
078        layout.marginHeight = 7;
079        child.setLayout(layout);
080        return child;
081      }
082      
083      /**
084       * Creates default items content.
085       * @param composite the composite.
086       * @return the control.
087       */
088      private Control createDefaultItems(Composite composite) {
089        PropertyResource propertyResource = PropertyResource.getInstance(project, true);
090        String reviewIdString = PropertyConstraints.DEFAULT_REVIEW_ID;
091        ReviewResource reviewResource = propertyResource.getReviewResource(reviewIdString, true);
092        ReviewId reviewId = null;
093        if (reviewResource != null) {
094          reviewId = reviewResource.getReviewId();
095        }
096        IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_ITEM_ENTRIES);
097        ReviewIdNewItemEntriesPage itemEntryPage = (ReviewIdNewItemEntriesPage) page;
098        final Map fieldItemIdFieldItemMap = itemEntryPage.getFieldItemIdFieldItemMap();
099        
100        // create type label and its combo.
101        Label typeLabel = new Label(composite, SWT.NONE);
102        typeLabel.setText(ReviewI18n.getString("ReviewIdEditDialog.label.type"));
103        this.typeCombo = new Combo(composite, SWT.READ_ONLY);
104        typeCombo.setData(typeCombo);
105        TypeKeyManager manager = TypeKeyManager.getInstance(project, reviewId);
106        String[] elements = (reviewId != null) ? manager.getElements() : new String[]{""};
107        typeCombo.setItems(elements);
108        String typeName = PropertyConstraints.ATTRIBUTE_VALUE_TYPE;
109        String typeKey = (reviewResource != null) ? reviewResource.getDefaultField(typeName) : "";
110        typeCombo.setText(ReviewI18n.getString(typeKey));
111        typeCombo.addListener(SWT.Selection, new Listener() {
112          public void handleEvent(Event event) {
113            String nameId = PropertyConstraints.ATTRIBUTE_VALUE_TYPE;
114            FieldItem fieldItem = (FieldItem) fieldItemIdFieldItemMap.get(nameId);
115            fieldItem.setDefaultKey(ReviewI18n.getKey(typeCombo.getText()));
116          }
117        });
118        
119        FormData typeLabelData = new FormData();
120        typeLabelData.width = (int) (clientWidth * RATIO);
121        typeLabelData.top = new FormAttachment(typeCombo, 0, SWT.CENTER);
122        typeLabel.setLayoutData(typeLabelData);
123        FormData typeComboData = new FormData();
124        typeComboData.left = new FormAttachment(typeLabel, 0);
125        typeComboData.right = new FormAttachment(100, 0);
126        typeCombo.setLayoutData(typeComboData);
127        
128        // create severity label and its combo.
129        Label severityLabel = new Label(composite, SWT.NONE);
130        severityLabel.setText(ReviewI18n.getString("ReviewIdEditDialog.label.severity"));
131        this.severityCombo = new Combo(composite, SWT.READ_ONLY);
132        severityCombo.setData(severityCombo);
133        severityCombo.setItems(SeverityKeyManager.getInstance(project, reviewId).getElements());
134        String severityName = PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY;
135        String severityKey = (reviewResource != null) ? reviewResource.getDefaultField(severityName)
136                                                      : "";
137        severityCombo.setText(ReviewI18n.getString(severityKey));
138        severityCombo.addListener(SWT.Selection, new Listener() {
139          public void handleEvent(Event event) {
140            String nameId = PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY;
141            FieldItem fieldItem = (FieldItem) fieldItemIdFieldItemMap.get(nameId);
142            fieldItem.setDefaultKey(ReviewI18n.getKey(severityCombo.getText()));
143          }
144        });
145        
146        FormData severityLabelData = new FormData();
147        severityLabelData.width = (int) (clientWidth * RATIO);
148        severityLabelData.top = new FormAttachment(severityCombo, 0, SWT.CENTER);
149        severityLabel.setLayoutData(severityLabelData);
150        FormData severityComboData = new FormData();
151        severityComboData.top = new FormAttachment(typeCombo, 5);
152        severityComboData.left = new FormAttachment(severityLabel, 0);
153        severityComboData.right = new FormAttachment(100, 0);
154        severityCombo.setLayoutData(severityComboData);
155        
156        // create resolution label and its combo.
157        Label resolutionLabel = new Label(composite, SWT.NONE);
158        resolutionLabel.setText(ReviewI18n.getString("ReviewIdEditDialog.label.resolution"));
159        this.resolutionCombo = new Combo(composite, SWT.READ_ONLY);
160        resolutionCombo.setData(resolutionCombo);
161        resolutionCombo.setItems(ResolutionKeyManager.getInstance(project, reviewId).getElements());
162        String resolutionName = PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION;
163        String resolutionKey = (reviewResource != null) 
164                                ? reviewResource.getDefaultField(resolutionName) : "";
165        resolutionCombo.setText(ReviewI18n.getString(resolutionKey));
166        resolutionCombo.addListener(SWT.Selection, new Listener() {
167          public void handleEvent(Event event) {
168            String nameId = PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION;
169            FieldItem fieldItem = (FieldItem) fieldItemIdFieldItemMap.get(nameId);
170            fieldItem.setDefaultKey(ReviewI18n.getKey(resolutionCombo.getText()));
171          }
172        });
173        
174        FormData resolutionLabelData = new FormData();
175        resolutionLabelData.width = (int) (clientWidth * RATIO);
176        resolutionLabelData.top = new FormAttachment(resolutionCombo, 0, SWT.CENTER);
177        resolutionLabel.setLayoutData(resolutionLabelData);
178        FormData resolutionComboData = new FormData();
179        resolutionComboData.top = new FormAttachment(severityCombo, 5);
180        resolutionComboData.left = new FormAttachment(resolutionLabel, 0);
181        resolutionComboData.right = new FormAttachment(100, 0);
182        resolutionCombo.setLayoutData(resolutionComboData);
183        
184        // create status label and its combo.
185        Label statusLabel = new Label(composite, SWT.NONE);
186        statusLabel.setText(ReviewI18n.getString("ReviewIdEditDialog.label.status"));
187        this.statusCombo = new Combo(composite, SWT.READ_ONLY);
188        statusCombo.setData(statusCombo);
189        statusCombo.setItems(StatusKeyManager.getInstance(project, reviewId).getElements());
190        String statusName = PropertyConstraints.ATTRIBUTE_VALUE_STATUS;
191        String statusKey = (reviewResource != null) ? reviewResource.getDefaultField(statusName) : "";
192        statusCombo.setText(ReviewI18n.getString(statusKey));
193        statusCombo.addListener(SWT.Selection, new Listener() {
194          public void handleEvent(Event event) {
195            String nameId = PropertyConstraints.ATTRIBUTE_VALUE_STATUS;
196            FieldItem fieldItem = (FieldItem) fieldItemIdFieldItemMap.get(nameId);
197            fieldItem.setDefaultKey(ReviewI18n.getKey(statusCombo.getText()));
198          }
199        });
200        
201        FormData statusLabelData = new FormData();
202        statusLabelData.width = (int) (clientWidth * RATIO);
203        statusLabelData.top = new FormAttachment(statusCombo, 0, SWT.CENTER);
204        statusLabel.setLayoutData(statusLabelData);
205        FormData statusComboData = new FormData();
206        statusComboData.top = new FormAttachment(resolutionCombo, 5);
207        statusComboData.left = new FormAttachment(statusLabel, 0);
208        statusComboData.right = new FormAttachment(100, 0);
209        statusCombo.setLayoutData(statusComboData);
210        return composite;
211      }
