073      /**
074       * Creates view preference frame and return the child composite.
075       * 
076       * @param parent the parent composite.
077       * @return the child composite.
078       */
079      private Composite createsGeneralComposite(Composite parent) {
080        Composite child = new Composite(parent, SWT.LEFT);
081        FormLayout layout = new FormLayout();
082        layout.marginWidth = 7;
083        layout.marginHeight = 7;
084        child.setLayout(layout);
085        return child;
086      }

088      /**
089       * Creates item entries content.
090       * @param composite the composite.
091       * @return the control.
092       */
093      private Control createItemEntriesContent(Composite composite) {
094        PropertyResource propertyResource = PropertyResource.getInstance(this.project, true);
095        String reviewIdString = PropertyConstraints.DEFAULT_REVIEW_ID;
096        ReviewResource reviewResource = propertyResource.getReviewResource(reviewIdString, true);
097        if (reviewResource != null) {
098          this.fieldItemIdFieldItemMap = reviewResource.getFieldItemMap();
099          this.fieldItemIdList = reviewResource.getFieldItemIdList();
100        }
101        this.itemCombo = new Combo(composite, SWT.READ_ONLY);
102        itemCombo.setData(itemCombo);
103        itemCombo.setItems((String[]) fieldItemIdList.toArray(new String[] {}));
104        itemCombo.setText((String) fieldItemIdList.get(0));
105        itemCombo.addListener(SWT.Selection, new Listener() {
106          public void handleEvent(Event event) {
107            Combo selectedItemCombo = (Combo) event.widget.getData();
108            handleItemIdComboSelection(selectedItemCombo);
109          }
110        });
111        FormData itemComboData = new FormData();
112        itemComboData.left = new FormAttachment(0, 0);
113        itemComboData.right = new FormAttachment(100, 0);
114        itemCombo.setLayoutData(itemComboData);
115        
116        createItemEntriesTable(composite);
117        return composite;
118      }

230      /**
231       * Creates new item entry.
232       */
233      protected void newItemEntry() {
234        String shortMessageKey = "ReviewIdEditDialog.dialogMessage.label.tab.itemEntries.new.short";
235        String longMessageKey = "ReviewIdEditDialog.dialogMessage.label.tab.itemEntries.new.long";    
236        InputDialog dialog = openDialog("", shortMessageKey, longMessageKey);
237        if (dialog.getReturnCode() != InputDialog.OK) {
238          return;
239        }
240        String itemName = dialog.getValue();
241        TableItem item = new TableItem(this.itemListTable, SWT.NONE);
242        String fieldItemId = this.itemCombo.getText();
243        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
244        if (fieldItem != null) {
245          List itemList = fieldItem.getEntryNameList();
246          itemList.add(itemName);
247          fillItemTable(fieldItemId);
248          updateDefaultItems(fieldItemId);
249          IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_FILTERS);
250          ReviewIdNewFilterPage filterPage = (ReviewIdNewFilterPage) page;
251          filterPage.updateFilterItems(fieldItemId);
252        }
253      }
254      
255      /**
256       * Edits the item entry.
257       */
258      protected void editItemEntry() {
259        String shortMessageKey = "ReviewIdEditDialog.dialogMessage.label.tab.itemEntries.edit.short";
260        String longMessageKey = "ReviewIdEditDialog.dialogMessage.label.tab.itemEntries.edit.long";
261        TableItem item = this.itemListTable.getSelection()[0];
262        String oldName = item.getText();
263        InputDialog dialog = openDialog(oldName, shortMessageKey, longMessageKey);
264        if (dialog.getReturnCode() != InputDialog.OK) {
265          return;
266        }
267        String newName = dialog.getValue();
268        item.setText(newName);
269        String fieldItemId = this.itemCombo.getText();
270        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
271        if (fieldItem != null) {
272          List itemList = fieldItem.getEntryNameList();
273          int index = itemList.indexOf(oldName);
274          itemList.remove(index);
275          itemList.add(index, newName);
276          fillItemTable(fieldItemId);
277          updateDefaultItems(fieldItemId);
278          IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_FILTERS);
279          ReviewIdNewFilterPage filterPage = (ReviewIdNewFilterPage) page;
280          filterPage.updateFilterItems(fieldItemId);
281        }
282      }

284      /**
285       * Removes item entries form the table.
286       */
287      protected void removeItemEntries() {
288        String fieldItemId = this.itemCombo.getText();
289        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
290        if (fieldItem != null) {
291          List itemList = fieldItem.getEntryNameList();
292          TableItem[] items = this.itemListTable.getSelection();
293          for (int i = 0; i < items.length; i++) {
294            String itemName = items[i].getText();
295            itemList.remove(itemName);
296          }
297          fillItemTable(fieldItemId);
298          updateDefaultItems(fieldItemId);
299          IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_FILTERS);
300          ReviewIdNewFilterPage filterPage = (ReviewIdNewFilterPage) page;
301          filterPage.updateFilterItems(fieldItemId);
302        }
303      }

305      /**
306       * Moves the selected item entry by one upward if <code>isUpward</code> is <code>true</code>.
307       * Otherwise, moves the selected item entry by one downward.
308       * @param isUpward <code>true</code> if moving the selected item entry by one upward.
309       * <code>false</code> if moving the selected item entry by one downward.
310       */
311      protected void moveItemEntry(boolean isUpward) { 
312        String fieldItemId = this.itemCombo.getText();
313        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
314        TableItem[] selectedItems = this.itemListTable.getSelection();
315        if (selectedItems.length > 0 && fieldItem != null) {
316          TableItem selectedItem = selectedItems[0];
317          String itemName = selectedItem.getText();
318          List itemList = fieldItem.getEntryNameList();
319          int index = itemList.indexOf(itemName);
320          if ((isUpward) ? index > 0 : index < itemList.size() - 1) {
321            int nextIndex = (isUpward) ? index - 1 : index + 1;
322            itemList.remove(index);
323            itemList.add(nextIndex, itemName);
324            fillItemTable(fieldItemId);
325            this.itemListTable.select(nextIndex);
326            handleItemListTableSelection();
327            updateDefaultItems(fieldItemId);
328            IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_FILTERS);
329            ReviewIdNewFilterPage filterPage = (ReviewIdNewFilterPage) page;
330            filterPage.updateFilterItems(fieldItemId);
331          }
332        }
333      }

335      /**
336       * Restores the item entries from the default review id.
337       */
338      protected void restoreItemEntries() {
339        // read from .jupiter.
340        PropertyResource propertyResource = PropertyResource.getInstance(project, true);
341        String defaultReviewID = PropertyConstraints.DEFAULT_REVIEW_ID;
342        ReviewResource reviewResource = propertyResource.getReviewResource(defaultReviewID, true);
343        if (reviewResource != null) {
344          String fieldItemId = this.itemCombo.getText();
345          FieldItem fieldItem = reviewResource.getFieldItem(fieldItemId);
346          this.fieldItemIdFieldItemMap.put(fieldItemId, fieldItem);
347          fillItemTable(fieldItemId);
348          updateDefaultItems(fieldItemId);
349          IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_FILTERS);
350          ReviewIdNewFilterPage filterPage = (ReviewIdNewFilterPage) page;
351          filterPage.updateFilterItems(fieldItemId);
352        }
353      }

355      /**
356       * Opens the dialog window.
357       * @param existingItemName the existing item name.
358       * @param shortMessageKey the short message key.
359       * @param longMessageKey the long message key.
360       * @return the input dialog the <code>InputDialog</code>.
361       */
362      private InputDialog openDialog(String existingItemName, String shortMessageKey, 
363                          String longMessageKey) {
364        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(this.itemCombo.getText());
365        if (fieldItem != null) {
366          final List itemList = fieldItem.getEntryNameList();
367          IInputValidator validator = new IInputValidator() {
368            public String isValid(String newText) {
369              if (!itemList.contains(ReviewI18n.getKey(newText))) {
370                return null;
371              }
372              else {
373                String errorKey = "ReviewIdEditDialog.dialogMessage.label.tab.itemEntries.error";
374                return ReviewI18n.getString(errorKey);
375              }
376            }
377          };
378          InputDialog dialog = new InputDialog(getShell(), ReviewI18n.getString(shortMessageKey), 
379                                               ReviewI18n.getString(longMessageKey), existingItemName,
380                                               validator);  //$NON-NLS-1$ //$NON-NLS-2$
381          dialog.open();
382          return dialog;
383        }
384        return null;
385      }

387      /**
388       * Handles the item id combo selection.
389       * @param itemCombo the item combo.
390       */
391      protected void handleItemIdComboSelection(Combo itemCombo) {
392        fillItemTable(itemCombo.getText());
393        handleItemListTableSelection();
394      }
395      
396      /**
397       * Handles the item list table selection.
398       */
399      protected void handleItemListTableSelection() {
400        boolean isOneItemEntry = (this.itemListTable.getSelectionCount() == 1);
401        int index = this.itemListTable.getSelectionIndex();
402        boolean isSelected = (index >= 0);
403        this.editButtonInItemEntries.setEnabled(isOneItemEntry);
404        this.removeButtonInItemEntries.setEnabled(isSelected);
405        this.upButtonInItemEntries.setEnabled((index > 0) && isOneItemEntry);
406        int lastIndex = this.itemListTable.getItemCount() - 1;
407        this.downButtonInItemEntries.setEnabled((index < lastIndex) && isOneItemEntry);
408      }

410      /**
411       * Fills item data in the item table.
412       * @param fieldItemId the field item id.
413       */
414      private void fillItemTable(String fieldItemId) {
415        removeItems();
416        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
417        if (fieldItem != null) {
418          for (Iterator i = fieldItem.getEntryNameList().iterator(); i.hasNext();) {
419            String itemEntry = ReviewI18n.getString((String) i.next());
420            TableItem item = new TableItem(this.itemListTable, SWT.NONE);
421            item.setText(itemEntry);
422          }
423        }
424      }
425      
426      /**
427       * Removes item data from the item table.
428       */
429      private void removeItems() {
430        TableItem[] items = this.itemListTable.getItems();
431        for (int i = 0; i < items.length; i++) {
432          TableItem item  = items[i];
433          item.dispose();
434        }
435      }
436      
437      /**
438       * Updates the default items folder.
439       * @param fieldItemId the field item id.
440       */
441      private void updateDefaultItems(String fieldItemId) {
442        String pageName = ReviewIdNewWizard.PAGE_DEFAULT_ITEMS;
443        ReviewIdNewDefaultItemsPage page = (ReviewIdNewDefaultItemsPage) getWizard().getPage(pageName);
444        FieldItem fieldItem = (FieldItem) this.fieldItemIdFieldItemMap.get(fieldItemId);
445        if (fieldItem != null) {
446          List itemNameList = fieldItem.getEntryNameList();
447          if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_TYPE)) {
448            page.setItemType((String[]) itemNameList.toArray(new String[] {}));
449          }
450          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY)) {
451            page.setItemSeverity((String[]) itemNameList.toArray(new String[] {}));
452          }
453          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION)) {
454            page.setItemResolution((String[]) itemNameList.toArray(new String[] {}));
455          }
456          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_STATUS)) {
457            page.setItemStatus((String[]) itemNameList.toArray(new String[] {}));
458          }
459        }
460      }
