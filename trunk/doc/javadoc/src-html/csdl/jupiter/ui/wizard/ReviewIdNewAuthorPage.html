001    package csdl.jupiter.ui.wizard;
002    
003    import java.util.Map;
004    
005    import org.eclipse.core.resources.IProject;
006    import org.eclipse.jface.wizard.IWizardPage;
007    import org.eclipse.jface.wizard.WizardPage;
008    import org.eclipse.swt.SWT;
009    import org.eclipse.swt.layout.FormAttachment;
010    import org.eclipse.swt.layout.FormData;
011    import org.eclipse.swt.layout.FormLayout;
012    import org.eclipse.swt.widgets.Combo;
013    import org.eclipse.swt.widgets.Composite;
014    import org.eclipse.swt.widgets.Event;
015    import org.eclipse.swt.widgets.Label;
016    import org.eclipse.swt.widgets.Listener;
017    
018    import csdl.jupiter.ReviewI18n;
019    import csdl.jupiter.ReviewPlugin;
020    import csdl.jupiter.file.PropertyConstraints;
021    import csdl.jupiter.file.PropertyResource;
022    import csdl.jupiter.model.review.ReviewId;
023    import csdl.jupiter.util.JupiterLogger;
024    
025    /**
026     * Provides author page.
027     * @author Takuya Yamashita
028     * @version $Id: ReviewIdNewAuthorPage.java,v 1.4 2005/03/25 18:32:06 takuyay Exp $
029     */
030    public class ReviewIdNewAuthorPage extends WizardPage {
031      /** Jupiter logger */
032      private JupiterLogger log = JupiterLogger.getLogger();
033    
034      private IProject project;
035      private Combo authorCombo;
036      private int clientWidth;
037      private static final double RATIO = 0.6;
038      /**
039       * Instantiates the config reviewer page.
040       * @param project the project.
041       * @param pageName the page name
042       * @param imageFilePath the image file path.
043       */
044      protected ReviewIdNewAuthorPage(IProject project, String pageName, String imageFilePath) {
045        super(pageName);
046        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
047        setTitle(ReviewI18n.getString("ReviewIdNewAuthorPage.label.title"));
048        setDescription(ReviewI18n.getString("ReviewIdNewAuthorPage.label.title.description"));
049        this.project = project;
050      }
051      
052      /**
053       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
054       */
055      public void createControl(Composite ancestor) {
056        this.clientWidth = ancestor.computeSize(SWT.DEFAULT, SWT.DEFAULT).x;
057        Composite parent = createsGeneralComposite(ancestor);
058        createProjectSelectionContent(parent);
059        setControl(parent);
060      }
061      
062      /**
063       * Creates view preference frame and return the child composite.
064       * 
065       * @param parent the parent composite.
066       * @return the child composite.
067       */
068      private Composite createsGeneralComposite(Composite parent) {
069        Composite child = new Composite(parent, SWT.LEFT);
070        FormLayout layout = new FormLayout();
071        layout.marginWidth = 7;
072        clientWidth = clientWidth - (7 * 2);
073        layout.marginHeight = 7;
074        child.setLayout(layout);
075        return child;
076      }
077      
078      /**
079       * Creates project selection content.
080       * @param parent the parent.
081       */
082      private void createProjectSelectionContent(Composite parent) {
083        Label authorLabel = new Label(parent, SWT.NONE);
084        authorLabel.setText(ReviewI18n.getString("ReviewIdNewAuthorPage.label.author"));
085        this.authorCombo = new Combo(parent, SWT.READ_ONLY);
086        authorCombo.setData(authorCombo);
087        PropertyResource reviewIdResource = PropertyResource.getInstance(this.project, true);
088        String defualtReviewId = PropertyConstraints.DEFAULT_REVIEW_ID;
089        ReviewId reviewId = reviewIdResource.getReviewId(defualtReviewId);
090        Map reviewers = reviewId.getReviewers();
091        String author = reviewId.getAuthor();
092        authorCombo.setItems((String[]) reviewers.keySet().toArray(new String[] {}));
093        authorCombo.setText(author);
094        authorCombo.addListener(SWT.Selection, new Listener() {
095          public void handleEvent(Event event) {
096            handleAuthorSelection(event);
097          }
098        });
099    
100        FormData authorLabelData = new FormData();
101        authorLabelData.width = (int) (clientWidth * RATIO);
102        authorLabelData.top = new FormAttachment(authorCombo, 0, SWT.CENTER);
103        authorLabel.setLayoutData(authorLabelData);
104        FormData authorComboData = new FormData();
105        authorComboData.left = new FormAttachment(authorLabel, 0);
106        authorComboData.right = new FormAttachment(100, 0);
107        authorCombo.setLayoutData(authorComboData);
108      }
109      
110      /**
111       * Returns the next page. Saves the values from this page in the model associated with the wizard.
112       * Initializes the widgets on the next page.
113       * @return the next page.
114       */
115      public IWizardPage getNextPage() {
116        return ((ReviewIdNewWizard) getWizard()).getPage(ReviewIdNewWizard.PAGE_ITEM_ENTRIES);
117      }
118      
119      /**
120       * Handles author selection
121       * @param event the event.
122       */
123      protected void handleAuthorSelection(Event event) {
124    //    ((ReviewIdNewWizard) getWizard()).setAuthor(this.authorCombo.getText());
125        getWizard().getContainer().updateButtons();
126      }
127    
128      /**
129       * Sets author items.
130       * @param items the array of the author combo items.
131       */
132      public void setAuthorItems(String[] items) {
133        this.authorCombo.setItems(items);
134      }
135      
136      /**
137       * Sets the default author.
138       * @param author the default author.
139       */
140      public void setDefaultAuthor(String author) {
141        this.authorCombo.setText(author);
142      }
143      
144      /**
145       * Gets the author of the review id.
146       * @return the author of the review id.
147       */
148      public String getAuthor() {
149        return this.authorCombo.getText();
150      }
151    
152    }
