001    package csdl.jupiter.ui.property;
002    
003    import java.text.SimpleDateFormat;
004    
005    import org.eclipse.core.resources.IFile;
006    import org.eclipse.jface.dialogs.Dialog;
007    import org.eclipse.swt.SWT;
008    import org.eclipse.swt.layout.FormAttachment;
009    import org.eclipse.swt.layout.FormData;
010    import org.eclipse.swt.layout.FormLayout;
011    import org.eclipse.swt.widgets.Composite;
012    import org.eclipse.swt.widgets.Control;
013    import org.eclipse.swt.widgets.Event;
014    import org.eclipse.swt.widgets.Label;
015    import org.eclipse.swt.widgets.Listener;
016    import org.eclipse.swt.widgets.Shell;
017    import org.eclipse.swt.widgets.Table;
018    import org.eclipse.swt.widgets.TableItem;
019    
020    import csdl.jupiter.ReviewI18n;
021    
022    /**
023     * Provides the review id removal dialog.
024     * @author Takuya Yamashita
025     * @version $Id: ReviewIdRemovalDialog.java,v 1.5 2004/08/21 10:56:15 takuyay Exp $
026     */
027    public class ReviewIdRemovalDialog extends Dialog {
028     
029      private IFile[] reviewIFiles;
030      private Label infoLabel;
031      private Table reviewFileTable;
032      private SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd");
033    
034      /**
035       * Instantiates review id removal dialog.
036       * @param parentShell the <code>Shell</code> instance.
037       * @param reviewIFiles the array of the review <code>IFile</code> instances.
038       */
039      protected ReviewIdRemovalDialog(Shell parentShell, IFile[] reviewIFiles) {
040        super(parentShell);
041        this.reviewIFiles = reviewIFiles;
042      }
043    
044      /**
045       * @see org.eclipse.jface.dialogs.Dialog#createDialogArea(org.eclipse.swt.widgets.Composite)
046       */
047      protected Control createDialogArea(Composite parent) {
048        Composite composite = (Composite) super.createDialogArea(parent);
049        Composite dialog = createsGeneralComposite(composite);
050        createMessageLabelContent(dialog);
051        if (this.reviewIFiles.length > 0) {
052          createReviewFileTableContent(dialog);
053        }
054        return composite;
055      }
056    
057      /**
058       * Creates view preference frame and return the child composite.
059       * @param parent the parent composite.
060       * @return the child composite.
061       */
062      private Composite createsGeneralComposite(Composite parent) {
063        Composite child = new Composite(parent, SWT.LEFT);
064        FormLayout layout = new FormLayout();
065        layout.marginWidth = 7;
066        layout.marginHeight = 7;
067        child.setLayout(layout);
068        return child;
069      }
070    
071      /**
072       * Creates the message label content.
073       * @param dialog the composite.
074       */
075      private void createMessageLabelContent(Composite dialog) {
076        this.infoLabel = new Label(dialog, SWT.NULL);
077        String infoMessage = "";
078        if (this.reviewIFiles.length > 0) {
079          infoMessage = ReviewI18n.getString("ReviewIdRemovalDialog.label.info.file");
080        }
081        else {
082          infoMessage = ReviewI18n.getString("ReviewIdRemovalDialog.label.info.noFile");
083        }
084        infoMessage += ReviewI18n.getString("ReviewIdRemovalDialog.label.info.confirmation");
085        infoLabel.setText(infoMessage);
086      }
087      
088      
089      /**
090       * Creates the review file table content.
091       * @param dialog the composite.
092       */
093      private void createReviewFileTableContent(Composite dialog) {
094        this.reviewFileTable = new Table(dialog, SWT.BORDER);
095        FormData reviewFileTableData = new FormData();
096        reviewFileTableData.left = new FormAttachment(infoLabel, 0, SWT.LEFT);
097        reviewFileTableData.right = new FormAttachment(100, 0);
098        reviewFileTableData.top = new FormAttachment(infoLabel, 5);
099        reviewFileTableData.bottom = new FormAttachment(100, 0);
100        reviewFileTable.setLayoutData(reviewFileTableData);
101        reviewFileTable.addListener(SWT.Selection, new Listener() {
102          public void handleEvent(Event e) {
103          }
104        });
105        fillReviewFileTable();
106      }
107    
108      /**
109       * Fills review file table with review ID files.
110       */
111      private void fillReviewFileTable() {
112        for (int i = 0; i < this.reviewIFiles.length; i++) {
113          TableItem item = new TableItem(this.reviewFileTable, SWT.NONE);
114          item.setText(reviewIFiles[i].getProjectRelativePath().toString());
115        }
116      }
117    }
