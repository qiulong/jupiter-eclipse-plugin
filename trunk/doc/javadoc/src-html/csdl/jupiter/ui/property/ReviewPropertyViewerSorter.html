001    package csdl.jupiter.ui.property;
002    
003    import java.util.HashMap;
004    import java.util.Map;
005    
006    import org.eclipse.jface.viewers.Viewer;
007    import org.eclipse.jface.viewers.ViewerSorter;
008    
009    import csdl.jupiter.model.review.ReviewId;
010    
011    /**
012     * 
013     * @author Takuya Yamashita
014     * @version $Id: ReviewPropertyViewerSorter.java,v 1.1 2004/08/19 12:40:32 takuyay Exp $
015     */
016    public class ReviewPropertyViewerSorter  {
017      
018      private static Map viewerSorterMap = new HashMap();
019      private static boolean isReverse;
020      /**
021       * The viewer sorter which sorts by review ID.
022       */
023      public static final ViewerSorter REVIEW_ID = new ViewerSorter() {
024        /**
025         * @see org.eclipse.jface.viewers.ViewerSorter#compare(org.eclipse.jface.viewers.Viewer, 
026         * java.lang.Object, java.lang.Object)
027         */
028        public int compare(Viewer viewer, Object objectElement1, Object objectElement2) {
029          if ((objectElement1 instanceof ReviewId) && (objectElement2 instanceof ReviewId)) {
030            ReviewId reviewId1 = (ReviewId) objectElement1;
031            ReviewId reviewId2 = (ReviewId) objectElement2;
032            if (isReverse) {
033              return reviewId2.getReviewId().compareTo(reviewId1.getReviewId());
034            }
035            else {
036              return reviewId1.getReviewId().compareTo(reviewId2.getReviewId());
037            }
038          }
039          else {
040            return super.compare(viewer, objectElement1, objectElement2);
041          }
042          
043        }
044      };
045      
046      /**
047       * The viewer sorter which sorts by description.
048       */
049      public static final ViewerSorter DESCRIPTION = new ViewerSorter() {
050        /**
051         * @see org.eclipse.jface.viewers.ViewerSorter#compare(org.eclipse.jface.viewers.Viewer, 
052         * java.lang.Object, java.lang.Object)
053         */
054        public int compare(Viewer viewer, Object objectElement1, Object objectElement2) {
055          if ((objectElement1 instanceof ReviewId) && (objectElement2 instanceof ReviewId)) {
056            ReviewId reviewId1 = (ReviewId) objectElement1;
057            ReviewId reviewId2 = (ReviewId) objectElement2;
058            if (isReverse) {
059              return reviewId2.getDescription().compareTo(reviewId1.getDescription());
060            }
061            else {
062              return reviewId1.getDescription().compareTo(reviewId2.getDescription());
063            }
064          }
065          else {
066            return super.compare(viewer, objectElement1, objectElement2);
067          }
068        }
069      }; 
070      
071      /**
072       * The viewer sorter which sorts by creation date.
073       */
074      public static final ViewerSorter DATE = new ViewerSorter() {
075        /**
076         * @see org.eclipse.jface.viewers.ViewerSorter#compare(org.eclipse.jface.viewers.Viewer, 
077         * java.lang.Object, java.lang.Object)
078         */
079        public int compare(Viewer viewer, Object objectElement1, Object objectElement2) {
080          if ((objectElement1 instanceof ReviewId) && (objectElement2 instanceof ReviewId)) {
081            ReviewId reviewId1 = (ReviewId) objectElement1;
082            ReviewId reviewId2 = (ReviewId) objectElement2;
083            if (isReverse) {
084              return reviewId1.getDate().compareTo(reviewId2.getDate());
085            }
086            else {
087              return reviewId2.getDate().compareTo(reviewId1.getDate());         
088            }
089          }
090          else {
091            return super.compare(viewer, objectElement1, objectElement2);
092          }
093        }
094      }; 
095      
096      /**
097       * Checks if the viewer sorters are reverse mode. Returns <code>true</code> if the sorters are
098       * reverse mode. Returns <code>false</code> otherwise.
099       * @return Returns <code>true</code> if the sorter is
100       * reverse mode. Returns <code>false</code> otherwise.
101       */
102      public static boolean isReverse() {
103        return isReverse;
104      }
105      
106      /**
107       * Sets the reverse status for viewer sorters. Sets <code>true</code> if the sorters are to be
108       * reverse mode. Sets <code>false</code> otherwise.
109       * @param reverse Sets <code>true</code> if the sorters are to be reverse mode. 
110       * Sets <code>false</code> otherwise.
111       */
112      public static void setReverse(boolean reverse) {
113        isReverse = reverse;
114      }
115      
116      static {
117        viewerSorterMap.put(ReviewPropertyPage.COLUMN_REVIEW_ID_KEY, REVIEW_ID);
118        viewerSorterMap.put(ReviewPropertyPage.COLUMN_DESCRIPTION_KEY, DESCRIPTION);
119        viewerSorterMap.put(ReviewPropertyPage.COLUMN_DATE_KEY, DATE);
120      }
121      
122      /**
123       * Gets the viewer sorter by the column name key.
124       * @param columnKey the coumn name key to get the assocaited viewer sorter.
125       * @return the viewer sorter by the column name key.
126       */
127      public static ViewerSorter getViewerSorter(String columnKey) {
128        return (ViewerSorter) viewerSorterMap.get(columnKey);
129      }
130    }
