053    /**
054     * Provides an action triggered when Code Review pop up menu on the Compilation Unit is selected. Do
055     * the action that the code review editor is opened to edit comments with class name, method name,
056     * and line number if applicable.
057     * 
058     * @author Takuya Yamashita
059     * @version $Id: ReviewAdditionActionDelegate.java,v 1.26 2005/03/28 14:39:31 takuyay Exp $
060     */
061    public class ReviewAdditionActionDelegate implements IEditorActionDelegate {
062      /** Jupiter logger */
063      private JupiterLogger log = JupiterLogger.getLogger();
064    
065      /** The target file path from a project */
066      private String targetFilePath = "";
067      /** The line number to be recored in the opened editor. */
068      private String lineNumber = "";
069      /** The selectedText when code review action is invoked. */
070      private String selectedText = "";
071      private ITextSelection texSelection;
072    
073      /**
074       * Runs logic implementation when the menu, whose this implementing target is defined in the
075       * plugin.xml, is selected. See the plugin.xml file. Clients should not call this method.
076       * 
077       * @param action The action proxy that handles the presentation portion of the action
078       * @see org.eclipse.ui.IActionDelegate#run(org.eclipse.jface.action.IAction)
079       */
080      public void run(IAction action) { 
081        boolean isDetermined = determineProjectReviewIdReviewerId();
082        if (!isDetermined) {
083          return;
084        }
085        IFile selectedFile = FileResource.getSelectedIFile();  
086        this.targetFilePath = (selectedFile != null) 
087                                 ? selectedFile.getProjectRelativePath().toString() : "";
088        
089        // Creates ReviewIssue instance, and fill it into editor view.
090        ReviewModel reviewModel = ReviewModel.getInstance();
091        IProject project = reviewModel.getProjectManager().getProject();
092        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
093        ReviewerId reviewerId = reviewModel.getReviewerIdManager().getReviewerId();
094        IFile iReviewFile = FileResource.getReviewFile(project, reviewId, reviewerId);
095        if (iReviewFile == null) {
096          String titleKey = "ReviewDialog.noReviewFileDetermined.simpleConfirm.messageDialog.title";
097          String title = ReviewI18n.getString(titleKey);
098          String messageKey = "ReviewDialog.noReviewFileDetermined.simpleConfirm.messageDialog.message";
099          String message = ReviewI18n.getString(messageKey);
100          ReviewDialog.openSimpleComfirmMessageDialog(title, message);
101          log.debug(message);
102          return;
103        }
104        // check file written permission
105        if (iReviewFile.isReadOnly()) {
106            String message = "Review file " + iReviewFile + " is readonly. Please make it writable to" +
107                             " save your issues.";
108            ReviewDialog.openSimpleComfirmMessageDialog("Review Management", message);
109        }
110        
111        try {
112          ReviewEditorView editorView = ReviewEditorView.bringViewToTop();
113          editorView.setEnable(true);
114          String line = getLineNumber();
115          if (!targetFilePath.equals("")) {
116            ReviewIssue reviewIssue = new ReviewIssue(new Date(), new Date(), 
117              reviewerId.getReviewerId(), "", targetFilePath, line, 
118              (Type) TypeKeyManager.getInstance(project, reviewId).getItem(0),
119              (Severity) SeverityKeyManager.getInstance(project, reviewId).getItem(0),
120              "", this.selectedText, "", "",
121              (Resolution) ResolutionKeyManager.getInstance(project, reviewId).getItem(0),
122              (Status) StatusKeyManager.getInstance(project, reviewId).getItem(0),
123              iReviewFile);
124            reviewIssue.setLinkStatus(true);
125            ReviewEditorViewAction.NEXT.setEnabled(false);
126            ReviewEditorViewAction.PREVIOUS.setEnabled(false);
127            editorView.setReviewIssue(reviewIssue);
128          }
129          else {
130            editorView.setNewEmptyReviewIssue(iReviewFile);
131          }
132          editorView.setItemFields(project, reviewId);
133          editorView.setFocus();
134        }
135        catch (ReviewException e) {
136         log.debug(e.getMessage());
137        }
138        int type = ReviewEvent.TYPE_COMMAND;
139        int kind = ReviewEvent.KIND_ADD;
140        ReviewPlugin.getInstance().notifyListeners(type, kind);
141      }
142    
143      /**
144       * Processes opening review table and review editor. If the necessary value such as project,
145       * review id, reviewer id, phase name key are not set, prompt the proper dialog to let user
146       * specify these value before the open.
147       * 
148       * @return <code>true</code> if the project, review id, and reviewer id, were determined.
149       */
150      public static boolean determineProjectReviewIdReviewerId() {
151        boolean isReviewSelectionWizardInvoked = false;
152        IProject project = FileResource.getActiveProject();
153        // assertion project should not be null.
154        if (project == null) {
155          ReviewDialog.processNonProjectSelectionDialog();
156          return false;
157        }
158            
159        // assertion review id should not be null.
160        ReviewModel reviewModel = ReviewModel.getInstance();
161        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
162        String reviewPhaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
163        IProject reviewModelProject = reviewModel.getProjectManager().getProject();
164        reviewModelProject = (reviewModelProject != null) ? reviewModelProject : project;
165        if (reviewId == null 
166            || (reviewId != null && !isActiveEditorInCurrentProject(reviewModelProject))) {
167          int result = ReviewDialog.processReviewIdSelectionWizardDialog(reviewPhaseNameKey,
168                                                                         project, false);
169          if (result == WizardDialog.CANCEL) {
170            return false;
171          }
172          isReviewSelectionWizardInvoked = true;
173        }
174        
175        // update review Id.
176        reviewModel = ReviewModel.getInstance();
177        project = reviewModel.getProjectManager().getProject();
178        reviewId = reviewModel.getReviewIdManager().getReviewId();
179        ReviewerId reviewerId = reviewModel.getReviewerIdManager().getReviewerId();
180        // assertion review file should not be null.
181        IFile iReviewFile = FileResource.getReviewFile(project, reviewId, reviewerId);
182        if (iReviewFile == null) {
183          String titleKey = "ReviewDialog.noReviewFileDetermined.simpleConfirm.messageDialog.title";
184          String title = ReviewI18n.getString(titleKey);
185          String messageKey = "ReviewDialog.noReviewFileDetermined.simpleConfirm.messageDialog.message";
186          String message = ReviewI18n.getString(messageKey);
187          ReviewDialog.openSimpleComfirmMessageDialog(title, message);
188          return false;
189        }
190        
191        // fill review issues into model.
192        ReviewIssueModelManager reviewIssueModelManager = ReviewIssueModelManager.getInstance();
193        ReviewPlugin plugin = ReviewPlugin.getInstance();
194        IPreferenceStore store = plugin.getPreferenceStore();
195        if (isReviewSelectionWizardInvoked) {
196          ReviewIssueModel reviewIssueModel = reviewIssueModelManager.getModel(project, reviewId);
197          reviewIssueModel.notifyListeners(ReviewIssueModelEvent.MERGE);
198          // fill the specified phase's column data into model
199          ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
200          ColumnDataModel columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKey);
201    
202          ReviewTableView view = ReviewTableView.getActiveView();
203          // set column data model into view.
204          // null happens when the view is not opened yet after Eclipse startup.
205          if (view == null) {
206            view = ReviewTableView.bringViewToTop();
207          }
208          view.createColumns(columnDataModel);
209          PropertyResource propertyResource = PropertyResource.getInstance(project, true);
210          String reviewIdString = reviewId.getReviewId();
211          ReviewResource reviewResource = propertyResource.getReviewResource(reviewIdString, true);
212          if (reviewResource != null) {
213            FilterPhase filterPhase = reviewResource.getFilterPhase(reviewPhaseNameKey);
214            view.setFilterStatus(filterPhase.isEnabled());
215          }
216        }
217        else {
218          reviewIssueModelManager.getCurrentModel();
219        }
220        return true;
221      }

223      /**
224       * Gets the line number of the active file if any.
225       * 
226       * @return the line number of the active file if any.
227       */
228      public String getLineNumber() {
229        ReviewPlugin plugin = ReviewPlugin.getInstance();
230        IWorkbenchWindow window = plugin.getWorkbench().getActiveWorkbenchWindow();
231        if (window != null) {
232          IWorkbenchPage page = window.getActivePage();
233          IEditorPart activeEditorPart = page.getActiveEditor();
234          if (activeEditorPart != null) {
235            IEditorSite editorSite = page.getActiveEditor().getEditorSite();
236            ISelectionProvider selectionProvider = editorSite.getSelectionProvider();
237            if (selectionProvider != null) {
238              ISelection selection = selectionProvider.getSelection();
239              if (selection instanceof ITextSelection) {
240                this.lineNumber = String.valueOf(((ITextSelection) selection).getStartLine() + 1);
241              }
242            }
243          }
244        }
245        return this.lineNumber;
246      }

248      /**
249       * Checks if the active editor belongs to the current project.
250       * 
251       * @param currentProject the <code>IProject</code>.
252       * @return <code>true</code> if the active editor belongs to the current project.
253       *         <code>false</code> otherwise.
254       */
255      private static boolean isActiveEditorInCurrentProject(IProject currentProject) {
256        return (currentProject.getName().equals(FileResource.getActiveProject().getName()));
257      }

259      /**
260       * Notifies this action delegate that the selection in the workbench has changed. Gets a fully
261       * qualified class name, method name, line number if applicable. Clients should not call this
262       * method.
263       * 
264       * @see org.eclipse.ui.IActionDelegate #selectionChanged(org.eclipse.jface.action.IAction,
265       *      org.eclipse.jface.viewers.ISelection)
266       */
267      public void selectionChanged(IAction action, ISelection selection) {
268        if (selection instanceof ITextSelection) {
269          ITextSelection textSelection = (ITextSelection) selection;
270          this.texSelection = textSelection;
271          this.lineNumber = String.valueOf(textSelection.getStartLine() + 1);
272          this.selectedText = textSelection.getText();
273        }
274        IFile selectedFile = FileResource.getSelectedIFile();
275        this.targetFilePath = (selectedFile != null) 
276                                 ? selectedFile.getProjectRelativePath().toString() : "";
277      }

279      /**
280       * Gets the <code>ICompilationUnit</code> instance. Returns <code>null</code> if corresponding
281       * compilation unit does not exist.
282       * 
283       * @param file the <code>File</code> instance to convert to the compilation unit.
284       * @return the <code>ICompilationUnit</code> instance.
285       */
286      private ICompilationUnit getCompilationUnit(IFile file) {
287        if ((file != null) && file.getLocation().toString().endsWith(".java")) {
288          return JavaCore.createCompilationUnitFrom(file);
289        }
290        else {
291          return null;
292        }
293      }
