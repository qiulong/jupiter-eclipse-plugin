001    package csdl.jupiter.ui.perspective;
002    
003    import org.eclipse.debug.ui.IDebugUIConstants;
004    import org.eclipse.jdt.ui.JavaUI;
005    import org.eclipse.search.ui.NewSearchUI;
006    import org.eclipse.ui.IFolderLayout;
007    import org.eclipse.ui.IPageLayout;
008    import org.eclipse.ui.IPerspectiveFactory;
009    import org.eclipse.ui.IWorkbench;
010    import org.eclipse.ui.IWorkbenchWindow;
011    import org.eclipse.ui.WorkbenchException;
012    import org.eclipse.ui.console.IConsoleConstants;
013    import org.eclipse.ui.progress.IProgressConstants;
014    
015    import csdl.jupiter.ReviewPlugin;
016    import csdl.jupiter.util.JupiterLogger;
017    
018    /**
019     * Provides a code review perspective.
020     * @author Takuya Yamashita
021     * @version $Id: ReviewPerspectiveFactory.java,v 1.6 2005/03/25 19:07:46 takuyay Exp $
022     */
023    public class ReviewPerspectiveFactory implements IPerspectiveFactory {
024      /** jupiter perspective ID. */
025      public static final String ID = "csdl.jupiter.ui.perspective";
026      
027      private static final String ID_REVIEW_EDITOR_VIEW = "csdl.jupiter.ui.view.editor";
028      private static final String ID_REVIEW_TABLE_VIEW = "csdl.jupiter.ui.view.table";
029      private static final String PACKAGE_EXPLORER_ID = "org.eclipse.jdt.ui.PackageExplorer";
030      private static final String BOTTOM = "bottom";
031    
032      /** Jupiter logger */
033      private static JupiterLogger log = JupiterLogger.getLogger();
034    
035      /**
036       * @see org.eclipse.ui.IPerspectiveFactory#createInitialLayout(org.eclipse.ui.IPageLayout)
037       */
038      public void createInitialLayout(IPageLayout layout) {
039        defineLayout(layout);
040      }
041    
042      /**
043       * Defines layout.
044       * @param layout the layout.
045       */
046      private void defineLayout(IPageLayout layout) {
047        String editorArea = layout.getEditorArea();  
048        
049        IFolderLayout bottom = layout.createFolder(BOTTOM, IPageLayout.BOTTOM, (float) 0.70,
050                                                  editorArea);
051        bottom.addView(ID_REVIEW_TABLE_VIEW);
052        IFolderLayout bottomRight = layout.createFolder("bottomRight", IPageLayout.RIGHT, (float) 0.50,
053                                                         BOTTOM);
054        bottomRight.addView(ID_REVIEW_EDITOR_VIEW);
055        bottomRight.addPlaceholder(IPageLayout.ID_PROBLEM_VIEW);
056        bottomRight.addPlaceholder(JavaUI.ID_JAVADOC_VIEW);
057        bottomRight.addPlaceholder(JavaUI.ID_SOURCE_VIEW);
058        bottomRight.addPlaceholder(NewSearchUI.SEARCH_VIEW_ID);
059        bottomRight.addPlaceholder(IConsoleConstants.ID_CONSOLE_VIEW);
060        bottomRight.addPlaceholder(IPageLayout.ID_BOOKMARKS);
061        bottomRight.addPlaceholder(IProgressConstants.PROGRESS_VIEW_ID);
062        
063        IFolderLayout folder = layout.createFolder("topLeft", IPageLayout.LEFT, (float) 0.25,
064                                                   editorArea);
065        folder.addView(JavaUI.ID_PACKAGES);
066        folder.addView(IPageLayout.ID_RES_NAV);
067        folder.addPlaceholder(JavaUI.ID_TYPE_HIERARCHY);
068        
069        layout.addView(IPageLayout.ID_OUTLINE, IPageLayout.RIGHT, (float) 0.75, editorArea);
070        
071        layout.addActionSet(IDebugUIConstants.LAUNCH_ACTION_SET);
072        layout.addActionSet(JavaUI.ID_ACTION_SET);
073        layout.addActionSet(JavaUI.ID_ELEMENT_CREATION_ACTION_SET);
074        layout.addActionSet(IPageLayout.ID_NAVIGATE_ACTION_SET);
075        
076        // views - java
077        layout.addShowViewShortcut(JavaUI.ID_PACKAGES);
078        layout.addShowViewShortcut(JavaUI.ID_TYPE_HIERARCHY);
079        layout.addShowViewShortcut(JavaUI.ID_SOURCE_VIEW);
080        layout.addShowViewShortcut(JavaUI.ID_JAVADOC_VIEW);
081      }
082      
083      /**
084       * Switches to the review perspective.
085       */
086      public static void showPerspective() {
087        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
088        IWorkbenchWindow window = workbench.getActiveWorkbenchWindow();
089        if (window != null) {
090          try {
091            workbench.showPerspective(ReviewPerspectiveFactory.ID, window);
092          }
093          catch (WorkbenchException e) {
094            log.error(e);
095          }
096        }
097      }
098    }
