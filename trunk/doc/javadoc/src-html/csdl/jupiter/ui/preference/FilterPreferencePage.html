110      /**
111       * Creates the Jupiter preference contents.
112       *
113       * @param parent the parent composite
114       *
115       * @return the new control
116       */
117      protected Control createContents(Composite parent) {
118        Composite top = createGeneralComposite(parent);
119        createLabelContent(top);
120        createFilterPreferenceGroup(top);
121        // Loads the preference store setting.
122        loadFieldEditors(false);
123        loadComposites();
124        // Initializes the enable setting for field editors and composites.
125        initializeEnableSetting(this.store.getBoolean(ENABLE_FILTER_STORE_KEY));
126        return top;
127      }

160      /**
161       * Initializes the all fields (either field editor or composite) to be either enabled or
162       * disabled, depending upon the <code>isEnabled</code> parameter status.
163       *
164       * @param isEnabled <code>true</code> if all fields are enabled.
165       */
166      private void initializeEnableSetting(boolean isEnabled) {
167        for (Iterator i = fields.iterator(); i.hasNext();) {
168          Object[] object = (Object[]) i.next();
169    
170          // Excludes enable filter check box.
171          if (object.length >= 2) {
172            FieldEditor fieldEditor = (FieldEditor) object[0];
173            Composite parent = (Composite) object[1];
174            fieldEditor.setEnabled(isEnabled, parent);
175            // if a filed editor or composite is hooked to the filedEditor property listener.
176            if (object.length >= 3) {
177              if (!isEnabled
178                  || (isEnabled && (fieldEditor instanceof BooleanFieldEditor)
179                  && ((BooleanFieldEditor) fieldEditor).getBooleanValue())) {
180                ((Scrollable) object[2]).setEnabled(isEnabled);
181              }
182            }
183          }
184        }
185      }

203       * Creates the enable filter check box sub group.
204       *
205       * @param parent the composite to be hooked.
206       */
207      private void createEnabledFilterSubGroup(Composite parent) {
208        // Creates check box field sub group.
209        Composite checkBoxEnableFieldSubGroup = new Composite(parent, SWT.NONE);
210        GridLayout checkBoxEnableLayout = new GridLayout();
211        checkBoxEnableLayout.numColumns = 2;
212        checkBoxEnableFieldSubGroup.setLayout(checkBoxEnableLayout);
213        Composite filterEnabledComposite = new Composite(checkBoxEnableFieldSubGroup, SWT.NONE);
214        String filterLabelEnabledKey = "FilterPreferencePage.filter.label.enabled";
215        this.filterEnabledField = new BooleanFieldEditor(ENABLE_FILTER_STORE_KEY,
216                                                      ReviewI18n.getString(filterLabelEnabledKey),
217                                                      filterEnabledComposite);
218        filterEnabledField.setPropertyChangeListener(new IPropertyChangeListener() {
219            public void propertyChange(PropertyChangeEvent event) {
220              initializeEnableSetting(new Boolean(event.getNewValue().toString()).booleanValue());
221            }
222          });
223        fields.add(new Object[] {filterEnabledField});
224      }

522      /**
523       * Restores all field editor's values.
524       *
525       * @param isDefautLoaded <code>true</code> if default field editors are loaded.
526       *        <code>false</code> otherwise.
527       */
528      private void loadFieldEditors(boolean isDefautLoaded) {
529        for (Iterator i = this.fields.iterator(); i.hasNext();) {
530          Object[] object = (Object[]) i.next();
531          FieldEditor fieldEditor = (FieldEditor) object[0];
532          fieldEditor.setPreferenceStore(this.store);
533          if (isDefautLoaded) {
534            fieldEditor.loadDefault();
535          }
536          else {
537            fieldEditor.load();
538          }
539        }
540      }

542      /**
543       * Restores all necessary composite's value.
544       */
545      private void loadComposites() {
546        String interval = this.store.getString(FILTER_INTERVAL_TEXT_KEY);
547        String severity = this.store.getString(FILTER_SEVERITY_COMBO_KEY);
548        String status = this.store.getString(FILTER_STATUS_COMBO_KEY);
549        String type = this.store.getString(FILTER_TYPE_COMBO_KEY);
550        String resolution = this.store.getString(FILTER_RESOLUTION_COMBO_KEY);
551        String assignedTo = this.store.getString(FILTER_ASSIGNED_TO_COMBO_KEY);
552        String reviewer = this.store.getString(FILTER_REVIEWER_COMBO_KEY);
553        String file = this.store.getString(FILTER_FILE_COMBO_KEY);
554        this.filterIntervalText.setText(interval);
555        this.filterSeverityCombo.select(this.filterSeverityCombo.indexOf(severity));
556        this.filterStatusCombo.select(this.filterStatusCombo.indexOf(status));
557        this.filterTypeCombo.select(this.filterTypeCombo.indexOf(type));
558        this.filterResolutionCombo.select(this.filterResolutionCombo.indexOf(resolution));
559        this.filterAssignedToCombo.select(this.filterAssignedToCombo.indexOf(assignedTo));
560        this.filterReviewerCombo.select(this.filterReviewerCombo.indexOf(reviewer));
561        this.filterFileCombo.select(this.filterFileCombo.indexOf(file));
562      }

576      /**
577       * Stores all <code>Composites</code> or its sub class instance.
578       */
579      private void storeComposites() {
580        String intervalString = this.filterIntervalText.getText();
581        int interval = 0;
582        try {
583          Integer.parseInt(intervalString);
584        }
585        catch (NumberFormatException e) {
586          // set interval = 0;
587        }
588        int severityIndex = this.filterSeverityCombo.getSelectionIndex();
589        String severityLabel = (severityIndex != -1) ? this.filterSeverityCombo.getItem(severityIndex)
590                                                     : "";
591        int statusIndex = this.filterStatusCombo.getSelectionIndex();
592        String statusLabel = (statusIndex != -1) ? this.filterStatusCombo.getItem(statusIndex) : "";
593        int typeIndex = this.filterTypeCombo.getSelectionIndex();
594        String typeLabel = (typeIndex != -1) ? this.filterTypeCombo.getItem(typeIndex) : "";
595        int resolutionIndex = this.filterResolutionCombo.getSelectionIndex();
596        String resolutionLabel = (resolutionIndex != -1)
597                                  ? this.filterResolutionCombo.getItem(resolutionIndex) : "";
598        int assignedToIndex = this.filterAssignedToCombo.getSelectionIndex();
599        String assignedToLabel = "";
600        // -1 happens if no reviews are loaded to review, i.e. no assigned to exists.
601        if (assignedToIndex != -1) {
602          assignedToLabel = this.filterAssignedToCombo.getItem(assignedToIndex);
603        }
604        int reviewerIndex = this.filterReviewerCombo.getSelectionIndex();
605        // -1 happens if no reviews are loaded to review, i.e. no reviews to exists.
606        String reviewerLabel = (reviewerIndex != -1) ? this.filterReviewerCombo.getItem(reviewerIndex)
607                                                     : "";
608        int fileIndex = this.filterFileCombo.getSelectionIndex();
609        String fileLabel = (fileIndex != -1) ? this.filterFileCombo.getItem(fileIndex)
610                                                     : "";
611        ReviewModel reviewModel = ReviewModel.getInstance();
612        IProject project = reviewModel.getProjectManager().getProject();
613        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();;
614        this.store.setValue(FILTER_INTERVAL_TEXT_KEY, interval + "");
615        SeverityKeyManager severityKeyManager = SeverityKeyManager.getInstance(project, reviewId);
616        this.store.setValue(FILTER_SEVERITY_COMBO_KEY, severityKeyManager.getKey(severityLabel));
617        StatusKeyManager statusKeyManager = StatusKeyManager.getInstance(project, reviewId);
618        this.store.setValue(FILTER_STATUS_COMBO_KEY, statusKeyManager.getKey(statusLabel));
619        TypeKeyManager typeKeyManager = TypeKeyManager.getInstance(project, reviewId);
620        this.store.setValue(FILTER_TYPE_COMBO_KEY, typeKeyManager.getKey(typeLabel));
621        ResolutionKeyManager resolutionKeyManager = ResolutionKeyManager.getInstance(project, reviewId);
622        this.store.setValue(FILTER_RESOLUTION_COMBO_KEY, resolutionKeyManager.getKey(resolutionLabel));
623        this.store.setValue(FILTER_ASSIGNED_TO_COMBO_KEY, assignedToLabel);
624        this.store.setValue(FILTER_REVIEWER_COMBO_KEY, reviewerLabel);
625        this.store.setValue(FILTER_FILE_COMBO_KEY, fileLabel);
626      }

628      /**
629       * Performs to save the current data in the editors.
630       *
631       * @return <code>false</code> to abort the container's OK processing and <code>true</code> to
632       *         allow the OK to happen.
633       */
634      public boolean performOk() {
635        storeFieldEditors();
636        storeComposites();
637        ReviewTableView view = ReviewTableView.getActiveView();
638        if (view != null) {
639          boolean isFilterEnabled = this.store.getBoolean(ENABLE_FILTER_STORE_KEY);
640          view.setFilterStatus(isFilterEnabled);
641        }
642        return super.performOk();
643      }

647       * implementation of this framework method simply calls performOk to simulate the pressing of
648       * the page's OK button.
649       */
650      public void performApply() {
651        performOk();
652      }
