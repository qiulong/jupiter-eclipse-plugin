140      /**
141       * Fetches the revision.
142       * @param listener the listener.
143       * @param cvsResource the cvs resource.
144       * @param revision the revision.
145       * @param monitor the monitor.
146       * @throws CVSException if problems occur.
147       */
148      private void fetchRevision(AnnotateListener listener, ICVSResource cvsResource, String revision,
149          IProgressMonitor monitor) throws CVSException {
150    
151        monitor = Policy.monitorFor(monitor);
152        monitor.beginTask(null, 100);
153    
154        final ICVSFolder folder = cvsResource.getParent();
155        final FolderSyncInfo info = folder.getFolderSyncInfo();
156        final ICVSRepositoryLocation location = KnownRepositories.getInstance().getRepository(
157            info.getRoot());
158    
159        final Session session = new Session(location, folder, true /* output to console */);
160        session.open(Policy.subMonitorFor(monitor, 10), false /* read-only */);
161        try {
162          final Command.QuietOption quietness = CVSProviderPlugin.getPlugin().getQuietness();
163          try {
164            CVSProviderPlugin.getPlugin().setQuietness(Command.VERBOSE);
165            final Command.LocalOption[] localOption;
166            if (revision == null) {
167              localOption = Command.NO_LOCAL_OPTIONS;
168            }
169            else {
170              localOption = new Command.LocalOption[1];
171              localOption[0] = Annotate.makeRevisionOption(revision);
172            }
173            final IStatus status = Command.ANNOTATE.execute(session, Command.NO_GLOBAL_OPTIONS,
174                localOption, new ICVSResource[]{cvsResource
175                }, listener, Policy.subMonitorFor(monitor, 90));
176            if (status.getCode() == CVSStatus.SERVER_ERROR) {
177              throw new CVSServerException(status);
178            }
179          }
180          finally {
181            CVSProviderPlugin.getPlugin().setQuietness(quietness);
182            monitor.done();
183          }
184        }
185        finally {
186          session.close();
187        }
188      }
189

191       * @see org.eclipse.team.internal.ccvs.ui.operations.CVSOperation#getTaskName()
192       */
193      protected String getTaskName() {
194        return "Revision Task";
195      }
