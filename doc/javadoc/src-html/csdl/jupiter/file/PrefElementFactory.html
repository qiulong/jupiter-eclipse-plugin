001    package csdl.jupiter.file;
002    
003    import java.util.ArrayList;
004    import java.util.Iterator;
005    import java.util.List;
006    
007    import org.jdom.Element;
008    
009    /**
010     * Provides preference element wrapper class.
011     * @author Takuya Yamashita
012     * @version $Id: PrefElementFactory.java,v 1.2 2004/08/20 16:48:54 takuyay Exp $
013     */
014    public class PrefElementFactory {
015      private static PrefElementFactory theInstance = new PrefElementFactory();
016      private Element preferenceElement;
017      
018      /**
019       * Prohibits clients from instantiating this.
020       */
021      private PrefElementFactory() {
022      }
023      
024      /**
025       * Gets the singleton instance.
026       * @param preferenceElement the preference element.
027       * @return the singleton instance.
028       */
029      public static PrefElementFactory getInstance(Element preferenceElement) {
030        theInstance.setPreferenceElement(preferenceElement);
031        return theInstance;
032      }
033      
034      /**
035       * Sets the 'Preference' <code>Element</code>.
036       * @param preferenceElement the 'Preference' <code>Element</code>.
037       */
038      private void setPreferenceElement(Element preferenceElement) {
039        this.preferenceElement = preferenceElement;
040      }
041      
042      /**
043       * Gets the 'Preference' <code>Element</code>.
044       * @return the 'Preference' <code>Element</code>.
045       */
046      public Element getPreferenceElement() {
047        return this.preferenceElement;
048      }
049      
050      /**
051       * Gets the 'UpdateUrl' <code>Element</code>.
052       * @return the 'UpdateUrl' <code>Element</code>.
053       */
054      public Element getUpdateUrlElement() {
055        Element generalElement = this.preferenceElement.getChild(PrefConstraints.ELEMENT_GENERAL);
056        Element updateUrlElement = generalElement.getChild(PrefConstraints.ELEMENT_UPDATE_URL);
057        return updateUrlElement;
058      }
059        
060      /**
061       * Gets the 'EnableUpdate' <code>Element</code>.
062       * @return the 'EnableUpdate' <code>Element</code>.
063       */
064      public Element getEnableUpdateElement() {
065        Element generalElement = this.preferenceElement.getChild(PrefConstraints.ELEMENT_GENERAL);
066        Element enableUpdateElement = generalElement.getChild(PrefConstraints.ELEMENT_ENABLE_UPDATE);
067        return enableUpdateElement;
068      }
069      
070      /**
071       * Gets the 'EnableFilter' <code>Element</code>.
072       * @return the 'EnableFilter' <code>Element</code>.
073       */
074      public Element getEnableFilterElement() {
075        Element generalElement = this.preferenceElement.getChild(PrefConstraints.ELEMENT_GENERAL);
076        Element enableFilterElement = generalElement.getChild(PrefConstraints.ELEMENT_ENABLE_FILTER);
077        return enableFilterElement;
078      }
079      
080      /**
081       * Gets the 'View' <code>Element</code>.
082       * @return the 'View' <code>Element</code>.
083       */
084      public Element getViewElement() {
085        return this.preferenceElement.getChild(PrefConstraints.ELEMENT_VIEW);
086      }
087      
088      /**
089       * Gets the 'Phase' <code>Element</code> if any. Returns <code>null</code> if not found.
090       * @param phaseNameKey the phase name key.
091       * @return the 'Phase' <code>Element</code> if any. Returns <code>null</code> if not found.
092       */
093      public Element getPhaseElement(String phaseNameKey) {
094        Element viewElement = getViewElement();
095        List phaseElementList = viewElement.getChildren(PrefConstraints.ELEMENT_PHASE);
096        for (Iterator i = phaseElementList.iterator(); i.hasNext();) {
097          Element phaseElement = (Element) i.next();
098          if (phaseElement.getAttributeValue(PrefConstraints.ATTRIBUTE_NAME).equals(phaseNameKey)) {
099            return phaseElement;
100          }
101        }
102        return null;
103      }
104      
105      /**
106       * Gets the list of the 'Phase' <code>Element</code> instances.
107       * @return the list of the 'Phase' <code>Element</code> instances.
108       */
109      public List getPhaseElementList() {
110        Element viewElement = this.preferenceElement.getChild(PrefConstraints.ELEMENT_VIEW);
111        return viewElement.getChildren(PrefConstraints.ELEMENT_PHASE);
112      }
113      
114      /**
115       * Gets the list of the 'ColumnEntry' <code>Element</code> instances given the phase name key.
116       * @param phaseNameKey the phase name key.
117       * @return the list of the 'ColumnEntry' <code>Element</code> instances given the phase name key.
118       */
119      public List getColumnEntryElementList(String phaseNameKey) {
120        Element viewElement = this.preferenceElement.getChild(PrefConstraints.ELEMENT_VIEW);
121        List phaseElementList = viewElement.getChildren(PrefConstraints.ELEMENT_PHASE);
122        List columnHeaderElementList = new ArrayList();
123        for (Iterator i = phaseElementList.iterator(); i.hasNext();) {
124          Element phaseElement = (Element) i.next();
125          if (phaseElement.getAttributeValue(PrefConstraints.ATTRIBUTE_NAME).equals(phaseNameKey)) {
126            String columnHeaderElementName = PrefConstraints.ELEMENT_COLUMN_ENTRY;
127            columnHeaderElementList = phaseElement.getChildren(columnHeaderElementName);
128          }
129        }
130        return columnHeaderElementList;
131      }
132    }
