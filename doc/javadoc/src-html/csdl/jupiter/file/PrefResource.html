001    package csdl.jupiter.file;
002    
003    import java.util.ArrayList;
004    import java.util.Iterator;
005    import java.util.List;
006    
007    import org.eclipse.jface.viewers.ColumnPixelData;
008    import org.jdom.Element;
009    
010    import csdl.jupiter.ReviewException;
011    import csdl.jupiter.ReviewI18n;
012    import csdl.jupiter.model.columndata.ColumnData;
013    import csdl.jupiter.model.columndata.ColumnDataModel;
014    import csdl.jupiter.util.JupiterLogger;
015    
016    /**
017     * Provides data-filled manager objects. Clients can pass an object to get proper data.
018     *
019     * @author Takuya Yamashita
020     * @version $Id: PrefResource.java,v 1.6 2005/03/28 14:39:52 takuyay Exp $
021     */
022    public class PrefResource {
023      /** Jupiter logger */
024      private JupiterLogger log = JupiterLogger.getLogger();
025      /** The singleton instance. */
026      private static PrefResource theInstance;
027      /** The PrefElementFactory instance. */
028      private PrefElementFactory factory;
029      
030      /**
031       * Prohibits the clients' instantiation.
032       */
033      private PrefResource() {
034        this.factory = PrefElementFactory.getInstance(PrefXmlSerializer.loadPreferenceElement());
035      }
036      
037      /**
038       * Gets the instance of the <code>PrefResource</code>.
039       * @return the instance of the <code>PrefResource</code>.
040       */
041      public static PrefResource getInstance() {
042        if (theInstance == null) {
043          theInstance = new PrefResource();
044        }
045        return theInstance;
046      }
047    
048      /**
049       * Stores the <code>ColumnDataManager</code> instance into configuration XML file.
050       * @param reviewPhaseNameKey the review phase name key.
051       * @param columnDataModel the <code>ColumnDataManager</code> instance.
052       */
053      public void storeColumnDataModel(String reviewPhaseNameKey, ColumnDataModel columnDataModel) {
054        Element phaseElement = this.factory.getPhaseElement(reviewPhaseNameKey);
055        phaseElement.removeContent();
056        List columnDataList = new ArrayList();
057        ColumnData[] columnDataArray = columnDataModel.getAllColumnDataArray();
058        for (int i = 0; i < columnDataArray.length; i++) {
059          Element columnHeaderElement = new Element(PrefConstraints.ELEMENT_COLUMN_ENTRY);
060          int width = columnDataArray[i].getColumnPixelData().width;
061          boolean resizable = columnDataArray[i].getColumnPixelData().resizable;
062          String columnNameKey = columnDataArray[i].getColumnNameKey();
063          columnHeaderElement.setAttribute(PrefConstraints.ATTRIBUTE_NAME, columnNameKey);
064          columnHeaderElement.setAttribute(PrefConstraints.ATTRIBUTE_WIDTH, width + "");
065          columnHeaderElement.setAttribute(PrefConstraints.ATTRIBUTE_RESIZALE, resizable + "");
066          boolean isEnabled = columnDataArray[i].isEnabled();
067          columnHeaderElement.setAttribute(PrefConstraints.ATTRIBUTE_ENABLE, isEnabled + "");
068          phaseElement.addContent(columnHeaderElement);
069        }
070        try {
071          PrefXmlSerializer.serializeDocument(phaseElement.getDocument());
072        }
073        catch (ReviewException e) {
074          log.error(e);
075        }
076      }
077    
078      /**
079       * Gets the default review phase name.
080       * @return the default review phase name.
081       */
082      public String getDefaultPhaseNameKey() {
083        return this.factory.getViewElement().getAttributeValue(PrefConstraints.ATTRIBUTE_DEFAULT);
084      }
085      
086      /**
087       * Gets the array of <code>String</code> review phase names or phase name keys depending upon
088       * the argument option. Returns the phase name keys if isKey is true. 
089       * Otherwise returns the phase names
090       * @param isKey true if returning values are array of phase name keys.
091       * @return the array of <code>String</code> review phase name keys or phase names.
092       */
093      public String[] getPhaseArray(boolean isKey) {
094        List phaseElementList = this.factory.getPhaseElementList();
095        List phaseList = new ArrayList();
096        for (Iterator i = phaseElementList.iterator(); i.hasNext();) {
097          Element phaseElement = (Element) i.next();
098          String phaseNameKey = phaseElement.getAttributeValue(PrefConstraints.ATTRIBUTE_NAME);
099          String phase = (isKey) ? phaseNameKey : ReviewI18n.getString(phaseNameKey);
100          phaseList.add(phase);
101        }
102        return (String[]) phaseList.toArray(new String[] {});
103      }
104      
105      /**
106       * Gets the update URL string.
107       * @return the update URL string.
108       */
109      public String getUpdateUrl() {
110        return this.factory.getUpdateUrlElement().getText();
111      }
112      
113      /**
114       * Gets the boolean enable update value.
115       * @return true if update is enabled.
116       */
117      public boolean getEnableUpdate() {
118        return new Boolean(this.factory.getEnableUpdateElement().getText()).booleanValue();
119      }
120      
121      /**
122       * Gets the boolean enable filter value.
123       * @return true if filter is enabled.
124       */
125      public boolean getEnableFilter() {
126        return new Boolean(this.factory.getEnableFilterElement().getText()).booleanValue();
127      }
128      
129      /**
130       * Loads ColumnData instances to the column data model.
131       * @param phaseNameKey the phase name key.
132       * @param columnDataModel the column data model.
133       */
134      public void loadColumnData(String phaseNameKey, ColumnDataModel columnDataModel) {
135        columnDataModel.clear();
136        columnDataModel.addAll(getColumnDataList(phaseNameKey));
137      }
138      
139      /**
140       * Gets the list of the ColumnData instances given the phase name key.
141       * @param phaseNameKey the phase name key.
142       * @return the list of the ColumnData instances given the phase name key.
143       */
144      public List getColumnDataList(String phaseNameKey) {
145        List columnHeaderElementList = this.factory.getColumnEntryElementList(phaseNameKey);
146        List columnDataList = new ArrayList();
147        for (Iterator i = columnHeaderElementList.iterator(); i.hasNext();) {
148          Element columnHeaderElement = (Element) i.next();
149          ColumnData columnData = createColumnData(columnHeaderElement);
150          columnDataList.add(columnData);
151        }
152        return columnDataList;
153      }
154      
155      /**
156       * Creates the ColumnData instance.
157       * @param columnHeaderElement the ColumnHeader element.
158       * @return the ColumnData instance.
159       */
160      private ColumnData createColumnData(Element columnHeaderElement) {
161        String columnNameKey = columnHeaderElement.getAttributeValue(PrefConstraints.ATTRIBUTE_NAME);
162        String enableString = columnHeaderElement.getAttributeValue(PrefConstraints.ATTRIBUTE_ENABLE);
163        boolean enable = new Boolean(enableString).booleanValue();
164        String widthString = columnHeaderElement.getAttributeValue(PrefConstraints.ATTRIBUTE_WIDTH);
165        int width = Integer.parseInt(widthString);
166        String resizableAttributeName = PrefConstraints.ATTRIBUTE_RESIZALE;
167        String resizableString  = columnHeaderElement.getAttributeValue(resizableAttributeName);
168        boolean resizable = new Boolean(resizableString).booleanValue();
169        ColumnPixelData pixelData = new ColumnPixelData(width, resizable);
170        return new ColumnData(columnNameKey, pixelData, enable);
171      }
172    }
