068      /**
069       * Gets file instance from the given project and the relative path to the review file to be
070       * created. Returns <code>null</code> if problems occur on the new file creation.
071       * 
072       * @param project the active project.
073       * @param relativePath the relative path to the review file to be created.
074       * @param isReviewFile sets <code>true</code> if the creating file is review file. sets
075       *        <code>false</code> otherwise.
076       * @return the file instance which is guaranteed to exist unless <code>null</code> is returned.
077       * @throws CoreException thrown if problems occur.
078       * @throws IOException thrown if problems occur.
079       * @throws ReviewException thrown if problems occur.
080       */
081      public static IFile createIFile(IProject project, String relativePath, boolean isReviewFile)
082          throws IOException, CoreException, ReviewException {
083        if (project == null) {
084          return null;
085        }
086        IFile iFile = project.getFile(relativePath);
087        return createIFile(iFile, isReviewFile);
088      }

090      /**
091       * Creates <code>IFile</code> instance from the <code>File</code>. Sets <code>true</code>
092       * if the file is review file. Sets <code>false</code> otherwise.
093       * 
094       * @param iFile the iFile
095       * @param isReviewFile Sets <code>true</code> if the file is review file. Sets
096       *        <code>false</code> otherwise.
097       * @return the iFile
098       * @throws IOException if problems occur.
099       * @throws CoreException if problems occur.
100       * @throws ReviewException if problems occur.
101       */
102      public static IFile createIFile(IFile iFile, boolean isReviewFile) throws IOException,
103          CoreException, ReviewException {
104        File file = iFile.getLocation().toFile();
105        if (iFile.exists()) {
106          if (!file.exists()) {
107            // Makes sure the parent directories of the review file are created.
108            file.getParentFile().mkdirs();
109            file.createNewFile();
110            iFile.refreshLocal(IResource.DEPTH_ONE, null);
111            iFile.setContents(new FileInputStream(file), true, false, null);
112            // Writes root XML elements. Otherwise the org.jdom.input.JDOMParseException exception
113            // will be thrown when the XML file is read.
114            if (isReviewFile) {
115              ReviewIssueXmlSerializer.writeEmptyCodeReview(file);
116            }
117          }
118        }
119        else {
120          if (!file.exists()) {
121            // makes sure the parent directories of the review file are created.
122            file.getParentFile().mkdirs();
123            file.createNewFile();
124            iFile.refreshLocal(IResource.DEPTH_ONE, null);
125            iFile.setContents(new FileInputStream(file), true, false, null);
126            // Writes root XML elements. Otherwise the org.jdom.input.JDOMParseException exception
127            // will be thrown when the XML file is read.
128            if (isReviewFile) {
129              ReviewIssueXmlSerializer.writeEmptyCodeReview(file);
130            }
131          }
132          else {
133            iFile.create(new FileInputStream(file), true, null);
134          }
135        }
136        return iFile;
137      }

139      /**
140       * Gets the relative file name to the review XML file. Returns null if the review id is
141       * not contained in the list of config file. Throws the <code>IllegalArgumentException</code>
142       * if at least either project, reviewId, or reviewerId, or both.
143       * 
144       * @param project the project.
145       * @param reviewId the review id.
146       * @param reviewerId the reviewer id.
147       * @return the relative file name to the review XML file. Returns null if the review id is
148       * not contained in the list of config file. 
149       */
150      public static String getPathToReviewFile(IProject project, ReviewId reviewId,
151          ReviewerId reviewerId) {
152        if (project == null || reviewId == null || reviewerId == null) {
153          throw new IllegalArgumentException("argument(s) is null");
154        }
155        String relativeFolder = getPathToReviewFolder(project, reviewId);
156        if (relativeFolder != null) {
157          String fileName = reviewId.getReviewId() + "-" + reviewerId.getReviewerId();
158          return relativeFolder + "/" + fileName + ".review";
159        }
160        return null;
161      }

163      /**
164       * Gets the review <code>IFile</code> instance of the XML file. The project, review id, and
165       * reviewer id determines an unique review file. Returns null if either project, reviewId, 
166       * or reviewerId is null, or if the review id is not contained in the list of config file.
167       * 
168       * @param project the project.
169       * @param reviewId the review id.
170       * @param reviewerId the reviewer id.
171       * @return the review <code>IFile</code> instance of the XML file. 
172       * Returns null if either project, reviewId, or reviewerId is null, 
173       * or if the review id is not contained in the list of config file.
174       */
175      public static IFile getReviewFile(IProject project, ReviewId reviewId, ReviewerId reviewerId) {
176        if (project == null || reviewId == null || reviewerId == null) {
177          return null;
178        }
179        String pathToReviewFile = getPathToReviewFile(project, reviewId, reviewerId);
180        if (pathToReviewFile != null) {
181          try {
182            return FileResource.createIFile(project, pathToReviewFile, true);
183          }
184          catch (Exception e) {
185            // ignore.
186          }
187        }
188        return null;
189      }

191      /**
192       * Gets the relative review folder. Returns null if the review id is not contained in the 
193       * list of config file.
194       * 
195       * @param project the project.
196       * @param reviewId the review id.
197       * @return the relative review folder. Returns null if the review id is not contained in the 
198       * list of config file.
199       */
200      public static String getPathToReviewFolder(IProject project, ReviewId reviewId) {
201        if (project == null) {
202          throw new IllegalArgumentException("project is null");
203        }
204        if (reviewId == null) {
205          throw new IllegalArgumentException("review id is null");
206        }
207        PropertyResource propertyResource = PropertyResource.getInstance(project, false);
208        List reviewIdList = propertyResource.getReviewIdList();
209        for (Iterator i = reviewIdList.iterator(); i.hasNext();) {
210          ReviewId searchingReviewId = (ReviewId) i.next();
211          if (searchingReviewId.getReviewId().equals(reviewId.getReviewId())) {
212            return searchingReviewId.getDirectory();
213          }
214        }
215        return null;
216      }

294          return (IFile) selectedResource;
295        }
296        else {
297          IFile activeFile = getActiveFile();
298          return (activeFile != null) ? activeFile : null;
299        }

302      /**
303       * Gets the active file which is active in the editor part.
304       * 
305       * @return the active <code>IFile</code> instance.
306       */
307      public static IFile getActiveFile() {
308        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
309        IWorkbenchWindow activeWindow = workbench.getActiveWorkbenchWindow();
310        if (activeWindow != null) {
311          IWorkbenchPage activePage = activeWindow.getActivePage();
312          if (activePage != null) {
313            IEditorPart part = activePage.getActiveEditor();
314            if (part != null) {
315              IEditorInput editorInput = part.getEditorInput();
316              if (editorInput instanceof IFileEditorInput) {
317                try {
318                  return ((IFileEditorInput) editorInput).getFile();
319                }
320                // This happens when the welcome page is opened at the beginning.
321                catch (ClassCastException e) {
322                  log.debug(e.getMessage());
323                }
324              }
325            }
326          }
327        }
328        return null;
329      }

331      /**
332       * Gets the set of the <code>IFile</code> instances associated with the <code>ReviewId</code>
333       * instance in a project.
334       * 
335       * @param project the project to search.
336       * @param reviewId the review ID instance which is associated with the files.
337       * @return the set of the review <code>IFile</code> instances associated with the review ID in
338       *         the project.
339       */
340      public static IFile[] getReviewIFiles(IProject project, ReviewId reviewId) {
341        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
342        String relativePath = reviewId.getDirectory();
343        List filesList = new ArrayList();
344        // Gathers all review File instances among all opened projects.
345        File folder = project.getFile(relativePath).getLocation().toFile();
346        if (folder.exists() && folder.isDirectory()) {
347          File[] files = folder.listFiles(new FileFilter() {
348            public boolean accept(File pathname) {
349              if (pathname.getName().endsWith(".review")) {
350                return true;
351              }
352              return false;
353            }
354          });
355          for (int j = 0; j < files.length; j++) {
356            try {
357              if (ReviewIssueXmlSerializer.isReviewIdAssociatedFile(reviewId.getReviewId(), files[j])) {
358                IFile iFile = project.getFile(relativePath + "/" + files[j].getName());
359                filesList.add(iFile);
360              }
361            }
362            catch (ReviewException e) {
363              log.error(e);
364            }
365          }
366        }
367        return (IFile[]) filesList.toArray(new IFile[]{});
368      }

387       * Gets the String array of the opened and review ID contained project names in workspace.
388       * 
389       * @return the String array of the opened and review ID contained project names in workspace.
390       */
391      public static String[] getOpenedAndReviewIdContainedProjects() {
392        IProject[] projects = ResourcesPlugin.getWorkspace().getRoot().getProjects();
393        Set projectSet = new LinkedHashSet();
394        for (int i = 0; i < projects.length; i++) {
395          if (projects[i].isOpen()) {
396            PropertyResource propertyResource = PropertyResource.getInstance(projects[i], false);
397            List reviewIds = propertyResource.getReviewIdList();
398            if (reviewIds.size() > 0) {
399              projectSet.add(projects[i].getName());
400            }
401          }
402        }
403        return (String[]) projectSet.toArray(new String[]{});
404      }

406      /**
407       * Links to the source code associated with the target file if any.
408       * 
409       * @param targetIFile the target review file to open.
410       * @param lineNumber the line number of the target file.
411       */
412      public static void goToLine(IFile targetIFile, int lineNumber) {
413        if (targetIFile == null || targetIFile.equals("")) {
414          return;
415        }
416        lineNumber = (lineNumber >= 0) ?  --lineNumber : -1;
417        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
418        final IWorkbenchPage page = workbench.getActiveWorkbenchWindow().getActivePage();
419        
420        //optimization: if the active editor has the same input as the selected target file then 
421        //activate the editor
422        IEditorPart editor = page.getActiveEditor();
423        if (editor != null) {
424        IEditorInput input = editor.getEditorInput();
425          if (input instanceof IFileEditorInput) {
426            IFile file = ((IFileEditorInput) input).getFile();
427            if (targetIFile.equals(file)) {
428              page.activate(editor);
429            }
430          }
431        }
432    
433        try {
434          // if jarIFile is java file,
435          String editorId = IDE.getEditorDescriptor(targetIFile).getId();
436          editor = IDE.openEditor(page, targetIFile, editorId);
437          IEditorInput editorInput = editor.getEditorInput();
438          if (editor instanceof ITextEditor) {
439            ITextEditor textEditor = (ITextEditor) editor;
440            IDocument document = textEditor.getDocumentProvider().getDocument(editorInput);
441            IRegion lineInformation = document.getLineInformation(lineNumber);
442            textEditor.selectAndReveal(lineInformation.getOffset(), 0);
443          }
444        }
445        catch (Exception e) {
446          log.debug(e.getMessage());
447        }
448    //    ReviewMarker.updateMarkers(targetIFile);
449      }
