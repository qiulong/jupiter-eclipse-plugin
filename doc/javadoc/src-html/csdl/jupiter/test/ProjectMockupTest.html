001    package csdl.jupiter.test;
002    
003    import junit.framework.TestCase;
004    
005    import org.eclipse.core.resources.IFolder;
006    import org.eclipse.core.resources.IProject;
007    import org.eclipse.core.resources.IResource;
008    import org.eclipse.core.runtime.CoreException;
009    import org.eclipse.core.runtime.IPath;
010    import org.eclipse.jdt.core.IClasspathEntry;
011    import org.eclipse.jdt.core.IJavaProject;
012    import org.eclipse.jdt.core.IPackageFragment;
013    import org.eclipse.jdt.core.JavaCore;
014    import org.eclipse.jdt.launching.JavaRuntime;
015    
016    /**
017     * Tests the <code>TestProjectMockup</code> class.
018     * @author Takuya Yamashita
019     * @version $Id: TestProjectMockup.java,v 1.3 2005/03/25 19:07:46 takuyay Exp $
020     */
021    public class ProjectMockupTest extends TestCase {
022      
023      private ProjectMockup projectMockup;
024      private static final String TEST_JUPITER_PROJECT = "testJupiterProject";
025      private static final String OUTPUT_DIR = "bin";
026      private static final String OUTPUT_LOCATION =  "/" + TEST_JUPITER_PROJECT + "/" + OUTPUT_DIR;
027      private static final String SRC_DIR = "src";
028      private static final String TEST_PACKAGE = "csdl.jupiter.test";
029      private static final String FILE_SEPARATED_TEST_PACKAGE = "csdl/jupiter/test";
030      /**
031       * Sets up the project mock up.
032       * @throws CoreException if problems occur.
033       */
034      public void setUp() throws CoreException {
035        this.projectMockup = new ProjectMockup(TEST_JUPITER_PROJECT, OUTPUT_DIR);
036      }
037      
038      /**
039       * Tears down the project mock up.
040       * @throws CoreException if problems occur.
041       */
042      public void tearDown() throws CoreException {
043        this.projectMockup.dispose();
044      }
045      
046      /**
047       * Tests the default project which does nothing after the <code>ProjectMockup</code>
048       * instantiation.
049       * @throws CoreException if problems occur.
050       */
051      public void testDefaultProject() throws CoreException {
052        IProject project = this.projectMockup.getProject();
053        assertEquals("Checking project name.", TEST_JUPITER_PROJECT, project.getName());
054        IJavaProject javaProject = this.projectMockup.getJavaProject();
055        assertEquals("Checking java project name.", TEST_JUPITER_PROJECT, javaProject.getElementName());
056        IPath outputLocation = javaProject.getOutputLocation();
057        assertEquals("Checking output location.", OUTPUT_LOCATION, outputLocation.toString());
058        IClasspathEntry[] classpathEntries = javaProject.getRawClasspath();
059        assertEquals("Checking classpath entry size.", 1, classpathEntries.length);    
060        IClasspathEntry classpathEntry = classpathEntries[0];
061        IClasspathEntry defaultJreContainerEntry = JavaRuntime.getDefaultJREContainerEntry();
062        assertEquals("Checking classpath entry.", defaultJreContainerEntry, classpathEntry);
063        IPath classpathLocation = classpathEntry.getOutputLocation();
064        assertEquals("Checking classpath location.", null, classpathLocation);
065        String natureId = project.getDescription().getNatureIds()[0];
066        assertEquals("Checking nature id.", JavaCore.NATURE_ID, natureId);
067      }
068      
069      /**
070       * Tests the project.
071       * @throws CoreException if problems occur.
072       */
073      public void testProject() throws CoreException {
074        IPackageFragment packageFragment = this.projectMockup.createPackage(TEST_PACKAGE, SRC_DIR);
075        String fullPath = "/" + TEST_JUPITER_PROJECT + "/" + SRC_DIR + "/" 
076                          + FILE_SEPARATED_TEST_PACKAGE;
077        IResource resource = packageFragment.getCorrespondingResource();
078        assertTrue("Checking source directory resource type.", resource instanceof IFolder);
079        assertEquals("Checking the full path.", fullPath, resource.getFullPath().toString());
080      }
081        
082    
083    }
