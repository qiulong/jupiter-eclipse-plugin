001    package csdl.jupiter.model.columndata;
002    
003    import org.eclipse.jface.viewers.ColumnPixelData;
004    
005    import csdl.jupiter.ReviewI18n;
006    
007    /**
008     * Provides the column data.
009     *
010     * @author Takuya Yamashita
011     * @version $Id: ColumnData.java,v 1.2 2004/07/27 02:20:12 takuyay Exp $
012     */
013    public class ColumnData {
014      /** The key for the column name. */
015      private String columnNameKey;
016      /** The column header name. */
017      private String columnName;
018      /** The <code>ColumnPixelData</code> instance to hold the column width. */
019      private ColumnPixelData columnPixelData;
020      /** The boolean flag to see if this column is enabled. */
021      private boolean isEnabled;
022    
023      /**
024       * Instantiates the object with the column name and <code>ColumnPixelData</code> instance.
025       * Note that the default column means that the column is provided in the config.xml file so that
026       * the column name supports i18n.
027       *
028       * @param columnNameKey the key for the column header name.
029       * @param columnPixelData the <code>ColumnPixelData</code> instance to hold the column width.
030       * @param isEnabled sets <code>true</code> if this column can be enabled (visible). 
031       * <code>false</code> otherwise.
032       */
033      public ColumnData(String columnNameKey, ColumnPixelData columnPixelData, boolean isEnabled) {
034        this.columnNameKey = columnNameKey;
035        this.columnName = ReviewI18n.getString(columnNameKey);
036        this.columnPixelData = columnPixelData;
037        this.isEnabled = isEnabled;
038      }
039    
040      /**
041       * Gets the column header name.
042       *
043       * @return the column header name.
044       */
045      public String getColumnName() {
046        return this.columnName;
047      }
048      
049      /**
050       * Gets the column header name key.
051       * @return the column header name key.
052       */
053      public String getColumnNameKey() {
054        return this.columnNameKey;
055      }
056      
057      /**
058       * Sets the <code>ColumnPixelData</code> instance.
059       * @param columnPixelData the <code>ColumnPixelData</code> instance.
060       */
061      public void setColumnPixelData(ColumnPixelData columnPixelData) {
062        this.columnPixelData = columnPixelData;
063      }
064      
065      /**
066       * Sets <code>true</code> if this column can be enabled (visible). <code>false</code> otherwise.
067       * @param isEnabled sets <code>true</code> if this column can be enabled (visible).
068       *  <code>false</code> otherwise.
069       */
070      public void setEnabled(boolean isEnabled) {
071        this.isEnabled = isEnabled;
072      }
073      
074      /**
075       * Checks if this column can be enabled (visible). Returns <code>true</code> if this column
076       * can be enabled (visible). <code>false</code> otherwise.
077       * @return <code>true</code> if this column can be enabled (visible). 
078       * <code>false</code> otherwise.
079       */
080      public boolean isEnabled() {
081        return this.isEnabled;
082      }
083    
084      /**
085       * Gets the <code>ColumnPixelData</code> instance to hold the column width.
086       *
087       * @return the <code>ColumnPixelData</code> instance to hold the column width.
088       */
089      public ColumnPixelData getColumnPixelData() {
090        return this.columnPixelData;
091      }
092    }
