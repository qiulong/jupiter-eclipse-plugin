001    package csdl.jupiter.model.reviewissue;
002    
003    import java.util.Date;
004    
005    import csdl.jupiter.ReviewException;
006    import csdl.jupiter.ReviewTestCase;
007    
008    
009    /**
010     * Provides test cases for the <code>ReviewIssue</code> class.
011     *
012     * @author Takuya Yamashita
013     * @version $Id: TestReviewIssue.java,v 1.6 2005/03/25 18:32:04 takuyay Exp $
014     */
015    public class TestReviewIssue extends ReviewTestCase {
016      /** The <code>ReviewIssue</code> instance to test. */
017      private ReviewIssue codeReview;
018      /**
019       * The <code>ReviewIssue</code> clone instance to test. This instance is made with the
020       * same parameter values as the <code>codeReview</code> reference.
021       */
022      private ReviewIssue codeReviewClone;
023      /**
024       * The <code>ReviewIssue</code> imitation instance to test. This instance is the almost same
025       * as the <code>codeReview</code> reference, but a parameter value is slightly different.
026       */
027      private ReviewIssue codeReviewImitation;
028    
029      /**
030       * Sets up the <code>ReviewIssue</code> instance.
031       *
032       * @throws Exception thrown if problems occur.
033       */
034      public void setUp() throws Exception {
035        super.setUp();
036        this.codeReview = new ReviewIssue(new Date(), new Date(), "reviewer", "assigned to",
037            "doc/review/foo/bar/Baz.java", "10", praise, trivial, "This is summary.",
038            "This is nice description.!? The second sentence continues.",
039            "This is annotation for review.", "This is revision comment.", validFixlater, unresolved,
040            null);
041        this.codeReviewClone = new ReviewIssue(new Date(), new Date(), "reviewer", "assigned to",
042            "doc/review/foo/bar/Baz.java", "10", praise, trivial, "This is summary.",
043            "This is nice description.!? The second sentence continues.",
044            "This is annotation for review.", "This is revision comment.", validFixlater, unresolved,
045            null);
046        
047        this.codeReviewImitation = new ReviewIssue(new Date(), new Date(), "reviewer", "assigned to",
048            "doc/review/foo/bar/Baz.java", "10", praise, trivial, "This is summary.",
049            "This is slightly different description.!? The second sentence continues.",
050            "This is annotation for review.", "This is revision comment.", validFixlater, unresolved,
051            null);
052      }
053      
054      /**
055       * Tests the <code>equals(Object)</code> method.
056       */
057      public void testEquals() {
058        assertEquals("Testing equals(): codeReview and codeReviewClone", true,
059            this.codeReview.equals(this.codeReview));
060        assertEquals("Testing equals(): codeReview and codeReviewClone", true,
061                     this.codeReview.equals(this.codeReviewClone));
062        assertEquals("Testing equals(): codeReview and codeReviewClone", false,
063            this.codeReview.equals(this.codeReviewImitation));
064      }
065      
066      /**
067       * Tests the <code>setCodeReview(ReviewIssue)</code> method.
068       * @throws ReviewException if problems occur.
069       */
070      public void testSetCodeReview() throws ReviewException {
071        this.codeReviewImitation.setReviewIssue(codeReview);
072        assertEquals("Testing equals(): codeReview and codeReviewClone", true,
073            this.codeReview.equals(this.codeReviewImitation));
074      }
075    }
