001    package csdl.jupiter.model.reviewissue;
002    
003    import java.util.Collections;
004    import java.util.Comparator;
005    import java.util.Iterator;
006    import java.util.LinkedList;
007    import java.util.List;
008    import java.util.Map;
009    import java.util.TreeMap;
010    
011    import org.eclipse.core.runtime.ListenerList;
012    import org.eclipse.jface.viewers.IStructuredContentProvider;
013    import org.eclipse.jface.viewers.Viewer;
014    
015    import csdl.jupiter.event.IReviewIssueModelListener;
016    import csdl.jupiter.event.ReviewIssueModelEvent;
017    import csdl.jupiter.ui.marker.ReviewMarker;
018    import csdl.jupiter.util.JupiterLogger;
019    
020    /**
021     * Provides a model for the Code ReviewIssue. Every time code review structure is changed such as
022     * addition, modification, deletion, this notifier notifies the registered ICodeReivew Listener's
023     * implementing class.
024     *
025     * @author Takuya Yamashita
026     * @version $Id: ReviewIssueModel.java,v 1.16 2005/03/25 18:32:05 takuyay Exp $
027     */
028    public class ReviewIssueModel implements IStructuredContentProvider {
029      /** Jupiter logger */
030      private JupiterLogger log = JupiterLogger.getLogger();
031    
032      /** The listener container to contains IReviewIssueModelListener listeners. */
033      private ListenerList listenerList = new ListenerList();
034      /** The map contains the code review id and it mapped <code>ReviewIssue</code> instance. */
035      private Map reviewIssueMap = new TreeMap();
036      /**
037       * The list container to contain the ReviewIssue id string.
038       * This is used for the order of the list.
039       */
040      private List reviewIssueOrdinalList = new LinkedList();
041      /** The comparator to be used for current sort. */
042      private Comparator comparator;
043      /** The code review instance to be notified to the listeners. */
044      private ReviewIssue notifyTargetReviewIssue;
045      /** The map of the String target file - String target file. */
046      private Map targetFileMap = new TreeMap();
047      
048      private boolean isReversed = false;
049    
050      /**
051       * Adds the IReviewIssueModelListener's implementing class to be notified when This model is
052       * changed.
053       *
054       * @param listener Description of the Parameter
055       */
056      public void addListener(IReviewIssueModelListener listener) {
057        listenerList.add(listener);
058      }
059    
060      /**
061       * Remove the IReviewIssueModelListener's implementing class from this listener list.
062       *
063       * @param listener Description of the Parameter
064       */
065      public void removeListener(IReviewIssueModelListener listener) {
066        listenerList.remove(listener);
067      }
068    
069      /**
070       * Adds the new CodeReivew instance to this model. If addition was succeed, then The event are
071       * notified to all the listeners.
072       *
073       * @param reviewIssue The ReviewIssue instance to be added.
074       *
075       * @return <code>true</code> if it's successfully added. <code>false</code> if the element
076       *         already exists.
077       */
078      public boolean add(ReviewIssue reviewIssue) {
079        if ((reviewIssue != null) && !this.contains(reviewIssue.getIssueId())) {
080          this.reviewIssueMap.put(reviewIssue.getIssueId(), reviewIssue);
081          ((LinkedList) this.reviewIssueOrdinalList).addFirst(reviewIssue);
082          ReviewMarker.addMarker(reviewIssue);
083          String targetFile = reviewIssue.getTargetFile();
084          this.targetFileMap.put(targetFile, targetFile);
085          this.notifyTargetReviewIssue = reviewIssue;
086          return true;
087        }
088        else {
089          this.notifyTargetReviewIssue = reviewIssue;
090          return false;
091        }
092      }
093      
094      /**
095       * Adds the list of the new code reviews to this model. No review is added if the
096       * <code>List</code> is null.
097       * @param reviewIssueList the list of the new code reviews.
098       */
099      public void addAll(List reviewIssueList) {
100        if (reviewIssueList != null) {
101          for (Iterator i = reviewIssueList.iterator(); i.hasNext();) {
102            this.add((ReviewIssue) i.next());
103          }
104        }
105      }
106    
107      /**
108       * Check if the ReviewIssue instance contains from the issue ID.
109       *
110       * @param issueId The issue Id of a <code>ReviewIssue</code> instance to be checked.
111       *
112       * @return <code>true</code> if it contains. <code>false</code> otherwise.
113       */
114      public boolean contains(String issueId) {
115        return this.reviewIssueMap.containsKey(issueId);
116      }
117    
118      /**
119       * Gets the <code>ReviewIssue</code> instance contained in this model. Clients can use this
120       * to modify the content of the code review instance. Return null if the instance does not
121       * exists in this model. Note that even though the content of the passing instance
122       * is the same as the that of returning instance, the address of instance might be different.
123       *
124       * @param issueId the issue Id of the <code>ReviewIssue</code> instance to be retrieved.
125       * @return the <code>ReviewIssue</code> instance contained in this model.
126       */
127      public ReviewIssue get(String issueId) {
128        ReviewIssue reviewIssue = (ReviewIssue) this.reviewIssueMap.get(issueId);
129        if (reviewIssue != null) {
130          this.notifyTargetReviewIssue = reviewIssue;
131          return this.notifyTargetReviewIssue;
132        }
133        else {
134          return null;
135        }
136      }
137      
138      /**
139       * Gets the array of the String target file.
140       * @return the array of the String target file.
141       */
142      public String[] getTargetFileArray() {
143        return (String[]) this.targetFileMap.values().toArray(new String[] {});
144      }
145    
146      /**
147       * Provides the size of this model.
148       *
149       * @return The size of this model.
150       */
151      public int size() {
152        return this.reviewIssueMap.size();
153      }
154    
155      /**
156       * Removes the ReviewIssue instance form this model.If deletion was succeed, then The event are
157       * notified to all the listeners.
158       *
159       * @param reviewIssue The ReviewIssue id key to be removed.
160       *
161       * @return <code>true</code> if it's successfully removed.
162       */
163      public boolean remove(ReviewIssue reviewIssue) {
164        if (reviewIssue != null) {
165          this.reviewIssueMap.remove(reviewIssue.getIssueId());
166          ReviewMarker.removeMarker(reviewIssue);
167          this.reviewIssueOrdinalList.remove(reviewIssue);
168          this.notifyTargetReviewIssue = reviewIssue;
169          return true;
170        }
171        else {
172          this.notifyTargetReviewIssue = reviewIssue;
173          return false;
174        }
175      }
176    
177      /**
178       * Clears the model contents.
179       */
180      public void clear() {
181        this.reviewIssueMap.clear();
182        this.reviewIssueOrdinalList.clear();
183        this.notifyTargetReviewIssue = null;
184        this.targetFileMap.clear();
185        ReviewMarker.clearMarkersInReviewId();
186      }
187    
188      /**
189       * Iterates the ReviewIssue instance.
190       *
191       * @return The Iterator instance.
192       */
193      public Iterator iterator() {
194        return reviewIssueMap.values().iterator();
195      }
196    
197      /**
198       * Notifies the listeners who implement the <code>IReviewIssueModelListener</code>. This
199       * method can be invoked anytime, but it is designed for the function to let listeners to know
200       * the model is update so that it is ideal after <code>add(ReviewIssue)</code> or
201       * <code>remove(ReviewIssue)</code> method invocation.
202       * 
203       * @param type the event type of the <code>ReviewIssueModelEvent</code>
204       * when notifying to listeners.
205       */
206      public void notifyListeners(int type) {
207        this.notifyListeners(new ReviewIssueModelEvent(this.notifyTargetReviewIssue, type));
208      }
209    
210      /**
211       * Notifies the ReviewIssue
212       *
213       * @param event the <code>CodeReviewModelEvent</code> instance.
214       */
215      private void notifyListeners(ReviewIssueModelEvent event) {
216        Object[] listeners = listenerList.getListeners();
217        for (int i = 0; i < listeners.length; i++) {
218          ((IReviewIssueModelListener) listeners[i]).reviewIssueModelChanged(event);
219        }
220      }
221    
222      /**
223       * Gets the elements of this model. Each element type is ReviewIssue instance.
224       *
225       * @see org.eclipse.jface.viewers.IStructuredContentProvider#getElements(java.lang.Object)
226       */
227      public Object[] getElements(Object inputElement) {
228        return this.reviewIssueOrdinalList.toArray();
229      }
230    
231      /**
232       * Sorts this model by the order defined by Comparator implementing class. Reverse the model if
233       * you have passed the same comparator instance as the previous comparator instance. Sorts by
234       * natural order if <code>comparator</code> parameter is <code>null</code>.
235       *
236       * @param comparator the comparator to determine how this model is sorted.
237       */
238      public void sortBy(Comparator comparator) {
239        if (this.comparator == comparator) {
240          Collections.reverse(reviewIssueOrdinalList);
241          isReversed = !isReversed;
242        }
243        else {
244          Collections.sort(reviewIssueOrdinalList, comparator);
245          isReversed = false;
246          this.comparator = comparator;
247        }
248      }
249      
250      /**
251       * Sorts this model by the previous comparator.
252       */
253      public void sortByPreviousComparator() {
254        if (comparator != null) {
255          Collections.sort(reviewIssueOrdinalList, comparator);
256          if (isReversed) {
257            Collections.reverse(reviewIssueOrdinalList);
258          }
259        }
260      }
261      
262      /**
263       * Clears the comparator used in this model.
264       */
265      public void clearComparator() {
266        this.comparator = null;
267      }
268    
269      /**
270       * <b>Description copied from interface:</b>{@link org.eclipse.jface.viewers.IContentProvider}
271       * Disposes of this content provider. This is called by the viewer when it is disposed.
272       *
273       * @see org.eclipse.jface.viewers.IContentProvider#dispose()
274       */
275      public void dispose() {
276        // nothing so far.
277      }

279      /**
280       * <b>Description copied from interface:</b>{@link org.eclipse.jface.viewers.IContentProvider}
281       * Notifies this content provider that the given viewer's input has been switched to a different
282       * element.  A typical use for this method is registering the content provider as a listener to
283       * changes on the new input (using model-specific means), and deregistering the viewer from the
284       * old input. In response to these change notifications, the content provider should update the
285       * viewer (see the add, remove, update and refresh methods on the viewers).  The viewer should
286       * not updated during this call, as it might be in the process of being disposed.
287       *
288       * @see org.eclipse.jface.viewers.IContentProvider#inputChanged(org.eclipse.jface.viewers.Viewer,
289       *      java.lang.Object, java.lang.Object)
290       */
291      public void inputChanged(Viewer viewer, Object oldInput, Object newInput) {
292        // nothing so far
293        log.debug("inputChanage was called.");
294      }
