001    package csdl.jupiter.model.reviewissue;
002    
003    import csdl.jupiter.ReviewTestCase;
004    import csdl.jupiter.util.ResourceBundleKey;
005    
006    /**
007     * Test for the Type class.
008     *
009     * @author Takuya Yamashita
010     * @version $Id: TestType.java,v 1.2 2005/03/25 18:32:03 takuyay Exp $
011     */
012    public class TestType extends ReviewTestCase {
013      /**
014       * Tests getKey() method. The return value should be equal to each string below.
015       */
016      public void testGetkey() {
017        String key = ResourceBundleKey.ITEM_KEY_TYPE_DEFECT;
018        assertEquals("Testing the key of DEFECT", key, defect.getKey());
019        key = ResourceBundleKey.ITEM_KEY_TYPE_QUESTION;
020        assertEquals("Testing the key of QUESTION", key, question.getKey());
021        key = ResourceBundleKey.ITEM_KEY_TYPE_EXTERNAL_ISSUE;
022        assertEquals("Testing the key of EXTERNAL_ISSUE", key, externalIssue.getKey());
023        key = ResourceBundleKey.ITEM_KEY_TYPE_PRAISE;
024        assertEquals("Testing the key of PRAISE", key, praise.getKey());
025      }
026    
027      /**
028       * Tests compareTo method. Since the order of dispositions is defect &lt; externalIssue &lt;
029       * question &lt; praise, <code>defect.compareTo(externalIssue))</code> should return
030       * <code>-1</code>, <code>externalIssue.compareTo(question))</code> should return
031       * <code>-1</code>, <code>question.compareTo(praise))</code> should return
032       * <code>-1</code>.
033       */
034      public void testCompareTo() {
035        assertEquals("Testing compareTo with defect and externalIssue", -1,
036          defect.compareTo(externalIssue));
037        assertEquals("Testing compareTo with externalIssue and question", -1,
038            externalIssue.compareTo(question));
039        assertEquals("Testing compareTo with question and praise", -1,
040            question.compareTo(praise));
041      }
042    }
