001    package csdl.jupiter.model.reviewissue;
002    
003    import csdl.jupiter.ReviewTestCase;
004    import csdl.jupiter.util.ResourceBundleKey;
005    
006    /**
007     * Tests for Resolution class.
008     *
009     * @author Takuya Yamashita
010     * @version $Id: TestResolution.java,v 1.2 2004/08/23 06:40:47 takuyay Exp $
011     */
012    public class TestResolution extends ReviewTestCase {
013      /**
014       * Test getKey() method. The return value should be equal to each string below.
015       */
016      public void testGetKey() {
017        String key = ResourceBundleKey.ITEM_KEY_UNSET;
018        assertEquals("Testing key of UNSET", key, unset.getKey());
019        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_NEEDSFIXING;
020        assertEquals("Testing key of VALID_NEEDS_FIXING", key, validNeedsfixing.getKey());
021        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_FIXLATER;
022        assertEquals("Testing key of VALID_FIX_LATER", key, validFixlater.getKey());
023        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_DUPLICATE;
024        assertEquals("Testing key of VALID_DUPLICATE", key, validDuplicate.getKey());
025        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_VALID_WONTFIX;
026        assertEquals("Testing key of VALID_WONT_FIX", key, validWontfix.getKey());
027        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_INVALID_WONTFIX;
028        assertEquals("Testing key of InVALID_WONTFIX", key, invalidWontfix.getKey());
029        key = ResourceBundleKey.ITEM_KEY_RESOLUTION_UNSURE_VALIDITY;
030        assertEquals("Testing key of UNSURE_VALIDITY", key, unsureValidity.getKey());
031      }
032    
033      /**
034       * Since the order of dispositions is UNSET &lt; VALID_NEEDS_FIXING &lt; VALID_WONT_FIX &lt;
035       * VALID_DUPLICATE &lt; VALID_FIX_LATER,
036       * <code>Resolution.VALID_NEEDS_FIXING.compareTo(Resolution.VALID_WONT_FIX))</code> should
037       * return <code>-1</code>, <code>Resolution.DEFERcompareTo(Resolution.VALID_DUPLICATE))</code>
038       * should return <code>-1</code>,
039       * <code>Resolution.VALID_DUPLICATE.compareTo(Resolution.VALID_FIX_LATER))</code> should return
040       * <code>-1</code>.
041       */
042      public void testCompareTo() {
043        assertEquals("Testing compareTo with UNSET and VALID_NEEDS_FIXING", -1,
044          unset.compareTo(validNeedsfixing));
045        assertEquals("Testing compareTo with VALID_NEEDS_FIXING and VALID_FIX_LATER", -1,
046          validNeedsfixing.compareTo(validFixlater));
047        assertEquals("Testing compareTo with VALID_FIX_LATER and VALID_DUPLICATE", -1, 
048            validFixlater.compareTo(validDuplicate));
049        assertEquals("Testing compareTo with VALID_DUPLICATE and VALID_WONTFIX", -1,
050          validDuplicate.compareTo(validWontfix));
051        assertEquals("Testing compareTo with VALID_WONTFIX and INVALID_WONTFIX", -1,
052            validWontfix.compareTo(invalidWontfix));
053        assertEquals("Testing compareTo with INVALID_WONTFIX and UNSURE_VALIDITY", -1,
054            invalidWontfix.compareTo(unsureValidity));
055      }
056    }
