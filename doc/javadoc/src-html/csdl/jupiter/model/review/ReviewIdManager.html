001    package csdl.jupiter.model.review;
002    
003    
004    /**
005     * Provides the singleton review id manager class.
006     * @author Takuya Yamashita
007     * @version $Id: ReviewIdManager.java,v 1.3 2005/03/28 14:39:31 takuyay Exp $
008     */
009    public class ReviewIdManager implements IReviewModelElementChangeListener {
010      private static ReviewIdManager theInstance = new ReviewIdManager();
011      private ReviewId reviewId;
012      
013      /**
014       * Prohibits clients from instantiating this.
015       */
016      private ReviewIdManager() {
017      }
018      
019      /**
020       * Gets the singleton instance.
021       * @return the singleton instance.
022       */
023      static ReviewIdManager getInstance() {
024        return theInstance;
025      }
026      
027      /**
028       * Sets the ReviewId if the object is the instance of the ReviewId.
029       * @param object The object to be notified.
030       */
031      public void elementChanged(Object object) {
032        if (object instanceof ReviewId) {
033          setReviewId((ReviewId) object);
034        }
035      }
036      
037      /**
038       * Sets the <code>ReviewId</code> instance.
039       * @param reviewId the <code>ReviewId</code> instance.
040       */
041      private void setReviewId(ReviewId reviewId) {
042        this.reviewId = reviewId;
043      }
044      
045      /**
046       * Gets the <code>ReviewId</code> instance. Returns null if there is no <code>ReviewId</code>
047       * instance is set.
048       * @return the <code>ReviewId</code> instance.
049       */
050      public ReviewId getReviewId() {
051        return this.reviewId;
052      }
053    }
