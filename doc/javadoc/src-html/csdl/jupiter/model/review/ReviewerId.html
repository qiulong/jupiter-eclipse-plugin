001    package csdl.jupiter.model.review;
002    
003    import csdl.jupiter.util.ResourceBundleKey;
004    
005    /**
006     * Provides reviewer information.
007     * @author Takuya Yamashita
008     * @version $Id: ReviewerId.java,v 1.3 2004/09/13 12:09:07 takuyay Exp $
009     */
010    public class ReviewerId {
011      /** The reviewer automatic key to use dynamic reviewer allocation. */
012      public static final String AUTOMATIC_KEY = ResourceBundleKey.ITEM_KEY_REVIEWER_AUTOMATIC;
013      /** The reviewer id. */
014      private String reviewerId;
015      /** The reviewer name. */
016      private String reviewerName;
017      
018      /**
019       * Instantiates reviewer instance.
020       * @param reviewerId the review id.
021       * @param reviewerName the review name.
022       */
023      public ReviewerId(String reviewerId, String reviewerName) {
024        this.reviewerId = reviewerId;
025        this.reviewerName = reviewerName;
026      }
027      
028      /**
029       * Gets the review id.
030       * @return Returns the reviewerId.
031       */
032      public String getReviewerId() {
033        return reviewerId;
034      }
035      
036      /**
037       * Gets the reviewer name.
038       * @return Returns the reviewerName.
039       */
040      public String getReviewerName() {
041        return reviewerName;
042      }
043    }
