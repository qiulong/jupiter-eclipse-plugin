001    package csdl.jupiter;
002    
003    /**
004     * Thrown when exceptions occur during code review processing.
005     *
006     * @author Takuya Yamashita
007     * @version $Id: ReviewException.java,v 1.2 2005/05/06 07:41:09 takuyay Exp $
008     */
009    public class ReviewException extends Exception {
010      /** The serial version UID. */
011      private static final long serialVersionUID = 3256728389887931954L;
012    
013      /**
014       * Thrown when exceptions occur during code review processing.
015       *
016       * @param detailMessage A detailed string describing the error.
017       */
018      public ReviewException(String detailMessage) {
019        super(detailMessage);
020      }
021    
022      /**
023       * Thrown when exceptions occur during code review processing.
024       *
025       * @param detailMessage A detailed string describing the error.
026       * @param cause the cause (which is saved for later retrieval by the {@link #getCause()} method).
027       *        (A <tt>null</tt> value is permitted, and indicates that the cause is nonexistent or
028       *        unknown.)
029       */
030      public ReviewException(String detailMessage, Throwable cause) {
031        super(detailMessage, cause);
032      }
033    }
