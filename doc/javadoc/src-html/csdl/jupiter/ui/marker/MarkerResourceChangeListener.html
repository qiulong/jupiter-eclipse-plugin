001    package csdl.jupiter.ui.marker;
002    
003    import org.eclipse.core.resources.IFile;
004    import org.eclipse.core.resources.IMarker;
005    import org.eclipse.core.resources.IMarkerDelta;
006    import org.eclipse.core.resources.IResource;
007    import org.eclipse.core.resources.IResourceChangeEvent;
008    import org.eclipse.core.resources.IResourceChangeListener;
009    import org.eclipse.core.resources.IResourceDelta;
010    import org.eclipse.core.resources.IResourceDeltaVisitor;
011    import org.eclipse.core.runtime.CoreException;
012    
013    import csdl.jupiter.event.ReviewIssueModelEvent;
014    import csdl.jupiter.file.FileResource;
015    import csdl.jupiter.model.reviewissue.ReviewIssue;
016    import csdl.jupiter.model.reviewissue.ReviewIssueModel;
017    import csdl.jupiter.model.reviewissue.ReviewIssueModelManager;
018    import csdl.jupiter.util.JupiterLogger;
019    
020    /**
021     * Provides the resource change listener to check whether markers are moved along with the
022     * text document modification.
023     * @author Takuya Yamashita
024     * @version $Id: MarkerResourceChangeListener.java,v 1.1 2005/03/26 07:13:36 takuyay Exp $
025     */
026    public class MarkerResourceChangeListener 
027                 implements IResourceChangeListener, IResourceDeltaVisitor {
028      /** Jupiter logger */
029      private JupiterLogger log = JupiterLogger.getLogger();
030    
031        /**
032         * Implements resource delta for the active file and accepts the implementing
033         * resource data visitor to check the marker change associating with the resource change.
034         * This method is called by platform when resources are changed.
035         * 
036         * @param event A resource change event to describe changes to resources.
037         */
038        public void resourceChanged(IResourceChangeEvent event) {
039          IResource resource = event.getResource();
040          if (((event.getType() & IResourceChangeEvent.POST_CHANGE) != 0)) {
041            try {
042              IFile activeFile = FileResource.getActiveFile();
043              if (activeFile == null) {
044                return;
045              }
046              IResourceDelta rootDelta = event.getDelta();
047              // only interested in the active file.
048              IResourceDelta resourceDelta = rootDelta.findMember(activeFile.getFullPath());
049              if (resourceDelta == null) {
050                return;
051              }
052              resourceDelta.accept(this);
053            }
054            catch (CoreException e) {
055              e.printStackTrace();
056            }
057          }
058        }
059    
060        /**
061         * Implements marker delta to check which markers are changed and retrieve the new line number
062         * and set it to the <code>ReviewIssue</code> instance in the <code>ReviewIssueModel</code>. 
063         * This method is called by EclipseSensorPlugin instance. Note that
064         * <code>true</code> is returned if the parameter of IResourceDelta instance has children.
065         * 
066         * @param delta IResourceDelta instance to contains delta resource.
067         * @return true if the resource delta's children should be visited; false if they should be
068         *         skipped.
069         * @throws CoreException if this method fails. Reasons include:
070         * <ul>
071         * <li> This marker does not exist.</li>
072         * </ul>
073         */
074        public boolean visit(IResourceDelta delta) throws CoreException {
075         // only interested in changed resources (not added or removed)
076          if ((delta.getKind() & IResourceDelta.CHANGED) == 0) {
077    //         log.debug("added or removed.");
078             return true;
079          }
080          //only interested in content changes
081          if ((delta.getFlags() & IResourceDelta.CONTENT) == 0) {
082    //         log.debug("not content change.");
083             return true;
084          }
085          IMarkerDelta[] markerdeltas = delta.getMarkerDeltas();
086          boolean isReviewMarker = false;
087          for (int i = 0; i < markerdeltas.length; i++) {
088            IMarkerDelta markerDelta = markerdeltas[i];
089            String markerType = markerDelta.getType();
090            if (!markerDelta.getType().equals(ReviewMarker.REVIEW_MARKER)) {
091              continue;
092            }
093            IMarker marker = markerDelta.getMarker();
094            Integer oldLineNumber = (Integer) markerDelta.getAttribute(IMarker.LINE_NUMBER);
095            Integer newLineNumber = (Integer) marker.getAttribute(IMarker.LINE_NUMBER);
096            if (newLineNumber == null || oldLineNumber == null 
097                || newLineNumber.equals(oldLineNumber)) {
098              continue;
099            }
100            log.debug("old marker line number: " + oldLineNumber.intValue());
101            log.debug("new marker line number: " + newLineNumber.intValue());
102            isReviewMarker = true;
103            String reviewIssueKey = ReviewMarker.ATTRIBUTE_REVIEW_ISSUE;
104            ReviewIssue savingReviewIssue = (ReviewIssue) marker.getAttribute(reviewIssueKey);
105            savingReviewIssue.setLine(newLineNumber.intValue() + "");
106          }
107          if (isReviewMarker) {
108            log.debug("review marker was moved along with resource change.");
109            ReviewIssueModelManager reviewIssueModelManager = ReviewIssueModelManager.getInstance();
110            ReviewIssueModel reviewIssueModel = reviewIssueModelManager.getCurrentModel();
111            reviewIssueModel.notifyListeners(ReviewIssueModelEvent.EDIT);
112          }
113          return true; // visit the children
114        }
115    
116    }
