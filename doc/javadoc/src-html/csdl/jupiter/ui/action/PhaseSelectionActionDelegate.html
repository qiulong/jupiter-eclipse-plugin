001    package csdl.jupiter.ui.action;
002    
003    import org.eclipse.jface.action.IAction;
004    import org.eclipse.swt.widgets.Control;
005    import org.eclipse.swt.widgets.Menu;
006    import org.eclipse.ui.IWorkbenchWindow;
007    import org.eclipse.ui.IWorkbenchWindowPulldownDelegate;
008    import org.eclipse.ui.actions.ActionDelegate;
009    
010    import csdl.jupiter.ReviewPlugin;
011    import csdl.jupiter.event.ReviewEvent;
012    import csdl.jupiter.model.review.ReviewModel;
013    import csdl.jupiter.ui.menu.PhaseSelectionMenu;
014    import csdl.jupiter.ui.perspective.ReviewPerspectiveFactory;
015    
016    /**
017     * Provides the action to correct all review files in the directory which has been specified in the
018     * preference page.
019     *
020     * @author Takuya Yamashita
021     * @version $Id: PhaseSelectionActionDelegate.java,v 1.9 2005/03/28 14:39:31 takuyay Exp $
022     */
023    public class PhaseSelectionActionDelegate extends ActionDelegate
024      implements IWorkbenchWindowPulldownDelegate {
025      
026      /**
027       * Initializes this action delegate with the workbench window it will work in.
028       *
029       * @param window the window that provides the context for this delegate
030       */
031      public void init(IWorkbenchWindow window) {
032      }
033    
034      /**
035       * Collects all review files in the directory which has been specified in the preference page.
036       *
037       * @param action the action proxy that handles the presentation portion of the action
038       */
039      public void run(IAction action) {
040        ReviewPerspectiveFactory.showPerspective();
041        String reviewPhaseNameKey = ReviewModel.getInstance().getPhaseManager().getPhaseNameKey();
042        boolean isSuccess = PhaseSelectionMenu.doUpdateMenuCommand(reviewPhaseNameKey);
043        if (isSuccess) {
044          int type = ReviewEvent.TYPE_COMMAND;
045          int kind = ReviewEvent.KIND_PHASE_SELECTION;
046          ReviewPlugin.getInstance().notifyListeners(type, kind);
047        }
048      }
049    
050      /**
051       * Returns the menu for this pull down action.
052       *
053       * @param parent the parent of <code>Control</code> instance.
054       *
055       * @return the <code>Menu</code> instance to be filled with the sets of the code review mode.
056       */
057      public Menu getMenu(Control parent) {
058        return PhaseSelectionMenu.createPhaseSelectionPullDownMenu(new Menu(parent));
059      }
060    }
