001    package csdl.jupiter.ui.preference;
002    
003    import java.util.ArrayList;
004    import java.util.Arrays;
005    import java.util.Iterator;
006    import java.util.List;
007    
008    import org.eclipse.jface.preference.BooleanFieldEditor;
009    import org.eclipse.jface.preference.FieldEditor;
010    import org.eclipse.jface.preference.IPreferenceStore;
011    import org.eclipse.jface.preference.PreferencePage;
012    import org.eclipse.jface.preference.StringFieldEditor;
013    import org.eclipse.jface.util.IPropertyChangeListener;
014    import org.eclipse.jface.util.PropertyChangeEvent;
015    import org.eclipse.swt.SWT;
016    import org.eclipse.swt.layout.GridData;
017    import org.eclipse.swt.layout.GridLayout;
018    import org.eclipse.swt.widgets.Composite;
019    import org.eclipse.swt.widgets.Control;
020    import org.eclipse.swt.widgets.Group;
021    import org.eclipse.ui.IWorkbench;
022    import org.eclipse.ui.IWorkbenchPreferencePage;
023    
024    import csdl.jupiter.ReviewI18n;
025    import csdl.jupiter.ReviewPlugin;
026    
027    /**
028     * Provides a root preference page for the Jupiter review system. Enable user to configure some
029     * features such as default reviewer name, update manager URL, etc.
030     *
031     * @author Takuya Yamashita
032     * @version $Id: GeneralPreferencePage.java,v 1.12 2005/03/25 18:32:10 takuyay Exp $
033     */
034    public class GeneralPreferencePage extends PreferencePage implements IWorkbenchPreferencePage {
035      /** The preference store to hold the existing preference values. */
036      private IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
037      /** The field editor list to hold the field editor. */
038      private List fieldEditors = new ArrayList();
039      /** The update URL store key. */
040      public static final String UPDATE_URL_KEY = "UpdateUrl";
041      /** The enable update store key. */
042      public static final String ENABLE_UPDATE_KEY = "EnableUpdate";
043    
044      /**
045       * Initializes this preference page for the given workbench. Loads the properties file setting to
046       * the preference store if the file exists.
047       * 
048       * <p>
049       * This method is called automatically as the preference page is being created and initialized.
050       * Clients must not call this method.
051       * </p>
052       *
053       * @param workbench the workbench
054       */
055      public void init(IWorkbench workbench) {
056      }
057    
058      /**
059       * Creates the root preference page of the jupiter configuration.
060       *
061       * @param parent the parent composite
062       *
063       * @return the new control
064       */
065      protected Control createContents(Composite parent) {
066        Composite top = new Composite(parent, SWT.LEFT);
067        top.setLayoutData(new GridData(GridData.FILL_HORIZONTAL));
068        top.setLayout(new GridLayout());
069        // Creates review preference group.
070        createGeneralPreferenceGroup(top);
071        // Loads the preference store.
072        loadFieldEditors(false);
073        return top;
074      }
075    
076      /**
077       * Creates preview preference group, which contains default reviewer's name.
078       * generalPreferenceGroup.setLayoutData(new GridData(GridData.FILL_HORIZONTAL));
079       *
080       * @param parent the parent composite.
081       */
082      private void createGeneralPreferenceGroup(Composite parent) {
083        final Group generalPreferenceGroup = new Group(parent, SWT.NONE);
084        String generalLabelPreferenceKey = "GeneralPreferencePage.general.label.preference";
085        generalPreferenceGroup.setText(ReviewI18n.getString(generalLabelPreferenceKey));
086        generalPreferenceGroup.setLayout(new GridLayout());
087        generalPreferenceGroup.setLayoutData(new GridData(GridData.FILL_HORIZONTAL));
088        // Creates text field sub group.
089        final Composite textFieldSubGroup = new Composite(generalPreferenceGroup, SWT.NONE);
090        textFieldSubGroup.setLayout(new GridLayout());
091        GridData fieldData = new GridData(GridData.FILL_HORIZONTAL);
092        fieldData.horizontalIndent = 2;
093        textFieldSubGroup.setLayoutData(fieldData);
094        String updateUrlKey = GeneralPreferencePage.UPDATE_URL_KEY;
095        String generalLabelUpdateUrlKey = "GeneralPreferencePage.general.label.updateUrl";
096        final StringFieldEditor updateUrlField = new StringFieldEditor(updateUrlKey,
097            ReviewI18n.getString(generalLabelUpdateUrlKey), textFieldSubGroup);
098    
099        // Creates sub enable update group.
100        Composite enableUpdateGroup = new Composite(generalPreferenceGroup, SWT.NONE);
101        enableUpdateGroup.setLayout(new GridLayout());
102        GridData enableData = new GridData(GridData.FILL_HORIZONTAL);
103        enableData.horizontalIndent = 2;
104        enableUpdateGroup.setLayoutData(enableData);
105        String enableUpdateKey = GeneralPreferencePage.ENABLE_UPDATE_KEY;
106        String generalLabelEnableAutoUpdateKey = "GeneralPreferencePage.general.label.enableAutoUpdate";
107        final BooleanFieldEditor enableUpdateField = new BooleanFieldEditor(enableUpdateKey,
108            ReviewI18n.getString(generalLabelEnableAutoUpdateKey), enableUpdateGroup);
109        enableUpdateField.setPropertyChangeListener(new IPropertyChangeListener() {
110            public void propertyChange(PropertyChangeEvent event) {
111              updateUrlField.setEnabled(enableUpdateField.getBooleanValue(), textFieldSubGroup);
112            }
113          });
114    
115        // Initializes the field status.
116        String enableUpdateStoreKey = GeneralPreferencePage.ENABLE_UPDATE_KEY;
117        updateUrlField.setEnabled(this.store.getBoolean(enableUpdateStoreKey), textFieldSubGroup);
118        // Collects all the FieldEditor instance to the List in order to load the setting in the
119        // preference store.
120        this.fieldEditors.addAll(Arrays.asList(
121            new FieldEditor[] {updateUrlField, enableUpdateField}));
122      }
123    
124      /**
125       * Stores all <code>FiledEditor</code> instances.
126       */
127      private void storeFieldEditors() {
128        for (Iterator i = this.fieldEditors.iterator(); i.hasNext();) {
129          FieldEditor fieldEditor = (FieldEditor) i.next();
130          if (fieldEditor != null) {
131            fieldEditor.setPreferenceStore(this.store);
132            fieldEditor.store();
133          }
134        }
135      }
136    
137      /**
138       * Restores all field editor's values.
139       *
140       * @param isDefautLoaded <code>true</code> if default field editors are loaded.
141       *        <code>false</code> otherwise.
142       */
143      private void loadFieldEditors(boolean isDefautLoaded) {
144        for (Iterator i = this.fieldEditors.iterator(); i.hasNext();) {
145          FieldEditor fieldEditor = (FieldEditor) i.next();
146          if (fieldEditor != null) {
147            fieldEditor.setPreferenceStore(this.store);
148            if (isDefautLoaded) {
149              fieldEditor.loadDefault();
150            }
151            else {
152              fieldEditor.load();
153            }
154          }
155        }
156      }
157    
158      /**
159       * Performs to save the current data in the editors.
160       *
161       * @return <code>false</code> to abort the container's OK processing and <code>true</code> to
162       *         allow the OK to happen.
163       */
164      public boolean performOk() {
165        storeFieldEditors();
166        return super.performOk();
167      }
168    
169      /**
170       * Performs special processing when this page's Apply button has been pressed. The default
171       * implementation of this framework method simply calls performOk to simulate the pressing of
172       * the page's OK button.
173       */
174      public void performApply() {
175        performOk();
176      }
177    
178      /**
179       * Performs reset of all field values to the default specified on the
180       * <code>csdl.jupiter.ReviewPlugin.initializeDefaultPreferences(IPreferenceStore)</code>
181       * method.
182       */
183      public void performDefaults() {
184        loadFieldEditors(true);
185      }
186    }
