064      /**
065       * Initializes this preference page for the given workbench.
066       *
067       * @param workbench the workbench.
068       */
069      public void init(IWorkbench workbench) {
070        ReviewModel reviewModel = ReviewModel.getInstance();
071        String reviewPhaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
072        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
073        ColumnDataModel columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKey);
074        PrefResource configManager = PrefResource.getInstance();
075        // Sets the initial phase to display in the view preference page.
076        if (columnDataModel.allSize() <= 0) {
077          this.reviewPhaseNameKey = configManager.getDefaultPhaseNameKey();
078        }
079        else {
080          this.reviewPhaseNameKey = reviewPhaseNameKey;
081        }
082        this.tempReviewPhaseNameKey = this.reviewPhaseNameKey;
083        String[] reviewPhaseNameKeys = configManager.getPhaseArray(true);
084        this.reviewPhaseMap = new HashMap();
085        for (int i = 0; i < reviewPhaseNameKeys.length; i++) {
086          columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKeys[i]);
087          ColumnData[] columnDataArray = columnDataModel.getAllColumnDataArray();
088          ArrayList columnDataList = new ArrayList(Arrays.asList(columnDataArray));
089          this.reviewPhaseMap.put(reviewPhaseNameKeys[i], columnDataList);
090        }
091      }

093      /**
094       * Creates preference page controls on demand.
095       *
096       * @param ancestor the parent for the preference page
097       *
098       * @return the <code>Control</code> instance.
099       */
100      protected Control createContents(Composite ancestor) {
101        Composite parent = createGeneralComposite(ancestor);
102        createLabelContent(parent);
103        createReviewPhaseContent(parent);
104        createViewPreferenceContent(parent);
105        fillTable((List) this.reviewPhaseMap.get(this.reviewPhaseNameKey));
106        Dialog.applyDialogFont(ancestor);
107        return parent;
108      }

110      /**
111       * Creates view preference frame and return the child composite.

113       * @return the child composite.
114       */

116        Composite child = new Composite(parent, SWT.NULL);

120        layout.numColumns = 1;
121        child.setLayout(layout);
122        GridData data = new GridData();
123        data.verticalAlignment = GridData.FILL;
124        data.horizontalAlignment = GridData.FILL;
125        child.setLayoutData(data);
126        return child;
127      }

141      /**
142       * Creates review phase content.
143       * @param parent the parent composite.
144       */
145      private void createReviewPhaseContent(Composite parent) {
146        Composite reviewPhaseSubGroup = new Composite(parent, SWT.NONE);
147        GridLayout reviewPhaseSubGroupLayout = new GridLayout();
148        reviewPhaseSubGroupLayout.marginWidth = 0;
149        reviewPhaseSubGroupLayout.marginHeight = 0;
150        reviewPhaseSubGroupLayout.numColumns = 1;
151        reviewPhaseSubGroup.setLayoutData(new GridData(GridData.FILL_BOTH));
152        reviewPhaseSubGroup.setLayout(reviewPhaseSubGroupLayout);
153        
154        this.reviewPhaseCombo = new Combo(reviewPhaseSubGroup, SWT.READ_ONLY);
155        GridData reviewPhaseData = new GridData(GridData.FILL_HORIZONTAL);
156        reviewPhaseCombo.setLayoutData(reviewPhaseData);
157        reviewPhaseCombo.setItems(PrefResource.getInstance().getPhaseArray(false));
158        ReviewPlugin plugin = ReviewPlugin.getInstance();
159        int index = reviewPhaseCombo.indexOf(ReviewI18n.getString(reviewPhaseNameKey));
160        reviewPhaseCombo.select(index);
161        reviewPhaseCombo.addListener(SWT.Selection, new Listener() {
162          public void handleEvent(Event e) { 
163            String reviewPhaseName = reviewPhaseCombo.getItem(reviewPhaseCombo.getSelectionIndex());
164            reviewPhaseNameKey = ReviewI18n.getKey(reviewPhaseName);
165            ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
166            ColumnDataModel columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKey);
167    //        PrefResource.getInstance().loadColumnDataManager(columnDataManager, reviewPhaseNameKey,
168    //                                                          false);
169            removeAllItems();
170            fillTable((List) reviewPhaseMap.get(reviewPhaseNameKey));
171          }
172        });
173      }

175      /**
176       * Creates view preference content.
177       * @param parent the parent composite.
178       */
179      private void createViewPreferenceContent(Composite parent) {
180        Composite viewPreferenceSubGroup = new Composite(parent, SWT.NONE);
181        GridLayout viewPreferenceSubGroupLayout = new GridLayout();
182        viewPreferenceSubGroupLayout.marginWidth = 0;
183        viewPreferenceSubGroupLayout.marginHeight = 0;
184        viewPreferenceSubGroupLayout.numColumns = 2;
185        viewPreferenceSubGroup.setLayoutData(new GridData(GridData.FILL_BOTH));
186        viewPreferenceSubGroup.setLayout(viewPreferenceSubGroupLayout);
187        
188        viewColumnTable = new Table(viewPreferenceSubGroup, SWT.CHECK | SWT.BORDER);
189        GridData gridData = new GridData(GridData.FILL_BOTH);
190    
191        //gd.widthHint = convertWidthInCharsToPixels(30);
192        gridData.heightHint = 300;
193        viewColumnTable.setLayoutData(gridData);
194        viewColumnTable.addListener(SWT.Selection, new Listener() {
195          public void handleEvent(Event e) {
196            handleSelection((TableItem) e.item);
197          }
198        });
199        
200    
201        Composite buttons = new Composite(viewPreferenceSubGroup, SWT.NULL);
202        buttons.setLayoutData(new GridData(GridData.VERTICAL_ALIGN_BEGINNING));
203        GridLayout layout = new GridLayout();
204        layout.marginHeight = 0;
205        layout.marginWidth = 0;
206        buttons.setLayout(layout);
207        modifyButton = new Button(buttons, SWT.PUSH);
208        modifyButton.setText(ReviewI18n.getString("ViewPreference.button.label.modify"));
209        modifyButton.setEnabled(false);
210        GridData data = new GridData();
211        data.horizontalAlignment = GridData.FILL;
212        // As to Eclipse Help, button layout is determined by layout.
213        //data.heightHint = convertVerticalDLUsToPixels(IDialogConstants.BUTTON_HEIGHT);
214        int widthHint = convertHorizontalDLUsToPixels(IDialogConstants.BUTTON_WIDTH);
215        int x = modifyButton.computeSize(SWT.DEFAULT, SWT.DEFAULT, true).x;
216        data.widthHint = Math.max(widthHint, x);
217        modifyButton.setLayoutData(data);
218        modifyButton.addListener(SWT.Selection, new Listener() {
219          public void handleEvent(Event e) {
220            modifyWidth();
221          }
222        });
223        
224        upButton = new Button(buttons, SWT.PUSH);
225        Image upImage = ReviewPlugin.createImageDescriptor("lib/icons/up.gif").createImage();
226        upButton.setImage(upImage);
227        upButton.setEnabled(false);
228        data = new GridData();
229        data.horizontalAlignment = GridData.FILL;
230        // data.heightHint = convertVerticalDLUsToPixels(IDialogConstants.BUTTON_HEIGHT);
231        widthHint = convertHorizontalDLUsToPixels(IDialogConstants.BUTTON_WIDTH);
232        x = upButton.computeSize(SWT.DEFAULT, SWT.DEFAULT, true).x;
233        data.widthHint = Math.max(widthHint, x);
234        upButton.setLayoutData(data);
235        upButton.addListener(SWT.Selection, new Listener() {
236          public void handleEvent(Event e) {
237            moveUpItem(viewColumnTable.getSelectionIndex());
238          }
239        });
240        
241        downButton = new Button(buttons, SWT.PUSH);
242        Image downImage = ReviewPlugin.createImageDescriptor("lib/icons/down.gif").createImage();
243        downButton.setImage(downImage);
244        downButton.setEnabled(false);
245        data = new GridData();
246        data.horizontalAlignment = GridData.FILL;
247        //data.heightHint = convertVerticalDLUsToPixels(IDialogConstants.BUTTON_HEIGHT);
248        widthHint = convertHorizontalDLUsToPixels(IDialogConstants.BUTTON_WIDTH);
249        x = downButton.computeSize(SWT.DEFAULT, SWT.DEFAULT, true).x;
250        data.widthHint = Math.max(widthHint, x);
251        downButton.setLayoutData(data);
252        downButton.addListener(SWT.Selection, new Listener() {
253          public void handleEvent(Event e) {
254            moveDownItem(viewColumnTable.getSelectionIndex());
255          }
256        });
257      }

259      /**
260       * Moves up the indexed item by one.
261       * @param index the index of the item to move up.
262       *
263       */
264      private void moveUpItem(int index) {
265        if (index > 0) {
266          List columnDataList = (List) this.reviewPhaseMap.get(this.reviewPhaseNameKey);
267          ColumnData columnData = (ColumnData) columnDataList.remove(index);
268          int newIndex = index - 1;
269          columnDataList.add(newIndex, columnData);
270          removeAllItems();
271          fillTable(columnDataList);
272          viewColumnTable.select(newIndex);
273        }
274      }
275      
276      /**
277       * Moves down the indexed item by one.
278       * @param index the index of the item to move down.
279       */
280      private void moveDownItem(int index) {
281        int lastIndex = viewColumnTable.getItemCount() - 1;
282        if (index < lastIndex) {
283          List columnDataList = (List) this.reviewPhaseMap.get(this.reviewPhaseNameKey);
284          ColumnData columnData = (ColumnData) columnDataList.remove(index);
285          int newIndex = index + 1;
286          columnDataList.add(newIndex, columnData);
287          removeAllItems();
288          fillTable(columnDataList);
289          viewColumnTable.select(newIndex);
290        }
291      }

293      /**
294       * Performs OK to save changes when this page's OK button has been pressed.
295       *
296       * @return whether it is okay to close the preference page
297       */
298      public boolean performOk() {
299        ReviewModel reviewModel = ReviewModel.getInstance();
300        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
301        for (Iterator i = this.reviewPhaseMap.entrySet().iterator(); i.hasNext();) {
302          Map.Entry entry = (Map.Entry) i.next();
303          String reviewPhaseNameKey = (String) entry.getKey();
304          List columnDataList = (List) entry.getValue();
305          ColumnDataModel columnDataModel = columnDataModelManager.getModel();
306          columnDataModel.clear();
307          columnDataModel.addAll(columnDataList);
308          PrefResource.getInstance().storeColumnDataModel(reviewPhaseNameKey, columnDataModel);
309        }
310        // Updates the view.
311        ColumnDataModel columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKey);
312        ReviewTableView view = ReviewTableView.getActiveView();
313        // null happens when the view is not opened yet after Eclipse startup.
314        if (view == null) {
315          view = ReviewTableView.bringViewToTop();
316        }
317        view.createColumns(columnDataModel);
318        view.getViewer().refresh();
319        return true;
320      }

322      /**
323       * Performs OK to save changes when this page's Apply button has been pressed.
324       */
325      public void performApply() {
326        performOk();
327      }

329      /**
330       * Performs Defaults to restore default values when this page's Defaults button has been pressed.
331       */
332      protected void performDefaults() {
333        String reviewPhaseName = reviewPhaseCombo.getItem(reviewPhaseCombo.getSelectionIndex());
334        reviewPhaseNameKey = ReviewI18n.getKey(reviewPhaseName);
335        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
336        ColumnDataModel columnDataModel = columnDataModelManager.getModel(reviewPhaseNameKey);
337        List columnDataList = (List) reviewPhaseMap.get(reviewPhaseNameKey);
338        columnDataList.clear();
339        columnDataList.addAll(new ArrayList(Arrays.asList(columnDataModel.getAllColumnDataArray())));
340        removeAllItems();
341        fillTable(columnDataList);
342      }
343    
344      /**
345       * Fills table with the column name, width, and enable status.
346       * @param columnDataList the <code>ColumnData</code> list.
347       */
348      private void fillTable(List columnDataList) {
349        for (Iterator i = columnDataList.iterator(); i.hasNext();) {
350          ColumnData columnData = (ColumnData) i.next();
351          String columnName = columnData.getColumnName();
352          int width = columnData.getColumnPixelData().width;
353          boolean isEnabled = columnData.isEnabled();
354          TableItem item = new TableItem(viewColumnTable, SWT.NONE);
355          item.setText(columnName + " [" + width + "]");
356          item.setChecked(isEnabled);
357        }
358      }
359      
360      /**
361       * Remove all items in the table.
362       */
363      private void removeAllItems() {
364        while (viewColumnTable.getItemCount() > 0) {
365          int lastIndex = viewColumnTable.getItemCount() - 1;
366          viewColumnTable.getItem(lastIndex).dispose();
367        }
368      }
369    
370      /**
371       * Modifies a column width.
372       */
373      private void modifyWidth() {
374        int index = viewColumnTable.getSelectionIndex();
375        IInputValidator validator = new IInputValidator() {
376          public String isValid(String newText) {
377            try {
378              Integer.parseInt(newText);
379              return null;
380            }
381            catch (NumberFormatException e) {
382              return ReviewI18n.getString("ViewPreference.button.messageDialog.error");
383            }
384          }
385        };
386        List columnDataList = (List) this.reviewPhaseMap.get(this.reviewPhaseNameKey);
387        ColumnData columnData = (ColumnData) columnDataList.get(index);
388        String initialWidthValue = columnData.getColumnPixelData().width + "";
389        InputDialog dialog = new InputDialog(getShell(), 
390            ReviewI18n.getString("ViewPreference.button.messageDialog.short"),
391            ReviewI18n.getString("ViewPreference.button.messageDialog.long"),
392            initialWidthValue, validator);  //$NON-NLS-1$ //$NON-NLS-2$
393        dialog.open();
394        if (dialog.getReturnCode() != InputDialog.OK) {
395          return;
396        }
397        String modifiedWidthValue = dialog.getValue();
398        int width = Integer.parseInt(modifiedWidthValue);
399        columnData.getColumnPixelData().width = width;
400        TableItem item = viewColumnTable.getItem(index);
401        item.setText(columnData.getColumnName() + " [" + width + "]");
402      }

404      /**
405       * Handles the table selection.
406       * @param tableItem the <code>TableItem</code> instance.
407       */
408      private void handleSelection(TableItem tableItem) {
409        int index = viewColumnTable.indexOf(tableItem);
410        List columnDataList = (List) this.reviewPhaseMap.get(this.reviewPhaseNameKey);
411        ColumnData columnData = (ColumnData) columnDataList.get(index);
412        columnData.setEnabled(tableItem.getChecked());    
413        boolean isSelected = (viewColumnTable.getSelectionCount() > 0);
414        modifyButton.setEnabled(isSelected);
415        upButton.setEnabled(isSelected);
416        downButton.setEnabled(isSelected);
417      }
