049      /** Jupiter logger */
050      private JupiterLogger log = JupiterLogger.getLogger();
051    
052      private static final String TABLE_COLUMN = "TableColumn";
053      private static final String COLUMN_KEY = "ColumnKey";
054      private IProject project;
055      private TableViewer tableViewer;
056      private Button newButton;
057      private Table table;
058      private Button removeButton;
059      private Button editButton;
060      private Composite composite;
061      /** The column review ID key. */
062      public static final String COLUMN_REVIEW_ID_KEY = "ReviewPropertyPage.label.column.reviewId";
063      /** The  column description key. */
064      public static final String COLUMN_DESCRIPTION_KEY = "ReviewPropertyPage.label.column.description";
065      /** The column date key. */
066      public static final String COLUMN_DATE_KEY = "ReviewPropertyPage.label.column.date";
067      /**
068       * Creates content.
069       * @param ancestor the composite.
070       * @return the control.
071       */
072      protected Control createContents(Composite ancestor) {
073        this.composite = ancestor;
074        this.project = (IProject) getElement();
075        noDefaultAndApplyButton();
076        Composite parent = createsGeneralComposite(ancestor);
077        createReviewIdTableContent(parent);
078        createButtonsContent(parent);
079        return parent;
080      }
081    
082      /**
083       * Creates view preference frame and return the child composite.
084       * @param parent the parent composite.
085       * @return the child composite.
086       */
087      private Composite createsGeneralComposite(Composite parent) {
088        Composite child = new Composite(parent, SWT.LEFT);
089        FormLayout layout = new FormLayout();
090        layout.marginWidth = 7;
091        layout.marginHeight = 7;
092        child.setLayout(layout);
093        return child;
094      }
095      
096      /**
097       * Creates review id table.
098       * @param parent the composite.
099       */
100      private void createReviewIdTableContent(Composite parent) {
101        this.table = new Table(parent, SWT.BORDER | SWT.FULL_SELECTION);
102        FormData tableData = new FormData();
103        tableData.left = new FormAttachment(0, 0);
104        tableData.right = new FormAttachment(80, 0);
105        tableData.top = new FormAttachment(0, 0);
106        tableData.bottom = new FormAttachment(100, 0);
107        table.setLayoutData(tableData);
108        table.setHeaderVisible(true);
109        table.setLinesVisible(true);
110        table.addListener(SWT.Selection, new Listener() {
111          public void handleEvent(Event e) {
112            handleReviewIdSelection();
113          }
114        });
115        TableColumn columnReviewId = new TableColumn(table, SWT.NONE);
116        columnReviewId.setText(ReviewI18n.getString(COLUMN_REVIEW_ID_KEY));
117        columnReviewId.setData(COLUMN_KEY, COLUMN_REVIEW_ID_KEY);
118        TableColumn columnDescription = new TableColumn(table, SWT.NONE);
119        String description = ReviewI18n.getString(COLUMN_DESCRIPTION_KEY);
120        columnDescription.setText(description);
121        columnDescription.setData(COLUMN_KEY, COLUMN_DESCRIPTION_KEY);
122        TableColumn columnDate = new TableColumn(table, SWT.NONE);
123        columnDate.setText(ReviewI18n.getString(COLUMN_DATE_KEY));
124        columnDate.setData(COLUMN_KEY, COLUMN_DATE_KEY);
125        
126        List columnList = new ArrayList();
127        columnList.add(columnReviewId);
128        columnList.add(columnDescription);
129        columnList.add(columnDate);
130        hookSelectionListener(columnList);
131        
132        TableLayout tableLayout = new TableLayout();
133        tableLayout.addColumnData(new ColumnWeightData(22));
134        tableLayout.addColumnData(new ColumnWeightData(48));
135        tableLayout.addColumnData(new ColumnWeightData(20));
136        table.setLayout(tableLayout);
137        
138        this.tableViewer = new TableViewer(table);
139        tableViewer.setLabelProvider(new ReviewPropertyLabelProvider());
140        tableViewer.setContentProvider(new ReviewPropertyContentProvider());
141        tableViewer.setSorter(ReviewPropertyViewerSorter.getViewerSorter(COLUMN_DATE_KEY));
142        tableViewer.setInput(PropertyResource.getInstance(this.project, true).getReviewIdList());
143        
144        // routine for the version 1 compatibility.
145        if (tableViewer.getTable().getItemCount() <= 0) {
146          // version 1.
147          List reviewIdList = Ver1PropertyHelper.getReviewIdList(this.project.getName(), false);
148          IFile jupiterConfigFile = project.getFile(PropertyXmlSerializer.PROPERTY_XML_FILE);
149          FileResource.remove(new IFile[] {jupiterConfigFile});
150          // create version 2 .jupiter file.
151          String defaultName = PropertyConstraints.DEFAULT_REVIEW_ID;
152          PropertyResource propertyResource = null;
153          for (Iterator i = reviewIdList.iterator(); i.hasNext();) {
154            ReviewId reviewId = (ReviewId) i.next();
155            propertyResource = PropertyResource.getInstance(this.project, true);
156            ReviewResource reviewResource = propertyResource.getReviewResource(defaultName, true);
157            ReviewId defaultReviewId = reviewResource.getReviewId();
158            String directory = defaultReviewId.getDirectory();
159            reviewId.setDirectory(directory);
160            reviewResource.setReviewId(reviewId);
161            try {
162              propertyResource.addReviewResource(reviewResource);
163            }
164            catch (ReviewException e) {
165              log.debug(e.getMessage());
166            }
167          }
168          tableViewer.setInput(propertyResource.getReviewIdList());
169        }
170        tableViewer.addDoubleClickListener(new IDoubleClickListener() {
171          public void doubleClick(DoubleClickEvent event) {
172            editReviewId();
173          }
174        });
175      }
176    
177      /**
178       * Creates buttons content.
179       * @param parent the parent.
180       */
181      private void createButtonsContent(Composite parent) {
182        this.newButton = new Button(parent, SWT.PUSH);
183        newButton.setText(ReviewI18n.getString("ReviewPropertyPage.label.button.new"));
184        newButton.addListener(SWT.Selection, new Listener() {
185          public void handleEvent(Event e) {
186              addReviewId();
187          }
188        });
189        FormData newButtonData = new FormData();
190        newButtonData.top = new FormAttachment(table, 0, SWT.TOP);
191        newButtonData.left = new FormAttachment(table, 10);
192        newButtonData.right = new FormAttachment(100, 0);
193        newButton.setLayoutData(newButtonData);
194        
195        this.editButton = new Button(parent, SWT.PUSH);
196        editButton.setText(ReviewI18n.getString("ReviewPropertyPage.label.button.edit"));
197        editButton.setEnabled(false);
198        editButton.addListener(SWT.Selection, new Listener() {
199          public void handleEvent(Event e) {
200              editReviewId();
201          }
202        });
203        FormData editButtonData = new FormData();
204        editButtonData.top = new FormAttachment(newButton, 5);
205        editButtonData.left = new FormAttachment(newButton, 0, SWT.LEFT);
206        editButtonData.right = new FormAttachment(100, 0);
207        editButton.setLayoutData(editButtonData);
208        
209        
210        this.removeButton = new Button(parent, SWT.PUSH);
211        removeButton.setText(ReviewI18n.getString("ReviewPropertyPage.label.button.remove"));
212        removeButton.setEnabled(false);
213        removeButton.addListener(SWT.Selection, new Listener() {
214          public void handleEvent(Event e) {
215            removeReviewId();
216          }
217        });
218        FormData removeButtonData = new FormData();
219        removeButtonData.top = new FormAttachment(editButton, 5);
220        removeButtonData.left = new FormAttachment(newButton, 0, SWT.LEFT);
221        removeButtonData.right = new FormAttachment(100, 0);
222        removeButton.setLayoutData(removeButtonData);
223      }

225      /**
226       * Hooks selection listener for each <code>TableColumn</code> element of the list.
227       * @param columnList the list of the <code>TableColumn</code> elements.
228       */
229      private void hookSelectionListener(List columnList) {
230        for (Iterator i = columnList.iterator(); i.hasNext();) {
231          TableColumn column = (TableColumn) i.next();
232          column.addListener(SWT.Selection, new Listener() {
233            public void handleEvent(Event event) {
234              String columnKey = (String) event.widget.getData(COLUMN_KEY);
235              sortBy(columnKey);
236            }
237          });
238        }
239      }
240      
241      /**
242       * Sorts by the <code>String</code> columnKey.
243       * @param columnKey the <code>String</code> columnKey.
244       */
245      protected void sortBy(String columnKey) {
246        ViewerSorter viewerSorter = ReviewPropertyViewerSorter.getViewerSorter(columnKey);
247        if (viewerSorter != null) {
248          ViewerSorter previousSorter = tableViewer.getSorter();
249          if (previousSorter == viewerSorter) {
250            ReviewPropertyViewerSorter.setReverse(!ReviewPropertyViewerSorter.isReverse());
251            // Resets sorter.
252            tableViewer.setSorter(null);
253          }
254          else {
255            ReviewPropertyViewerSorter.setReverse(false);
256          }
257          tableViewer.setSorter(viewerSorter);
258        }
259      }
260    
261      /**
262       * Adds review ID so as to open review id addition wizard.
263       */
264      private void addReviewId() {
265        ReviewDialog.processConfigWizardDialog(this.project);
266        List reviewIdList = PropertyResource.getInstance(this.project, true).getReviewIdList();
267        tableViewer.setInput(reviewIdList);
268      }
269      
270      /**
271       * Edits the selected review ID.
272       */
273      private void editReviewId() {
274        int selectedIndex = table.getSelectionIndex();
275        if (selectedIndex >= 0) {
276          ReviewId reviewId = (ReviewId) tableViewer.getElementAt(selectedIndex);
277          Dialog dialog = new ReviewIdEditDialog(composite.getShell(), this.project, reviewId);
278          dialog.open();
279          this.tableViewer.setInput(PropertyResource.getInstance(this.project, true).getReviewIdList());
280        }
281      }
282      
283      /**
284       * Removes the selected review ID
285       */
286      private void removeReviewId() {
287        int selectedIndex = table.getSelectionIndex();
288        if (selectedIndex >= 0) {
289          ReviewId reviewId = (ReviewId) tableViewer.getElementAt(selectedIndex);
290          IFile[] reviewIFiles = FileResource.getReviewIFiles(this.project, reviewId);
291          Dialog dialog = new ReviewIdRemovalDialog(composite.getShell(), reviewIFiles);
292          dialog.open();
293          if (dialog.getReturnCode() == Dialog.OK) {
294            // remove review files associated with the review id.
295            FileResource.remove(reviewIFiles);
296            try {
297              PropertyResource propertyResource = PropertyResource.getInstance(this.project, false);
298              propertyResource.removeReviewResource(reviewId);
299            }
300            catch (ReviewException e) {
301              log.error(e);
302            }
303          }
304        }
305        tableViewer.setInput(PropertyResource.getInstance(this.project, true).getReviewIdList());
306      }
307    
308      /**
309       * Handles review id selection
310       */
311      protected void handleReviewIdSelection() {
312        int index = this.table.getSelectionIndex();
313        boolean isSelected = (index >= 0);
314        this.newButton.setEnabled(isSelected);
315        this.editButton.setEnabled(isSelected);   
316        this.removeButton.setEnabled(isSelected);
317        TableItem item = this.table.getItem(index);
318        ReviewId reviewId = (ReviewId) item.getData();
319        if (reviewId.getReviewId().equals(PropertyConstraints.DEFAULT_REVIEW_ID)) {
320          this.removeButton.setEnabled(false);
321        }
322      }
