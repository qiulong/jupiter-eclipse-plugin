001    package csdl.jupiter.ui.property;
002    
003    import java.text.SimpleDateFormat;
004    
005    import org.eclipse.jface.viewers.ITableLabelProvider;
006    import org.eclipse.jface.viewers.LabelProvider;
007    import org.eclipse.swt.graphics.Image;
008    
009    import csdl.jupiter.ReviewI18n;
010    import csdl.jupiter.model.review.ReviewId;
011    
012    /**
013     * Provides config property label provider.
014     * @author Takuya Yamashita
015     * @version $Id: ReviewPropertyLabelProvider.java,v 1.1 2004/08/19 12:40:32 takuyay Exp $
016     */
017    public class ReviewPropertyLabelProvider extends LabelProvider implements ITableLabelProvider {
018       
019      private SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd");
020      /**
021       * @see org.eclipse.jface.viewers.ITableLabelProvider#getColumnImage(java.lang.Object, int)
022       */
023      public Image getColumnImage(Object element, int columnIndex) {
024        return null;
025      }
026    
027      /**
028       * @see org.eclipse.jface.viewers.ITableLabelProvider#getColumnText(java.lang.Object, int)
029       */
030      public String getColumnText(Object element, int columnIndex) {
031        String columnText = element.toString();
032        if (element instanceof ReviewId) {
033          ReviewId reviewId = (ReviewId) element;
034          switch (columnIndex) {
035            case 0:
036              columnText = reviewId.getReviewId();
037              break;
038            case 1:
039              columnText = ReviewI18n.getString(reviewId.getDescription());
040              break;
041            case 2:
042              columnText = this.format.format(reviewId.getDate());
043              break;
044            default:
045              break;
046          }
047        } 
048        return columnText;
049      }
050    
051    }
