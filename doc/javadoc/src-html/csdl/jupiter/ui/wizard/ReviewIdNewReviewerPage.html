056      /**
057       * Creates control.
058       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
059       */
060      public void createControl(Composite ancestor) {
061        Composite parent = createsGeneralComposite(ancestor);

068      /**
069       * Creates view preference frame and return the child composite.
070       * @param parent the parent composite.
071       * @return the child composite.
072       */

074        Composite child = new Composite(parent, SWT.LEFT);
075        FormLayout layout = new FormLayout();
076        layout.marginWidth = 7;
077        layout.marginHeight = 7;
078        child.setLayout(layout);

082      /**
083       * Creates view preference content.
084       * @param parent the parent composite.
085       */
086      private void createrReviewerListContent(Composite parent) {    
087        this.reviewerListTable = new Table(parent, SWT.BORDER | SWT.MULTI);
088        FormData reviewerListTableData = new FormData();
089        reviewerListTableData.left = new FormAttachment(0, 0);
090        reviewerListTableData.right = new FormAttachment(80, 0);
091        reviewerListTableData.top = new FormAttachment(0, 0);

093        reviewerListTable.setLayoutData(reviewerListTableData);
094        reviewerListTable.addListener(SWT.Selection, new Listener() {
095          public void handleEvent(Event e) {
096            handleReviewListSelection();
097          }
098        });
099        
100        this.addButton = new Button(parent, SWT.PUSH);
101        addButton.setText(ReviewI18n.getString("ReviewIdNewReviewerPage.label.button.add"));
102        addButton.addListener(SWT.Selection, new Listener() {
103          public void handleEvent(Event e) {
104              addReviewer();
105          }
106        });
107        FormData addButtonData = new FormData();
108        addButtonData.top = new FormAttachment(reviewerListTable, 0, SWT.TOP);
109        addButtonData.left = new FormAttachment(reviewerListTable, 10);
110        addButtonData.right = new FormAttachment(100, 0);
111        addButton.setLayoutData(addButtonData);
112        
113        
114        this.removeButton = new Button(parent, SWT.PUSH);
115        removeButton.setText(ReviewI18n.getString("ReviewIdNewReviewerPage.label.button.remove"));
116        removeButton.setEnabled(false);
117        removeButton.addListener(SWT.Selection, new Listener() {
118          public void handleEvent(Event e) {
119            removeReviewer();
120          }
121        });
122        FormData removeButtonData = new FormData();
123        removeButtonData.top = new FormAttachment(addButton, 5);
124        removeButtonData.left = new FormAttachment(addButton, 0, SWT.LEFT);
125        removeButtonData.right = new FormAttachment(100, 0);
126        removeButton.setLayoutData(removeButtonData);
127      }
128      
129      /**
130       * Fills the reviewer table with reviewers. Sets <code>true</code> if items are just updated.
131       * Sets <code>false</code> if items are read from the property resource.
132       * @param isUpdate <code>true</code> if items are just updated.
133       * <code>false</code> if items are read from the property resource.
134       */
135      public void fillReviewerTable(boolean isUpdate) {
136        removeAllItemsInReviewerTable();
137        if (!isUpdate) {
138          PropertyResource reviewIdResource = PropertyResource.getInstance(this.project, true);
139          Map reviewersMap = reviewIdResource.getReviewers(PropertyResource.DEFAULT_ID);
140          this.reviewers = new TreeMap(reviewersMap);
141        }
142        for (Iterator i = this.reviewers.keySet().iterator(); i.hasNext();) {
143          String reviewerId = (String) i.next();
144          TableItem item = new TableItem(this.reviewerListTable, SWT.NONE);
145          item.setText(reviewerId);
146        }
147      }
148      
149      /**
150       * Removes all items in the review table.
151       */
152      private void removeAllItemsInReviewerTable() {
153        TableItem[] items = this.reviewerListTable.getItems();
154        for (int i = 0; i < items.length; i++) {
155          items[i].dispose();
156        }
157      }

209      /**
210       * Adds a reviewer to the reviewer list.
211       */
212      protected void addReviewer() {
213        IInputValidator validator = new IInputValidator() {
214          public String isValid(String newText) {
215            if (!reviewers.containsKey(newText)) {
216              return null;
217            }
218            else {
219              String errorKey = "ReviewIdNewReviewerPage.dialogMessage.label.error";
220              return ReviewI18n.getString(errorKey);
221            }
222          }
223        };
224        InputDialog dialog = new InputDialog(getShell(), 
225            ReviewI18n.getString("ReviewIdNewReviewerPage.dialogMessage.label.short"),
226            ReviewI18n.getString("ReviewIdNewReviewerPage.dialogMessage.label.long"),
227            null, validator);  //$NON-NLS-1$ //$NON-NLS-2$
228        dialog.open();
229        if (dialog.getReturnCode() != InputDialog.OK) {
230          return;
231        }
232        String reviewerId = dialog.getValue();
233        ReviewerId reviewer = new ReviewerId(reviewerId, reviewerId);
234        this.reviewers.put(reviewerId, reviewer);
235        fillReviewerTable(true);
236        updateAuthorCandidates(this.reviewers);
237        getWizard().getContainer().updateButtons();
238        setPageComplete(isTableNonEmpty());
239      }
240      
241      /**
242       * Removes a reviewer form the reviewer list.
243       */
244      private void removeReviewer() {
245        TableItem[] items = this.reviewerListTable.getSelection();
246        for (int i = 0; i < items.length; i++) {
247          TableItem item = items[i];
248          this.reviewers.remove(item.getText());
249          item.dispose();
250        }
251        fillReviewerTable(true);
252        this.removeButton.setEnabled(this.reviewerListTable.getItemCount() > 0);
253        getWizard().getContainer().updateButtons();
254        setPageComplete(isTableNonEmpty());
255      }
256      
257      /**
258       * Handles the table selection.
259       */
260      private void handleReviewListSelection() {
261        boolean isSelected = (this.reviewerListTable.getSelectionIndex() >= 0);
262        this.addButton.setEnabled(isSelected);
263        this.removeButton.setEnabled(isSelected);
264      }

266      /**

268       * @return the reviewers.
269       */
270      public Map getReviews() {
271        return this.reviewers;
272      }
273      
274      /**
275       * Updates the author candidate list.
276       * @param reviewers the reviewers which would be the candidates for the author.
277       */
278      private void updateAuthorCandidates(Map reviewers) {
279        IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_AUTHOR);
280        ReviewIdNewAuthorPage configAuthorPage = (ReviewIdNewAuthorPage) page;
281        String selectedAuthor = configAuthorPage.getAuthor();
282        configAuthorPage.setAuthorItems((String[]) this.reviewers.keySet().toArray(new String[] {}));
283        configAuthorPage.setDefaultAuthor(selectedAuthor);
284      }

308        return (this.reviewers.size() > 0);
309      }
