023    /**
024     * Provides review ID selection page.
025     * @author Takuya Yamashita
026     * @version $Id: ReviewIdSelectionPage.java,v 1.24 2005/03/25 18:32:06 takuyay Exp $
027     */
028    public class ReviewIdSelectionPage extends WizardPage {
029      private Combo reviewIdCombo;
030      private Combo projectCombo;
031      private Combo reviewerIdCombo;
032      private String reviewId;
033      private Text reviewIdDescriptionText;
034      private IProject selectedProject;
035      private boolean isMultipleChoiceEnabled;
036      /**
037       * Instantiates review id selection page.
038       * @param defaultProject the selectedProject to be shown as a default in the dialog.
039       * @param isMultipleChoiceEnabled set <code>true</code> if the selectedProject choice
040       * is multiple.
041       * <code>false</code> if the selectedProject choice is only the passing selectedProject.
042       * @param pageName the page name.
043       */
044      public ReviewIdSelectionPage(IProject defaultProject, boolean isMultipleChoiceEnabled,
045                                   String pageName) {
046        super(pageName);
047        String imageFilePath = "lib/icons/selection_wiz.gif";
048        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
049        setTitle(ReviewI18n.getString("ReviewIdSelectionPage.label.title"));
050        setDescription(ReviewI18n.getString("ReviewIdSelectionPage.label.title.description"));
051        this.selectedProject = defaultProject;
052        this.isMultipleChoiceEnabled = isMultipleChoiceEnabled;
053      }
054    
055      /**
056       * Creates control.
057       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
058       */
059      public void createControl(Composite ancestor) {
060        Composite parent = createsGeneralComposite(ancestor);
061        createProjectSelectionContent(parent);
062        createReviewIdSelectionContent(parent);
063        craeteReviewIdDescriptionContent(parent);
064        createReviewerIdSelectionContent(parent);
065        //  set the composite as the control for this page
066        setControl(parent);
067      }
068    
069      /**
070       * Creates view preference frame and return the child composite.
071       * 
072       * @param parent the parent composite.
073       * @return the child composite.
074       */
075      private Composite createsGeneralComposite(Composite parent) {
076        Composite child = new Composite(parent, SWT.LEFT);
077        FormLayout layout = new FormLayout();
078        layout.marginWidth = 7;
079        layout.marginHeight = 7;
080        child.setLayout(layout);
081        return child;
082      }
083        
084      /**
085       * Handles the selectedProject selection event.
086       * @param e the event.
087       */
088      private void handleProjectSelection(Event e) {
089        String projectName = ((Combo) e.widget.getData()).getText();
090        this.selectedProject = FileResource.getProject(projectName);
091        PropertyResource reviewIdResource = PropertyResource.getInstance(this.selectedProject, false);
092        this.reviewIdCombo.setItems(reviewIdResource.getReviewIdNames());
093        reviewIdCombo.select(0);
094        this.reviewId = reviewIdCombo.getItem(0);
095        ReviewId reviewId = reviewIdResource.getReviewId(this.reviewId);
096        this.reviewIdDescriptionText.setText(reviewId.getDescription());
097        String[] items = reviewIdResource.getReviewerIdNames(this.reviewId);
098        this.reviewerIdCombo.setItems(items);
099        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
100        String reviewerId = store.getString(ReviewIdSelectionWizard.REVIEWER_ID);
101        int index = reviewerIdCombo.indexOf(reviewerId);
102        reviewerIdCombo.select((index != -1) ? index : 0);
103      }
104      
105      /**
106       * Handles the review id selection event.
107       * @param e the event.
108       */
109      private void handleReviewIdSelection(Event e) {
110        this.reviewId = ((Combo) e.widget.getData()).getText();
111        PropertyResource reviewIdResource = PropertyResource.getInstance(this.selectedProject, false);
112        this.reviewerIdCombo.setItems(reviewIdResource.getReviewerIdNames(reviewId));
113        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
114        String reviewerId = store.getString(ReviewIdSelectionWizard.REVIEWER_ID);
115        int index = reviewerIdCombo.indexOf(reviewerId);
116        reviewerIdCombo.select((index != -1) ? index : 0);
117        ReviewId reviewId = reviewIdResource.getReviewId(this.reviewId);
118        this.reviewIdDescriptionText.setText(reviewId.getDescription());
119      }

123      /**
124       * Creates selectedProject selection content.
125       * @param parent the parent.
126       */
127      private void createProjectSelectionContent(Composite parent) {
128        Label projectLabel = new Label(parent, SWT.NONE);
129        projectLabel.setText(ReviewI18n.getString("ReviewIdSelectionPage.label.project"));
130        this.projectCombo = new Combo(parent, SWT.READ_ONLY);
131        if (this.isMultipleChoiceEnabled) {
132          if (this.selectedProject != null) {
133            projectCombo.setItems(FileResource.getOpenedAndReviewIdContainedProjects());
134            int index = projectCombo.indexOf(this.selectedProject.getName());
135            projectCombo.select((index >= 0) ? index : 0);
136          }
137        }
138        else {
139          if (this.selectedProject != null) {
140            projectCombo.setItems(new String[] {this.selectedProject.getName()});
141            projectCombo.select(0);
142          }
143        }
144        projectCombo.setData(projectCombo);
145        projectCombo.addListener(SWT.Selection, new Listener() {
146          public void handleEvent(Event e) {
147            handleProjectSelection(e);
148          }
149        });
150    
151        FormData projectLabelData = new FormData();
152        projectLabelData.top = new FormAttachment(projectCombo, 0, SWT.CENTER);
153        projectLabel.setLayoutData(projectLabelData);
154        FormData projectComboData = new FormData();
155        projectComboData.left = new FormAttachment(projectLabel, 40);
156        projectComboData.right = new FormAttachment(100, 0);
157        projectCombo.setLayoutData(projectComboData);
158      }
159    
160      /**
161       * Creates review id selection content.
162       * 
163       * @param parent the parent.
164       */
165      private void createReviewIdSelectionContent(Composite parent) {
166        Label reviewIdLabel = new Label(parent, SWT.NONE);
167        reviewIdLabel.setText(ReviewI18n.getString("ReviewIdSelectionPage.label.reviewId"));
168        this.reviewIdCombo = new Combo(parent, SWT.READ_ONLY);
169        PropertyResource reviewIdResource = PropertyResource.getInstance(this.selectedProject, false);
170        reviewIdCombo.setItems(reviewIdResource.getReviewIdNames());
171        if (reviewIdCombo.getItemCount() > 0) {
172          reviewIdCombo.select(0);
173          this.reviewId = reviewIdCombo.getItem(0);
174        }
175        reviewIdCombo.setData(reviewIdCombo);
176        reviewIdCombo.addListener(SWT.Selection, new Listener() {
177          public void handleEvent(Event e) {
178            handleReviewIdSelection(e);
179          }
180        });
181    
182        FormData reviewIdLabelData = new FormData();
183        reviewIdLabelData.top = new FormAttachment(reviewIdCombo, 0, SWT.CENTER);
184        reviewIdLabel.setLayoutData(reviewIdLabelData);
185        FormData reviewIdComboData = new FormData();
186        reviewIdComboData.top = new FormAttachment(projectCombo, 5);
187        reviewIdComboData.left = new FormAttachment(projectCombo, 0, SWT.LEFT);
188        reviewIdComboData.right = new FormAttachment(100, 0);
189        reviewIdCombo.setLayoutData(reviewIdComboData);
190      }
191      
192      /**
193       * Creates review id description content.
194       * @param parent the parent.
195       */
196      private void craeteReviewIdDescriptionContent(Composite parent) {
197        Label reviewIdDescriptionLabel = new Label(parent, SWT.NONE);
198        String reviewIdDescriptionKey = "ReviewIdSelectionPage.label.reviewIdDescription";
199        reviewIdDescriptionLabel.setText(ReviewI18n.getString(reviewIdDescriptionKey));
200        this.reviewIdDescriptionText = new Text(parent, SWT.SINGLE | SWT.BORDER | SWT.READ_ONLY);
201        PropertyResource reviewIdResource = PropertyResource.getInstance(this.selectedProject, false);
202        ReviewId reviewId = reviewIdResource.getReviewId(this.reviewId);
203        String reviewIdDescription = "";
204        if (reviewId != null) {
205          reviewIdDescription = reviewId.getDescription();
206        }
207        reviewIdDescriptionText.setText(reviewIdDescription);
208    
209        FormData reviewIdDescriptionLabelData = new FormData();
210        reviewIdDescriptionLabelData.top = new FormAttachment(reviewIdDescriptionText, 0, SWT.CENTER);
211        reviewIdDescriptionLabel.setLayoutData(reviewIdDescriptionLabelData);
212        FormData reviewIdDescriptionTextData = new FormData();
213        reviewIdDescriptionTextData.top = new FormAttachment(reviewIdCombo, 5);
214        reviewIdDescriptionTextData.left = new FormAttachment(projectCombo, 0, SWT.LEFT);
215        reviewIdDescriptionTextData.right = new FormAttachment(100, 0);
216        reviewIdDescriptionText.setLayoutData(reviewIdDescriptionTextData);
217      }
218      
219      /**
220       * Creates reviewer id selection content.
221       * @param parent the parent.
222       */
223      private void createReviewerIdSelectionContent(Composite parent) {
224        Label reviewerIdLabel = new Label(parent, SWT.NONE);
225        reviewerIdLabel.setText(ReviewI18n.getString("ReviewIdSelectionPage.label.reviewerId"));
226        this.reviewerIdCombo = new Combo(parent, SWT.READ_ONLY);
227        PropertyResource reviewIdResource = PropertyResource.getInstance(this.selectedProject, false);
228        reviewerIdCombo.setItems(reviewIdResource.getReviewerIdNames(this.reviewId));
229        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
230        String reviewerId = store.getString(ReviewIdSelectionWizard.REVIEWER_ID);
231        int index = reviewerIdCombo.indexOf(reviewerId);
232        reviewerIdCombo.select((index != -1) ? index : 0);
233    
234        FormData reviewerIdLabelData = new FormData();
235        reviewerIdLabelData.top = new FormAttachment(reviewerIdCombo, 0, SWT.CENTER);
236        reviewerIdLabel.setLayoutData(reviewerIdLabelData);
237        FormData reviewerIdComboData = new FormData();
238        reviewerIdComboData.top = new FormAttachment(reviewIdDescriptionText, 5);
239        reviewerIdComboData.left = new FormAttachment(projectCombo, 0, SWT.LEFT);
240        reviewerIdComboData.right = new FormAttachment(100, 0);
241        reviewerIdCombo.setLayoutData(reviewerIdComboData);
242      }
