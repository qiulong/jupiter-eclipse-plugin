036    /**
037     * Provides review file configuration page.
038     * @author Takuya Yamashita
039     * @version $Id: ReviewIdNewFilePage.java,v 1.6 2005/03/25 18:32:06 takuyay Exp $
040     */
041    public class ReviewIdNewFilePage extends WizardPage {
042    
043      private IProject project;
044      private Set files;
045      private Table fileListTable;
046      private Button addButtonInFile;
047      private Button removeButtonInFile;
048    
049      /**
050       * Instantiates the review file configuration page.
051       * @param project the project.
052       * @param pageName the page name
053       * @param imageFilePath the image file path.
054       */
055      protected ReviewIdNewFilePage(IProject project, String pageName, String imageFilePath) {
056        super(pageName);
057        setImageDescriptor(ReviewPlugin.createImageDescriptor(imageFilePath));
058        setTitle(ReviewI18n.getString("ReviewIdNewFilePage.label.title"));
059        setDescription(ReviewI18n.getString("ReviewIdNewFilePage.label.title.description"));
060        this.project = project;
061      }
062    
063      /**
064       * @see org.eclipse.jface.dialogs.IDialogPage#createControl(org.eclipse.swt.widgets.Composite)
065       */
066      public void createControl(Composite ancestor) {
067        Composite parent = createsGeneralComposite(ancestor);
068        createFileTable(parent);
069        //  set the composite as the control for this page
070        setControl(parent);
071        setPageComplete(isTableNonEmpty());
072      }
073      
074      /**
075       * Creates view preference frame and return the child composite.
076       * @param parent the parent composite.
077       * @return the child composite.
078       */
079      private Composite createsGeneralComposite(Composite parent) {
080        Composite child = new Composite(parent, SWT.LEFT);
081        FormLayout layout = new FormLayout();
082        layout.marginWidth = 7;
083        layout.marginHeight = 7;
084        child.setLayout(layout);
085        return child;
086      }
087      
088      /**
089       * Creates the file table.
090       * @param composite the composite.
091       * @return the control.
092       */
093      private Control createFileTable(Composite composite) {
094        this.fileListTable = new Table(composite, SWT.BORDER | SWT.MULTI);
095        FormData fileListTableData = new FormData();
096        fileListTableData.left = new FormAttachment(0, 0);
097        fileListTableData.right = new FormAttachment(80, 0);
098        fileListTableData.top = new FormAttachment(0, 0);
099        fileListTableData.bottom = new FormAttachment(100, 0);
100        fileListTableData.height = 150;
101        fileListTable.setLayoutData(fileListTableData);
102        fileListTable.addListener(SWT.Selection, new Listener() {
103          public void handleEvent(Event e) {
104            handleFileListSelection();
105          }
106        });
107        fillFileTable(this.project.getName());
108        
109        this.addButtonInFile = new Button(composite, SWT.PUSH);
110        String addKey = "ReviewIdEditDialog.label.tab.file.button.add";
111        addButtonInFile.setText(ReviewI18n.getString(addKey));
112        addButtonInFile.addListener(SWT.Selection, new Listener() {
113          public void handleEvent(Event e) {
114              addFile();
115          }
116        });
117        FormData addButtonData = new FormData();
118        addButtonData.top = new FormAttachment(fileListTable, 0, SWT.TOP);
119        addButtonData.left = new FormAttachment(fileListTable, 10);
120        addButtonData.right = new FormAttachment(100, 0);
121        addButtonInFile.setLayoutData(addButtonData);
122        
123        
124        this.removeButtonInFile = new Button(composite, SWT.PUSH);
125        String removeKey = "ReviewIdEditDialog.label.tab.file.button.remove";
126        removeButtonInFile.setText(ReviewI18n.getString(removeKey));
127        removeButtonInFile.setEnabled(false);
128        removeButtonInFile.addListener(SWT.Selection, new Listener() {
129          public void handleEvent(Event e) {
130            removeFile();
131          }
132        });
133        FormData removeButtonData = new FormData();
134        removeButtonData.top = new FormAttachment(addButtonInFile, 5);
135        removeButtonData.left = new FormAttachment(addButtonInFile, 0, SWT.LEFT);
136        removeButtonData.right = new FormAttachment(100, 0);
137        removeButtonInFile.setLayoutData(removeButtonData);
138        
139        return composite;
140      }
141      
142      /**
143       * Fills the file table with files.
144       * @param projectName the project name.
145       */
146      public void fillFileTable(String projectName) {
147        IProject project = FileResource.getProject(projectName);
148        PropertyResource propertyResource = PropertyResource.getInstance(project, true);
149        String defaultReviewId = PropertyConstraints.DEFAULT_REVIEW_ID;
150        ReviewResource reviewResource = propertyResource.getReviewResource(defaultReviewId, true);
151        if (reviewResource != null) {
152          this.files = reviewResource.getFileSet();
153          for (Iterator i = this.files.iterator(); i.hasNext();) {
154            String file = (String) i.next();
155            TableItem item = new TableItem(this.fileListTable, SWT.NONE);
156            item.setText(file);
157          }
158        }
159      }
160      
161      /**
162       * Handles the file list table selection.
163       */
164      protected void handleFileListSelection() {
165        boolean isSelected = (this.fileListTable.getSelectionIndex() >= 0);
166        this.addButtonInFile.setEnabled(isSelected);
167        this.removeButtonInFile.setEnabled(isSelected);
168      }
169      
170      /**
171       * Adds a file to the file list.
172       */
173      protected void addFile() {
174        IWorkbench workbench = PlatformUI.getWorkbench();
175        Shell shell = workbench.getActiveWorkbenchWindow().getShell();
176        FileFolderSelectionDialog dialog = new FileFolderSelectionDialog(shell, true, IResource.FILE);
177        dialog.setTitle(ReviewI18n.getString("ReviewIdEditDialog.label.tab.file.add.title"));
178        dialog.setMessage(ReviewI18n.getString("ReviewIdEditDialog.label.tab.file.add.message"));
179        //dialog.setInput(project.getLocation().toFile());
180        try {
181          IFileStore store = EFS.getStore(project.getLocationURI());
182          dialog.setInput(store);
183        }
184        catch (Exception e) {
185          e.printStackTrace();
186          // Ignore  
187        }
188        
189        if (dialog.open() == FileFolderSelectionDialog.OK) {
190          Object[] results = (Object[]) dialog.getResult();
191          for (int i = 0; i < results.length; i++) {
192            LocalFile file = (LocalFile) results[i];
193            String filePath = file.toString();
194            String projectPath = project.getLocation().toFile().toString();
195            int index = projectPath.length();
196            String projectToFilePath = filePath.substring(index + 1);
197            String targetFile = project.getFile(projectToFilePath).getProjectRelativePath().toString();
198            if (this.files.add(targetFile)) {
199              TableItem item = new TableItem(this.fileListTable, SWT.NONE);
200              item.setText(targetFile); 
201            }
202          }
203        }
204        setPageComplete(isTableNonEmpty());
205      }
206    
207      /**
208       * Removes a file from the file list.
209       */
210      protected void removeFile() {
211        TableItem[] items = this.fileListTable.getSelection();
212        for (int i = 0; i < items.length; i++) {
213          TableItem item = items[i];
214          this.files.remove(item.getText());
215          item.dispose();
216        }
217        this.removeButtonInFile.setEnabled(this.fileListTable.getItemCount() > 0);
218        setPageComplete(isTableNonEmpty());
219      }

222       * Returns the next page. Saves the values from this page in the model associated with the wizard.
223       * Initializes the widgets on the next page.
224       * @return the next page.
225       */
226      public IWizardPage getNextPage() {
227        return ((ReviewIdNewWizard) getWizard()).getPage(ReviewIdNewWizard.PAGE_REVIEWER);
228      }
229      
230      /**
231       * Checks if the wizard can flip to the next page.
232       * @return <code>true</code> if it can flip to the next page.
233       */
234      public boolean canFlipToNextPage() {
235        return isPageComplete();
236      }
237      
238      /**
239       * Checks if the table has at least one review file.
240       * @return <code>true</code> if the table has at least one review file.
241       */
242      private boolean isTableNonEmpty() {
243    //    return (this.files.size() > 0);
244        return true;
245      }
246      
247      /**
248       * Gets the set of the review files.
249       * @return the set of the review files.
250       */
251      public Set getFiles() {
252        return this.files;
253      }
