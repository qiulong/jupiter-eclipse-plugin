003    import org.eclipse.core.resources.IProject;
004    import org.eclipse.jface.preference.IPreferenceStore;
005    import org.eclipse.jface.viewers.IStructuredSelection;
006    import org.eclipse.jface.wizard.Wizard;
007    import org.eclipse.jface.wizard.WizardPage;
008    import org.eclipse.ui.INewWizard;
009    import org.eclipse.ui.IWorkbench;
010    import org.eclipse.ui.dialogs.WizardNewFileCreationPage;
011    
012    import csdl.jupiter.ReviewPlugin;
013    import csdl.jupiter.file.FileResource;
014    import csdl.jupiter.file.PropertyResource;
015    import csdl.jupiter.model.review.ReviewId;
016    import csdl.jupiter.model.review.ReviewModel;
017    import csdl.jupiter.model.review.ReviewerId;
018    import csdl.jupiter.ui.marker.ReviewMarker;
019    import csdl.jupiter.util.JupiterLogger;
020    
021    /**
022     * Provides review id selection wizard.
023     * @author Takuya Yamashita
024     * @version $Id: ReviewIdSelectionWizard.java,v 1.24 2005/03/28 14:39:52 takuyay Exp $
025     */
026    public class ReviewIdSelectionWizard extends Wizard implements INewWizard {
027      /** Jupiter logger */
028      private JupiterLogger log = JupiterLogger.getLogger();
029    
030      private static final String PAGE_REVIEW_ID_SELECTION = "ReviewIdSelection";
031      private WizardNewFileCreationPage configFileCreationPage;
032      /** the reviewer id to be kept for the next default reviewer. */
033      public static final String REVIEWER_ID = "ReviewerId";
034      
035    //  private IStructuredSelection selection;
036      
037      private IWorkbench workbench;
038      private String reviewPhaseNameKey;
039      private boolean isMultipleChoiceEnabled;
040      private IProject defaultProject;
041      
042      /**
043       * Instantiates review id selection wizard. After this wizard is finished,
044       * 1) review ID (determined by defaultProject name and review id) will be set to content
045       *  provider model,
046       * 2) reviewer ID was set to preference store.
047       * 
048       * @param reviewPhaseNameKey the review phase name key.
049       * @param defaultProject the defaultProject to be shown as a default in the dialog. Clients can
050       * not pass defaultProject
051       * as <code>null</code>. so that they should check the null before instantiating this.
052       * @param isMultipleChoiceEnabled set <code>true</code> if the defaultProject choice is multiple.
053       * <code>false</code> if the defaultProject choice is only the passing defaultProject.
054       */
055      public ReviewIdSelectionWizard(String reviewPhaseNameKey, IProject defaultProject,
056                                     boolean isMultipleChoiceEnabled) {
057        super();
058        this.reviewPhaseNameKey = reviewPhaseNameKey;
059        this.defaultProject = defaultProject;
060        this.isMultipleChoiceEnabled = isMultipleChoiceEnabled;
061      }
062    
063      /**
064       * Performs finish. After this method is invoked.
065       * 1) review ID (determined by defaultProject name and review id) will be set to content
066       *  provider model,
067       * 2) reviewer ID was set to preference store.
068       * @see org.eclipse.jface.wizard.Wizard#performFinish()
069       */
070      public boolean performFinish() {
071        // clear all markers in the current project first.
072        ReviewMarker.clearMarkersInReviewId();
073        ReviewIdSelectionPage page = (ReviewIdSelectionPage) getPage(PAGE_REVIEW_ID_SELECTION);
074        String projectName = page.getProjectName();
075        String reviewIdName = page.getReviewIdName();
076        String reviewerIdName = page.getReviewerIdName();
077        IPreferenceStore store = ReviewPlugin.getInstance().getPreferenceStore();
078        // for the next default reviewer in the selection page.
079        store.setValue(ReviewIdSelectionWizard.REVIEWER_ID, reviewerIdName);
080        // new structure 7/23/04
081        IProject project = FileResource.getProject(projectName);
082        ReviewModel.getInstance().notifyListeners(project);
083        PropertyResource reviewIdResource = PropertyResource.getInstance(project, false);
084        ReviewId reviewId = reviewIdResource.getReviewId(reviewIdName);
085        log.debug("directory: " + ((reviewId != null) ? reviewId.getDirectory() : "null"));
086        ReviewModel.getInstance().notifyListeners(reviewId);
087        ReviewerId reviewerId = new ReviewerId(reviewerIdName, "");
088        ReviewModel.getInstance().notifyListeners(reviewerId);
089        return true;
090      }
091      
092      /**
093       * Initializes the wizard.
094       * @see org.eclipse.ui.IWorkbenchWizard#init(org.eclipse.ui.IWorkbench, 
095       * org.eclipse.jface.viewers.IStructuredSelection)
096       */
097      public void init(IWorkbench workbench, IStructuredSelection selection) {
098        this.workbench = workbench;
099    //    this.selection = selection;
100      }
101    
102      /**
103       * Adds pages.
104       * @see org.eclipse.jface.wizard.IWizard#addPages()
105       */
106      public void addPages() {
107        WizardPage reviewIdSelectionPage = new ReviewIdSelectionPage(this.defaultProject,
108                                                                     this.isMultipleChoiceEnabled,
109                                                                     PAGE_REVIEW_ID_SELECTION);
110        addPage(reviewIdSelectionPage);
111      }
112      
113      /**
114       * Checks if the wizard can finish.
115       * @return <code>true</code> if it can finish.
116       * @see org.eclipse.jface.wizard.IWizard#canFinish()
117       */
118      public boolean canFinish() {
119        return true;
120      }
121    }
