440      /**
441       * Fills new values in combos
442       * @param phaseName the phase name.
443       */
444      protected void fillNewValueInCombos(String phaseName) {
445        FilterPhase filterPhase = (FilterPhase) this.phaseNameFilterPhaseMap.get(phaseName);
446            
447        FilterEntry entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_INTERVAL);
448        this.intervalCheckButton.setSelection(entry.isEnabled());
449        this.intervalFilterText.setText(ReviewI18n.getString(entry.getValueKey()));
450        
451        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_REVIEWER);
452        this.reviewerCheckButton.setSelection(entry.isEnabled());
453        this.reviewerFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
454        
455        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_TYPE);
456        this.typeCheckButton.setSelection(entry.isEnabled());
457        this.typeFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
458        
459        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY);
460        this.severityCheckButton.setSelection(entry.isEnabled());
461        this.severityFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
462        
463        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_ASSIGNED_TO);
464        this.assignedToCheckButton.setSelection(entry.isEnabled());
465        this.assignedToFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
466        
467        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION);
468        this.resolutionCheckButton.setSelection(entry.isEnabled());
469        this.resolutionFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
470        
471        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_STATUS);
472        this.statusCheckButton.setSelection(entry.isEnabled());
473        this.statusFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
474        
475        entry = filterPhase.getFilterEntry(PropertyConstraints.ATTRIBUTE_VALUE_FILE);
476        this.fileCheckButton.setSelection(entry.isEnabled());
477        this.fileFilterCombo.setText(ReviewI18n.getString(entry.getValueKey()));
478        
479        this.enabledCheckButton.setSelection(filterPhase.isEnabled());
480        handleEnabledCheck();
481      }

483      /**
484       * Handles the enabled check status.
485       */
486      protected void handleEnabledCheck() {
487        boolean isEnabled = this.enabledCheckButton.getSelection();
488        this.intervalCheckButton.setEnabled(isEnabled);
489        this.reviewerCheckButton.setEnabled(isEnabled);
490        this.typeCheckButton.setEnabled(isEnabled);
491        this.severityCheckButton.setEnabled(isEnabled);
492        this.assignedToCheckButton.setEnabled(isEnabled);
493        this.resolutionCheckButton.setEnabled(isEnabled);
494        this.statusCheckButton.setEnabled(isEnabled);
495        this.fileCheckButton.setEnabled(isEnabled);
496        if (isEnabled) {
497          handleFilterEnabledCheck(intervalCheckButton.getSelection(), intervalFilterText);
498          handleFilterEnabledCheck(reviewerCheckButton.getSelection(), reviewerFilterCombo);
499          handleFilterEnabledCheck(typeCheckButton.getSelection(), typeFilterCombo);
500          handleFilterEnabledCheck(severityCheckButton.getSelection(), severityFilterCombo);
501          handleFilterEnabledCheck(assignedToCheckButton.getSelection(), assignedToFilterCombo);
502          handleFilterEnabledCheck(resolutionCheckButton.getSelection(), resolutionFilterCombo);
503          handleFilterEnabledCheck(statusCheckButton.getSelection(), statusFilterCombo);
504          handleFilterEnabledCheck(fileCheckButton.getSelection(), fileFilterCombo);
505        }
506        else {
507          this.intervalFilterText.setEnabled(isEnabled);
508          this.reviewerFilterCombo.setEnabled(isEnabled);
509          this.typeFilterCombo.setEnabled(isEnabled);
510          this.severityFilterCombo.setEnabled(isEnabled);
511          this.assignedToFilterCombo.setEnabled(isEnabled);
512          this.resolutionFilterCombo.setEnabled(isEnabled);
513          this.statusFilterCombo.setEnabled(isEnabled);
514          this.fileFilterCombo.setEnabled(isEnabled);
515        }
516      }

518      /**
519       * updates the filter enable status.
520       * @param filterName the filter name. null if the filter in the phase is to be set.
521       * @param isEnabled the enabled status of the filter. <code>true</code> if the filter is enabled.
522       */
523      private void updateFilterEnabled(String filterName, boolean isEnabled) {
524        String phaseName = this.phaseCombo.getText();
525        FilterPhase filterPhase = (FilterPhase) this.phaseNameFilterPhaseMap.get(phaseName);
526        if (filterName != null) {
527          FilterEntry entry = filterPhase.getFilterEntry(filterName);
528          entry.setEnabled(isEnabled);
529        }
530        else {
531          filterPhase.setEnabled(isEnabled);
532        }
533      }

536       * updates the filter value.
537       * @param filterName the filter name.
538       * @param key the key of the filter.
539       */
540      private void updateFilterValue(String filterName, String key) {
541        String phaseName = this.phaseCombo.getText();
542        FilterPhase filterPhase = (FilterPhase) this.phaseNameFilterPhaseMap.get(phaseName);
543        FilterEntry entry = filterPhase.getFilterEntry(filterName);
544        entry.setKey(key);
545      }

556      /**
557       * Updates the filter item folder.
558       * @param fieldItemId the field item id.
559       */
560      public void updateFilterItems(String fieldItemId) {
561        IWizardPage page = getWizard().getPage(ReviewIdNewWizard.PAGE_ITEM_ENTRIES);
562        ReviewIdNewItemEntriesPage itemEntryPage = (ReviewIdNewItemEntriesPage) page;
563        Map fieldItemIdFieldItemMap = itemEntryPage.getFieldItemIdFieldItemMap();
564        FieldItem fieldItem = (FieldItem) fieldItemIdFieldItemMap.get(fieldItemId);
565        if (fieldItem != null) {
566          List itemNameList = fieldItem.getEntryNameList();
567          if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_TYPE)) {
568            String currentType = this.typeFilterCombo.getText();
569            this.typeFilterCombo.setItems((String[]) itemNameList.toArray(new String[] {}));
570            this.typeFilterCombo.setText(currentType);
571          }
572          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_SEVERITY)) {
573            String currentSeverity = this.severityFilterCombo.getText();
574            this.severityFilterCombo.setItems((String[]) itemNameList.toArray(new String[] {}));
575            this.severityFilterCombo.setText(currentSeverity);
576          }
577          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_RESOLUTION)) {
578            String currentResolution = this.resolutionFilterCombo.getText();
579            this.severityFilterCombo.setItems((String[]) itemNameList.toArray(new String[] {}));
580            this.severityFilterCombo.setText(currentResolution);
581          }
582          else if (fieldItemId.equals(PropertyConstraints.ATTRIBUTE_VALUE_STATUS)) {
583            String currentStatus = this.statusFilterCombo.getText();
584            this.statusFilterCombo.setItems((String[]) itemNameList.toArray(new String[] {}));
585            this.statusFilterCombo.setText(currentStatus);
586          }
587        }
588      }
