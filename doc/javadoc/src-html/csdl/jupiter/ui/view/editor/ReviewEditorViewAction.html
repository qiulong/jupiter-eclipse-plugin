215      /**
216       * Processes saving an issue.
217       * 
218       * @return <code>true</code> if successfully saved. <code>false</code> otherwise.
219       */
220      private static boolean processSave() {
221        ReviewEditorView editorView = ReviewEditorView.getActiveView();
222        if (editorView == null) {
223          return false;
224        }
225        ReviewIssue savingReviewIssue = editorView.getReviewIssue();
226        String activeTabNameKey = editorView.getActiveTabNameKey();
227        if (savingReviewIssue == null) {
228          return false;
229        }
230        ReviewIssueModelManager reviewIssueModelManager = ReviewIssueModelManager.getInstance();
231        ReviewIssueModel reviewIssueModel = reviewIssueModelManager.getCurrentModel();
232    
233        // make sure the editingCodeReview is contained.
234        if (reviewIssueModel.contains(savingReviewIssue.getIssueId())) {
235          ReviewIssue originalReviewIssue = reviewIssueModel.get(savingReviewIssue.getIssueId());
236          if (!originalReviewIssue.contentEquals(savingReviewIssue)) {
237            try {
238              originalReviewIssue.setReviewIssue(savingReviewIssue);
239              reviewIssueModel.notifyListeners(ReviewIssueModelEvent.EDIT);
240            }
241            catch (ReviewException e) {
242              e.printStackTrace();
243            }
244          }
245        }
246        else {
247          reviewIssueModel.add(savingReviewIssue);
248          ReviewTableView view = ReviewTableView.getActiveView();
249          if (view == null) {
250            view = ReviewTableView.bringViewToTop();
251          }
252          view.getViewer().refresh();
253          Table table = view.getTable();
254          table.select(0);
255          updateNextPreviousIcon();
256          reviewIssueModel.notifyListeners(ReviewIssueModelEvent.ADD);
257        }
258        return true;
259      }

261      /**
262       * Checks if the <code>ReviewIssue</code> instance is valid. Throws
263       * <code>CodeReviewException</code> with error message if it is not valid.
264       * 
265       * @param codeReview the <code>ReviewIssue</code> instance.
266       * @param activeTabNameKey the active tab name key.
267       * @throws ReviewException if the instance is not valid.
268       */
269      private static void validateCodeReview(ReviewIssue codeReview, String activeTabNameKey)
270          throws ReviewException {
271        String message = "";
272        if (activeTabNameKey.equals(ResourceBundleKey.PHASE_INFIVIDUAL)) {
273          if (codeReview.getType().getKey().equals(ResourceBundleKey.ITEM_KEY_UNSET)) {
274            message += "\n" + ReviewI18n.getString(ResourceBundleKey.COLUMN_HEADER_TYPE);
275          }
276          if (codeReview.getSeverity().getKey().equals(ResourceBundleKey.ITEM_KEY_UNSET)) {
277            message += "\n" + ReviewI18n.getString(ResourceBundleKey.COLUMN_HEADER_SEVERITY);
278          }
279          if (!message.equals("")) {
280            throw new ReviewException(message);
281          }
282        }
283        else if (activeTabNameKey.equals(ResourceBundleKey.PHASE_TEAM)) {
284          if (codeReview.getAssignedTo().equals("")) {
285            message += "\n" + ReviewI18n.getString(ResourceBundleKey.COLUMN_HEADER_ASSGINED_TO);
286          }
287          if (codeReview.getResolution().getKey().equals(ResourceBundleKey.ITEM_KEY_UNSET)) {
288            message += "\n" + ReviewI18n.getString(ResourceBundleKey.COLUMN_HEADER_RESOLUTION);
289          }
290          if (!message.equals("")) {
291            throw new ReviewException(message);
292          }
293        }
294      }

296      /**
297       * Updates the next and previous icon.
298       */
299      public static void updateNextPreviousIcon() {
300        ReviewTableView tableView = ReviewTableView.getActiveView();
301        if (tableView != null) {
302          Table table = tableView.getTable();
303          int index = table.getSelectionIndex();
304          if (index != -1) {
305            int length = table.getItemCount();
306            NEXT.setEnabled(index < length - 1);
307            PREVIOUS.setEnabled(index > 0);
308          }
309        }
310      }
