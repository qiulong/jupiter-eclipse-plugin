001    package csdl.jupiter.ui.view.table;
002    
003    import org.eclipse.swt.widgets.Table;
004    
005    /**
006     * Provides the table index manager.
007     * @author Takuya Yamashita
008     * @version $Id: TableIndexManager.java,v 1.4 2004/08/27 04:56:02 takuyay Exp $
009     */
010    public class TableIndexManager {
011      private static TableIndexManager theInstance = new TableIndexManager();
012      private int index = 0;
013      private int size = 0;
014      /**
015       * Prohibits clients from instantiating this.
016       */
017      private TableIndexManager() {
018      }
019      
020      /**
021       * Gets the sigleton instance.
022       * @return the sigleton instance.
023       */
024      public static TableIndexManager getInstance() {
025        theInstance.computeCurrentIndexAndSize();
026        return theInstance;
027      }
028      
029      /**
030       * Computes the current index.
031       */
032      private void computeCurrentIndexAndSize() {
033        Table table = ReviewTableView.getInstance().getTable();
034        int selectionIndex = table.getSelectionIndex();
035        int newSize = table.getItemCount();
036        // if this size != new size (the filter case), then decrements index by one.
037        this.index = (selectionIndex >= 0) ?  selectionIndex 
038                                           : (newSize == this.size) ? this.index 
039                                                                     : (this.index > 0) ? --this.index
040                                                                                        : 0; 
041        this.size = newSize;
042    
043      }
044      
045      /**
046       * Gets the current index.
047       * @return the current index.
048       */
049      public int getIndex() {
050        return this.index;
051      }
052      
053      /**
054       * Increments the current index and returns the incremented index.
055       * @return the incremented index.
056       */
057      public int increment() {
058        if (this.index < size - 1) {
059          this.index++;
060        }
061        return this.index;
062      }
063      
064      /**
065       * Decrements the current index and returns the decremented index.
066       * @return the decremented index.
067       */
068      public int decrement() {
069        if (this.index > 0) {
070          this.index--;
071        }
072        return this.index;
073      }
074      
075      /**
076       * Gets the size of the table elements.
077       * @return the size of the table elements.
078       */
079      public int getSize() {
080        return this.size;
081      }
082    }
