275        GOTO_REVISION_SOURCE = new Action() {
276          public void run() {
277            ReviewTableView view = ReviewTableView.getActiveView();
278            ISelection selection = view.getViewer().getSelection();
279            Object object = ((IStructuredSelection) selection).getFirstElement();
280            if (object == null) {
281              return;
282            }
283            ReviewIssue reviewIssue = (ReviewIssue) object;
284            String targetFile = reviewIssue.getTargetFile();
285            if (targetFile.equals("")) {
286              return;
287            }
288            IProject project = FileResource.getProject(reviewIssue.getReviewIFile());
289            IFile targetIFile = project.getFile(targetFile);
290            //          String lineNumberString = reviewIssue.getLine();
291            //          boolean isNumber = ((lineNumberString != null) && !lineNumberString.equals(""));
292            //          int lineNumber = Integer.parseInt((isNumber) ? lineNumberString : "0");
293            //          FileResource.goToLine(targetIFile, lineNumber);
294    
295            final ICVSResource cvsResource = CVSWorkspaceRoot.getCVSResourceFor(targetIFile);
296            if (cvsResource == null) {
297              return;
298            }
299            String revision = getRevision(cvsResource);
300            if (revision == null) {
301              return;
302            }
303            try {
304              new RevisionOperation(cvsResource, revision).run();
305            }
306            catch (InvocationTargetException e) {
307              e.printStackTrace();
308            }
309            catch (InterruptedException e) {
310              e.printStackTrace();
311            }
312          }
313          /**
314           * Get the revision for the CVS resource. Returns null if the revision
315           * could not be determined.
316           * 
317           * @param cvsResource The CVS resource
318           * @return The revision of the resource.
319           */
320          private String getRevision(ICVSResource cvsResource) {
321            final ResourceSyncInfo info;
322            try {
323              info = cvsResource.getSyncInfo();
324              if (info == null) {
325                return null;
326              }
327            }
328            catch (CVSException e) {
329              return null;
330            }
331            return info.getRevision();
332          }
333        };

366        NOTIFY_EDITOR = new Action() {
367          private IProject previousProject = null;
368          private ReviewId previousReviewId = null;
369          public void run() {
370            ReviewTableView view = ReviewTableView.getActiveView();
371            if (view == null) {
372              log.warning("Review table view is null.");
373              return;
374            }
375            TableViewer viewer = view.getViewer();
376            ISelection selection = viewer.getSelection();
377            Object object = ((IStructuredSelection) selection).getFirstElement();        
378            ReviewEditorView[] editorViews = ReviewEditorView.getViews();
379            ReviewModel reviewModel = ReviewModel.getInstance();
380            IProject project = reviewModel.getProjectManager().getProject();
381            ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
382            for (int i = 0; i < editorViews.length; i++) {
383              ReviewEditorView editorView = editorViews[i];
384              if (editorView != null && !editorView.isDisposed()) {
385                if (object != null && project != null && reviewId != null) {
386                  ReviewIssue notifyingReviewIssue = (ReviewIssue) object;
387                  if (previousProject != project || previousReviewId != reviewId) {
388                    editorView.setItemFields(project, reviewId);
389                    previousProject = project;
390                    previousReviewId = reviewId;
391                  }
392                  editorView.setReviewIssue(notifyingReviewIssue);
393                  editorView.setEnable(true);
394                }
395                else {
396                  editorView.clearAllFields();
397                  editorView.setEnable(false);
398                }
399              }
400            }
401            ReviewEditorViewAction.updateNextPreviousIcon();
402          }
403        };

454        DELETE = new Action() {
455          public void run() {
456            ReviewTableView view = ReviewTableView.getActiveView();
457            ISelection selection = view.getViewer().getSelection();
458            IStructuredSelection structureSelection = (IStructuredSelection) selection;
459            Table table = view.getTable();
460            int deletingIndex = table.getSelectionIndex();
461            log.debug("structure selection size: " + structureSelection.size());
462            if ((structureSelection.size() > 0)
463                && MessageDialog.openConfirm(view.getSite().getShell(), ReviewI18n
464                    .getString("ReviewTableView.messageDialog.confirm.title"), ReviewI18n
465                    .getString("ReviewTableView.messageDialog.confirm.message"))) {
466              ReviewIssueModel model = ReviewIssueModelManager.getInstance().getCurrentModel();
467              for (Iterator i = ((IStructuredSelection) selection).iterator(); i.hasNext();) {
468                model.remove((ReviewIssue) i.next());
469                model.notifyListeners(ReviewIssueModelEvent.DELETE);
470              }
471              view.getViewer().refresh();
472              table = view.getTable();
473              int length = table.getItemCount();
474              if (length > 0) {
475                int index = (deletingIndex < length) ? deletingIndex : deletingIndex - 1;
476                table.select(index);
477                ReviewTableViewAction.NOTIFY_EDITOR.run();
478              }
479              else {
480                ReviewEditorView.setViewEnable(false);
481              }
482              int type = ReviewEvent.TYPE_COMMAND;
483              int kind = ReviewEvent.KIND_DELETE;
484              ReviewPlugin.getInstance().notifyListeners(type, kind);
485            }
486            else {
487              openOneTableItemSelectionDialog();
488            }
489          }
490        };
