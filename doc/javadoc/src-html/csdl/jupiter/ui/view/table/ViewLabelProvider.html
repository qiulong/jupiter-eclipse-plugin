032    /**
033     * Provides the label for the code review view.
034     *
035     * @author Takuya Yamashita
036     * @version $Id: ViewLabelProvider.java,v 1.10 2005/03/28 14:39:30 takuyay Exp $
037     */
038    class ViewLabelProvider extends LabelProvider implements ITableLabelProvider {
039      /** Jupiter logger */
040      private JupiterLogger log = JupiterLogger.getLogger();
041    
042      /** The date format for the creation date. */
043      private static final DateFormat DATE_FORMATTER = new SimpleDateFormat("yyyy-MM-dd");
044      
045      private Image linkImage;
046    
047      /**
048       * Returns the label text for the given column of the given element.
049       *
050       * @param object The object instance to cast the ReviewIssue instance type defined in
051       *        <code>ReviewIssueModel</code> instance.
052       * @param columnIndex the zero-based index of the column in which the label appears.
053       *
054       * @return The column text string. No null value.
055       */
056      public String getColumnText(Object object, int columnIndex) {
057        ReviewIssue reviewIssue = (ReviewIssue) object;
058        
059        ReviewModel reviewModel = ReviewModel.getInstance();
060        IProject project = reviewModel.getProjectManager().getProject();
061        ReviewId reviewId = reviewModel.getReviewIdManager().getReviewId();
062        String phaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
063        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
064        ColumnDataModel columnDataModel = columnDataModelManager.getModel(phaseNameKey);
065        
066        String[] columnNameKeys = columnDataModel.getEnabledColumnNameKeyArray();
067        // make sure that columnIndex is not the index out of bounds.
068        if (columnNameKeys.length > columnIndex) {
069          String columnNameKey = columnNameKeys[columnIndex];
070          if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_CREATED_DATE)) {
071            return DATE_FORMATTER.format(reviewIssue.getCreationDate());
072          }
073          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_MODIFIED_DATE)) {
074            return DATE_FORMATTER.format(reviewIssue.getModificationDate());
075          }
076          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_REVIEWER)) {
077            return reviewIssue.getReviewer();
078          }
079          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_ASSGINED_TO)) {
080            return reviewIssue.getAssignedTo();
081          }
082          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_SUMMARY)) {
083            return reviewIssue.getSummary();
084          }
085          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_DESCRIPTION)) {
086            return reviewIssue.getDescription();
087          }
088          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_ANNOTATION)) {
089            return reviewIssue.getAnnotation();
090          }
091          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_REVISION)) {
092            return reviewIssue.getRevision();
093          }
094          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_TYPE)) {
095            TypeKeyManager manager = TypeKeyManager.getInstance(project, reviewId);
096            return manager.getLocalizedLabel(reviewIssue.getType().getKey());
097          }
098          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_SEVERITY)) {
099            String severityKey = reviewIssue.getSeverity().getKey();
100            SeverityKeyManager manager = SeverityKeyManager.getInstance(project, reviewId);
101            return manager.getLocalizedLabel(severityKey);
102          }
103          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_RESOLUTION)) {
104            String resolutionKey = reviewIssue.getResolution().getKey();
105            ResolutionKeyManager manager = ResolutionKeyManager.getInstance(project, reviewId);
106            return manager.getLocalizedLabel(resolutionKey);
107          }
108          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_STATUS)) {
109            StatusKeyManager manager = StatusKeyManager.getInstance(project, reviewId);
110            return manager.getLocalizedLabel(reviewIssue.getStatus().getKey());
111          }
112          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_FILE)) {
113            return reviewIssue.getTargetFile();
114          }
115          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_LINE)) {
116            return reviewIssue.getLine();
117          }
118          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_ID)) {
119            return reviewIssue.getIssueId();
120          }
121          else if (columnNameKey.equals(ResourceBundleKey.COLUMN_HEADER_LINK_ICON)) {
122            return "";
123          }
124        }
125        return "";
126      }
127    
128      /**
129       * Returns the label image for the given column of the given element.
130       *
131       * @param object the object representing the entire row, or null indicating that no input object
132       *        is set in the viewer.
133       * @param columnIndex the zero-based index of the column in which the label appears.
134       *
135       * @return The label image for the given column of the given element.
136       */
137      public Image getColumnImage(Object object, int columnIndex) {
138        ReviewModel reviewModel = ReviewModel.getInstance();    
139        String phaseNameKey = reviewModel.getPhaseManager().getPhaseNameKey();
140        ColumnDataModelManager columnDataModelManager = ColumnDataModelManager.getInstance();
141        ColumnDataModel columnDataModel = columnDataModelManager.getModel(phaseNameKey);
142        String[] columnNameKeys = columnDataModel.getEnabledColumnNameKeyArray();
143        if (columnNameKeys.length > columnIndex) {
144          if (columnNameKeys[columnIndex].equals(ResourceBundleKey.COLUMN_HEADER_LINK_ICON)) {
145            return getImage(object);
146          }
147        }
148        return null;
149      }
150    
151      /**
152       * Gets the image from <code>object</code>. Return the linked icon image if <code>IFile</code>
153       * path to the target review file exist. Otherwise return null.
154       *
155       * @param object the element for which to provide the label image
156       *
157       * @return The image used to label the element, or null if there is no image for the given
158       *         object.
159       */
160      public Image getImage(Object object) {
161        ReviewIssue codeReview = (ReviewIssue) object;
162        IProject project = FileResource.getProject(codeReview.getReviewIFile());
163        IJavaProject javaProject = JavaCore.create(project);
164        String targetFileString = codeReview.getTargetFile();
165        IFile targetIFile = null;
166        if (!targetFileString.equals("")) {
167          targetIFile = project.getFile(targetFileString);
168        }
169        try {
170          URL url = ReviewPlugin.getInstance().getInstallURL();
171          url = new URL(url, "lib/icons/link.gif");
172          IJavaElement[] javaElements = new IJavaElement[] {javaProject};
173          boolean isTargetIFileAvailable = (targetIFile != null) ? targetIFile.exists() : false;
174          if (linkImage == null) {
175            linkImage = ImageDescriptor.createFromURL(url).createImage();
176          }
177          // Sets link status for the sort by link image.
178          codeReview.setLinkStatus(isTargetIFileAvailable);
179          return (isTargetIFileAvailable ? linkImage : null);
180        }
181        catch (MalformedURLException e) {
182          log.debug(e.getMessage());
183          return null;
184        }
185      }
