001    package csdl.jupiter.ui.view.table;
002    
003    import java.util.Iterator;
004    import java.util.List;
005    import java.util.Map;
006    import java.util.TreeMap;
007    
008    /**
009     * Provides the phase based filter entry data structure.
010     * @author Takuya Yamashita
011     * @version $Id: FilterPhase.java,v 1.3 2004/09/13 12:09:04 takuyay Exp $
012     */
013    public class FilterPhase {
014      private String phaseName;
015      private boolean isEnabled;
016      private Map filterNameFilterEntryMap;
017      private List filterEntryList;
018      
019      /**
020       * Instantiates the phase based filter entry data structure.
021       * @param phaseName the phase name.
022       * @param isEnabled <code>true</code> if the phase based filter is enabled.
023       * @param filterEntryList the list of the <code>FilterEntry</code>.
024       */
025      public FilterPhase(String phaseName, boolean isEnabled, List filterEntryList) {
026        this.phaseName = phaseName;
027        this.isEnabled = isEnabled;
028        this.filterEntryList = filterEntryList;
029        this.filterNameFilterEntryMap = new TreeMap();
030        for (Iterator i = filterEntryList.iterator(); i.hasNext();) {
031          FilterEntry entry = (FilterEntry) i.next();
032          this.filterNameFilterEntryMap.put(entry.getFilterName(), entry);
033        }
034      }
035      
036      /**
037       * Gets the FilterEntry instance. Returns null if not found.
038       * @param filterName the filter name.
039       * @return the FilterEntry instance. Returns null if not found.
040       */
041      public FilterEntry getFilterEntry(String filterName) {
042        return (FilterEntry) this.filterNameFilterEntryMap.get(filterName);
043      }
044      
045      /**
046       * Checks if the phase based filter is enabled. Returns <code>true</code> if the filter is 
047       * enabled.
048       * @return <code>true</code> if the filter is enabled.
049       */
050      public boolean isEnabled() {
051        return isEnabled;
052      }
053      
054      /**
055       * Sets the enable status of the filter in the phase.
056       * @param isEnabled the enable status of the filter in the phase. <code>true</code> if
057       * the filter in the phase is enabled.
058       */
059      public void setEnabled(boolean isEnabled) {
060        this.isEnabled = isEnabled;
061      }
062      
063      /**
064       * Gets the phase name.
065       * @return the phase name.
066       */
067      public String getPhaseName() {
068        return this.phaseName;
069      }
070      
071      /**
072       * Iterates over <code>FilterEntry</code>.
073       * @return the iterator of the <code>FilterEntry</code>.
074       */
075      public Iterator iterator() {
076        return this.filterEntryList.iterator();
077      }
078    }
