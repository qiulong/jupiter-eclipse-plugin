001    package csdl.jupiter.event;
002    
003    import java.io.File;
004    
005    import org.eclipse.core.resources.IFile;
006    import org.eclipse.core.resources.IProject;
007    import org.eclipse.core.runtime.IPath;
008    
009    import csdl.jupiter.file.FileResource;
010    import csdl.jupiter.model.review.ReviewModel;
011    import csdl.jupiter.util.JupiterLogger;
012    
013    
014    /**
015     * Provides file event manager. Clients can get the latest event associated file.
016     * @author Takuya Yamashita
017     * @version $Id: EventFileManager.java,v 1.4 2005/03/28 14:39:31 takuyay Exp $
018     */
019    public class EventFileManager {
020      private JupiterLogger log = JupiterLogger.getLogger();
021      private static EventFileManager theInstance = new EventFileManager();
022      private String filePath = null;
023      /**
024       * Prohibits clients from instantiating this.
025       */
026      private EventFileManager() {
027        this.filePath = getSelectedFile();
028      }
029      
030      /**
031       * Gets the selected file path. Returns empty string if there are not an applicable path.
032       * @return The selected file path.
033       */
034      private String getSelectedFile() {
035        String file = "";
036        IFile selectedFile = FileResource.getSelectedIFile();
037        if (selectedFile != null) {
038          IPath selectedPath = selectedFile.getLocation();
039          if (selectedPath != null) {
040            file = selectedPath.toFile().toString();
041          }
042        }
043        return file;
044      }
045      
046      /**
047       * Gets the singleton instance.
048       * @return the singleton instance.
049       */
050      public static EventFileManager getInstance() {
051        return theInstance;
052      }
053      
054      /**
055       * Sets the event associated file path.
056       * @param filePath the fully qualified file path.
057       */
058      public void setEventFilePath(String filePath) {
059        ReviewModel reviewModel = ReviewModel.getInstance();
060        IProject project = reviewModel.getProjectManager().getProject();
061        if (filePath != null && project != null) {
062          filePath = (!filePath.equals("")) ? filePath : File.separator;
063          this.filePath = new File(project.getLocation().toString(), filePath).toString();
064        }
065        else {
066          this.filePath = getSelectedFile();
067        }
068      }
069      
070      /**
071       * Gets the event associated file path. Returns empty string if there is no associated file.
072       * @return the fully qualified event associated file path. 
073       * Returns empty string if there is no associated file.
074       */
075      public String getEventFilePath() {
076        log.debug("Event file path: " + this.filePath);
077        return this.filePath;
078      }
079    }
