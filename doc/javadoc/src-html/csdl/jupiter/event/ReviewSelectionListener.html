001    package csdl.jupiter.event;
002    
003    import org.eclipse.core.resources.IFile;
004    import org.eclipse.core.resources.IResource;
005    import org.eclipse.core.runtime.IAdaptable;
006    import org.eclipse.jdt.core.IJavaElement;
007    import org.eclipse.jdt.core.IMember;
008    import org.eclipse.jface.viewers.ISelection;
009    import org.eclipse.jface.viewers.IStructuredSelection;
010    import org.eclipse.ui.IEditorPart;
011    import org.eclipse.ui.IFileEditorInput;
012    import org.eclipse.ui.ISelectionListener;
013    import org.eclipse.ui.IWorkbenchPart;
014    
015    import csdl.jupiter.file.FileResource;
016    
017    /**
018     * Provides the listener to listen to the selection change. 
019     *
020     * @author Takuya Yamashita
021     * @version $Id: ReviewSelectionListener.java,v 1.8 2005/03/25 18:32:02 takuyay Exp $
022     */
023    public class ReviewSelectionListener implements ISelectionListener {
024      /** The selection. */
025      private static IStructuredSelection structuredSelection;
026      
027      /**
028       * Gets structured selection.
029       * @return the structured selection.
030       */
031      public static IStructuredSelection getStructuredSelection() {
032        return structuredSelection;
033      }  
034      
035      /**
036       * Notified when selection is changed. Brings the opened code review editor to top if a code
037       * review file is selected and the editor associated with the file is opened.
038       *
039       * @param workbenchPart the workbench part containing the selection
040       * @param selection the current selection
041       */
042      public void selectionChanged(IWorkbenchPart workbenchPart, ISelection selection) {
043        IEditorPart editorPart = workbenchPart.getSite().getPage().getActiveEditor();
044        if (editorPart != null && editorPart.getEditorInput() instanceof IFileEditorInput) {
045          IFile activeFile = ((IFileEditorInput) editorPart.getEditorInput()).getFile();
046          FileResource.setSelectedResource(activeFile);
047        }
048        if (selection instanceof IStructuredSelection) {
049          structuredSelection = (IStructuredSelection) selection;
050          if (structuredSelection.size() == 1) {
051            Object object = structuredSelection.getFirstElement();
052            if (object instanceof IAdaptable) {
053              IAdaptable adaptable = (IAdaptable) object;
054              IResource resource = (IResource) adaptable.getAdapter(IResource.class);
055              FileResource.setSelectedResource(resource);
056              IJavaElement javaElement = (IJavaElement) adaptable.getAdapter(IJavaElement.class);
057              if (javaElement instanceof IMember) {
058                IMember member = (IMember) javaElement;
059                IFile selectedFile = (IFile) member.getResource();
060                FileResource.setSelectedResource(selectedFile);
061              }
062            }
063          }
064        }
065      }
066    }
