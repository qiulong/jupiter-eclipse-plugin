024    /**
025     * Provides the utility of the code review dialogs.
026     * @author Takuya Yamashita
027     * @version $Id: ReviewDialog.java,v 1.11 2005/05/06 07:41:09 takuyay Exp $
028     */
029    public class ReviewDialog {
030      private static IWorkbench workbench = PlatformUI.getWorkbench();
031      /**
032       * Processes non review ID notification dialog.
033       * @param projectName the project name to be shown.
034       * @return the return code of the message dialog.
035       */
036      public static int processNonReviewIdNotificationDialog(String projectName) {
037        String jupiterIconKey = "ReviewDialog.jupiterIcon";
038        final String jupiterIcon = ReviewI18n.getString(jupiterIconKey);
039        String messagePrefix = "ReviewDialog.nonReviewIdNotification.messageDialog.message.prefix";
040        String messageSuffix = "ReviewDialog.nonReviewIdNotification.messageDialog.message.suffix";
041        String create = "ReviewDialog.nonReviewIdNotification.messageDialog.button.create";
042        String cancel = "ReviewDialog.nonReviewIdNotification.messageDialog.button.cancel";
043        MessageDialog dialog = new MessageDialog(workbench.getWorkbenchWindows()[0].getShell(),
044            ReviewI18n.getString("ReviewDialog.nonReviewIdNotification.messageDialog.title"),
045            ReviewPlugin.createImageDescriptor(jupiterIcon).createImage(),
046            ReviewI18n.getString(messagePrefix)
047            + projectName
048            + ReviewI18n.getString(messageSuffix),
049            MessageDialog.INFORMATION,
050            new String[] {ReviewI18n.getString(create), ReviewI18n.getString(cancel)},
051            0);
052        return dialog.open();
053      }
054    
055      /**
056       * Processes configuration wizard dialog.
057       * @param project the project.
058       * @return the return code of the dialog.
059       */
060      public static int processConfigWizardDialog(IProject project) {
061        INewWizard wizard = new ReviewIdNewWizard(project);
062        wizard.init(workbench, null);
063        // Instantiates the wizard container with the wizard and opens it
064        IWorkbenchWindow window = workbench.getActiveWorkbenchWindow();
065        WizardDialog dialog = new WizardDialog(window.getShell(), wizard);
066        dialog.create();
067        return dialog.open();
068      }

070      /**
071       * Processes version check dialog.
072       * @param localVersionId the local version ID.
073       * @param serverVersionId the server version ID.
074       * @return the result code of the dialog.
075       */
076      public static int processVersionCheckDialog(String localVersionId, String serverVersionId) {
077        String jupiterIconKey = "ReviewDialog.jupiterIcon";
078        final String jupiterIcon = ReviewI18n.getString(jupiterIconKey);
079        MessageDialog dialog = new MessageDialog(workbench.getWorkbenchWindows()[0].getShell(),
080            ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.title"),
081            ReviewPlugin.createImageDescriptor(jupiterIcon).createImage(),
082            ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.message.first")
083            + localVersionId
084            + ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.message.between")
085            + serverVersionId
086            + ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.message.last"),
087            MessageDialog.QUESTION,
088            new String[] {
089              ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.button.update"),
090              ReviewI18n.getString("ReviewDialog.versionCheck.messageDialog.button.cancel")
091              },
092            0);
093        return dialog.open();
094      }

096      /**
097       * Processes non project selection notification dialog.
098       * @return the result code of the dialog.
099       */
100      public static int processNonProjectSelectionDialog() {
101        String messageKey = "ReviewDialog.nonProjectSelection.messageDialog.info.message";
102        boolean result = MessageDialog.openConfirm(workbench.getWorkbenchWindows()[0].getShell(),
103            ReviewI18n.getString("ReviewDialog.nonProjectSelection.messageDialog.info.title"),
104            ReviewI18n.getString(messageKey));

106      }

108      /**
109       * Processes unspecified field notification dialog.
110       * @param message the message to be shown in the dialog.
111       * @return the result code of the dialog.
112       */
113      public static int processUnspecifiedFieldNotificationDialog(String message) {
114        String titleKey = "ReviewDialog.unspecifiedFieldNotification.messageDialog.info.title";
115        String messageKey = "ReviewDialog.unspecifiedFieldNotification.messageDialog.info.message";
116        boolean result = MessageDialog.openConfirm(workbench.getWorkbenchWindows()[0].getShell(),
117            ReviewI18n.getString(titleKey), 
118            ReviewI18n.getString(messageKey)
119            + "\n" + message);
120        return (result) ? WizardDialog.OK : WizardDialog.CANCEL;
121      }
122    
123      /**
124       * Processes review id selection wizard. The default project would be based upon the selected 
125       * project, which can be given <code>FileResource.getSelectedProject()</code>.
126       * After this wizard is finished, 1) review ID instance (determined by project name and review id)
127       * will be set to content provider model, 2) reviewer ID string was set to preference store.
128       * <p>
129       * Clients should not pass <code>IProject</code> to be null. If so,
130       * <code>IllegalArgumentException</code> will be thrown.
131       * 
132       * @param reviewPhaseNameKey the review phase name key.
133       * @param project the project to be shown as a default in the dialog.
134       * @param isMultipleChoiceEnabled set <code>true</code> if the project choice is mulitpile.
135       * <code>false</code> if the project choice is only the passing project.
136       * @return the result code of the dialog. e.g WizardDialog.CANEL. 
137       * See <code>WizardDialog</code> constants.
138       */
139      public static int processReviewIdSelectionWizardDialog(String reviewPhaseNameKey, 
140                                                             IProject project, 
141                                                             boolean isMultipleChoiceEnabled) {
142        if (project == null) {
143          throw new IllegalArgumentException("project should not be null.");
144        }
145        // process review id creation wizard if there is no review ID in the project.
146        List reviewIdList = PropertyResource.getInstance(project, false).getReviewIdList();
147        if (reviewIdList.size() <= 0) {
148          int result = processNonReviewIdNotificationDialog(project.getName());
149          if (result == WizardDialog.CANCEL) {
150            return WizardDialog.CANCEL;
151          }
152          result = processConfigWizardDialog(project);
153          if (result == WizardDialog.CANCEL) {
154            return WizardDialog.CANCEL;
155          }
156        }
157        
158        INewWizard wizard = new ReviewIdSelectionWizard(reviewPhaseNameKey, project, 
159                                                        isMultipleChoiceEnabled);
160        IWorkbench workbench = PlatformUI.getWorkbench();
161        wizard.init(workbench, null);
162        // Instantiates the wizard container with the wizard and opens it
163        IWorkbenchWindow window = workbench.getActiveWorkbenchWindow();
164        WizardDialog dialog = new WizardDialog(window.getShell(), wizard);
165        dialog.create();
166        return dialog.open();
167      }

177      }

179      /**
180       * Opens new update wizard and process update.
181       */
182      public static void proccessOpenNewUpdatesWizard() { 
183        IWorkbench workbench = ReviewPlugin.getInstance().getWorkbench();
184        IWorkbenchWindow activeWindow = workbench.getActiveWorkbenchWindow();
185        if (activeWindow != null) {
186          Shell shell = workbench.getActiveWorkbenchWindow().getShell();
187          IWizard wizard = new InstallWizard(null);
188          WizardDialog dialog = new ResizableInstallWizardDialog(shell, wizard, "");
189          dialog.create();
190          //dialog.getShell().setText(UpdateUI.getString("InstallWizardAction.title")); //$NON-NLS-1$
191          dialog.getShell().setText(UpdateUI.getPluginId());
192          dialog.getShell().setSize(600, 500);
193          dialog.open();
194        }
195      }
