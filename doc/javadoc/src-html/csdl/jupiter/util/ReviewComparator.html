023      public static final Comparator REVIEWER = new Comparator() {
024          /**
025           * Compares ReviewIssue instance by reviewer's name
026           *
027           * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
028           */
029          public int compare(Object object1, Object object2) {
030            String reviewer1 = ((ReviewIssue) object1).getReviewer();
031            String reviewer2 = ((ReviewIssue) object2).getReviewer();
032            return reviewer1.compareTo(reviewer2);
033          }
034        };

178      /** The Comparator implementing class, which provides sort by line number. */
179      public static final Comparator LINE = new Comparator() {
180          /**
181           * Compares ReviewIssue instance by line number.
182           *
183           * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
184           */
185          public int compare(Object object1, Object object2) {
186            int object1IntValue = Integer.MAX_VALUE;
187            try {
188              object1IntValue = Integer.parseInt(((ReviewIssue) object1).getLine());
189            }
190            catch (NumberFormatException e) {
191              // use int max value;
192            }
193            int object2IntValue = Integer.MAX_VALUE;
194            try {
195              object2IntValue = Integer.parseInt(((ReviewIssue) object2).getLine());
196            }
197            catch (NumberFormatException e) {
198              // use int max value;
199            }
200            return object1IntValue - object2IntValue;
201          }
202        };

216      public static final Comparator LINK_ICON = new Comparator() {
217        /**
218         * Compares ReviewIssue instance by link icon. The small number lists first.
219         *
220         * @see java.util.Comparator#compare(java.lang.Object, java.lang.Object)
221         */
222        public int compare(Object object1, Object object2) {
223          String object1Ordinal = (((ReviewIssue) object1).isLinked() ? "0" : "1");
224          String orjbec2Ordinal = (((ReviewIssue) object2).isLinked() ? "0" : "1");
225          return object1Ordinal.compareTo(orjbec2Ordinal);
226        }

250      /**
251       * Gets the <code>Comparator</code> instance from the specified string. Returns <code>null</code>
252       * if the corresponding <code>Comparator</code> instance does not exist.
253       *
254       * @param comparatorNameKey the comparator name key to retrieve the corresponding
255       *        <code>Comparator</code>
256       *
257       * @return the <code>Comparator</code> instance. Returns <code>null</code> if the corresponding
258       *         <code>Comparator</code> instance does not exist.
259       */
260      public static Comparator getComparator(String comparatorNameKey) {
261        return (Comparator) comparators.get(comparatorNameKey);
262      }
