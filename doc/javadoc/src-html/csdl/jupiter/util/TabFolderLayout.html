011    package csdl.jupiter.util;
012    
013    import org.eclipse.swt.SWT;
014    import org.eclipse.swt.graphics.Point;
015    import org.eclipse.swt.graphics.Rectangle;
016    import org.eclipse.swt.widgets.Composite;
017    import org.eclipse.swt.widgets.Control;
018    import org.eclipse.swt.widgets.Layout;
019    
020    /**
021     * This layout controls the position and size of the children of a tab folder.
022     * Note that I (Takuya Yamashita) changed only the checkstyle errors.
023     * 
024     * @since 2.1
025     */
026    public class TabFolderLayout extends Layout {
027    
028      /**
029       * @see org.eclipse.swt.widgets.Layout
030       * #computeSize(org.eclipse.swt.widgets.Composite, int, int, boolean)
031       */
032      protected Point computeSize (Composite composite, int wHint, int hHint, boolean flushCache) {
033        if (wHint != SWT.DEFAULT && hHint != SWT.DEFAULT) {
034          return new Point(wHint, hHint);
035        }
036          
037        Control [] children = composite.getChildren ();
038        int count = children.length;
039        int maxWidth = 0, maxHeight = 0;
040        for (int i = 0; i < count; i++) {
041          Control child = children [i];
042          Point pt = child.computeSize (SWT.DEFAULT, SWT.DEFAULT, flushCache);
043          maxWidth = Math.max (maxWidth, pt.x);
044          maxHeight = Math.max (maxHeight, pt.y);
045        }
046        
047        if (wHint != SWT.DEFAULT) {
048          maxWidth = wHint;
049        }
050        if (hHint != SWT.DEFAULT) {
051          maxHeight = hHint;
052        }
053        
054        return new Point(maxWidth, maxHeight);  
055        
056      }
057    
058      /**
059       * @see org.eclipse.swt.widgets.Layout#layout(org.eclipse.swt.widgets.Composite, boolean)
060       */
061      protected void layout (Composite composite, boolean flushCache) {
062        Rectangle rect = composite.getClientArea();
063      
064        Control[] children = composite.getChildren();
065        for (int i = 0; i < children.length; i++) {
066          children[i].setBounds(rect);
067        }
068      }
069    }
